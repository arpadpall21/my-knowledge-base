(04) Create / Remove / Replace HTML elements practice

CODE : 
    <form name="A_form">
        <input type="number" value="55">
        <input type="text">
    </form>
<script>
// create new element with its necessary attributes!
    var inp = document.createElement('input');
    var att = document.createAttribute("value");
        att.value = "I'm Created with JavaScript!";
        inp.setAttributeNode(att);
            inp.setAttribute('type', 'text');
            inp.setAttribute('class', 'element-input');             // CSS selector created for styling
    // input element ready to insert 
    document.forms.namedItem('A_form').replaceChild(inp, document.forms.namedItem('A_form').elements[0]);           // replacement Executed!
    
// remove the second element!
    document.forms.A_form.removeChild(document.forms['A_form'].elements[1]);                                        // remove the second input
    
// create and insert a Submit button!
    var submit = document.createElement('input');
    var subAttr = document.createAttribute('submit');
        subAttr.value = " Submit (JS created)";
        submit.setAttributeNode(subAttr);
    var subAtt2 = document.createAttribute('type');
        subAtt2.value = "submit";
        submit.setAttributeNode(subAtt2);
    // submit button ready to insert!
    document.forms.namedItem('A_form').appendChild(submit);                                                         // insert the created Submit button!
</script>
