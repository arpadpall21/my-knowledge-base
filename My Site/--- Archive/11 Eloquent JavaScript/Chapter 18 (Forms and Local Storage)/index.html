<script>
    var table = document.getElementById("conway");
    var tableRows = table.querySelectorAll("tr");
    var cells = document.querySelectorAll("#conway tr td");
    var playState;
    var gameState;
    
// start button 
    function runGame() {
        for(let i = 0; cells.length > i; i++) {
            cells[i].removeEventListener("mousedown", addBgColor, true);
        }
        document.getElementById("status").textContent = "Play";
        document.getElementById("status").style.color = "yellow";
        
        getCurrentPattern();
        playState = setInterval(nouveauState, 1000);
    }
    
// stop button 
    function stopGame() {
        for(let i = 0; cells.length > i; i++) {
            cells[i].addEventListener("mousedown", addBgColor, true);
        }
        document.getElementById("status").textContent = "Stand By";
        document.getElementById("status").style.color = "green";
        
        clearInterval(playState);
    }
    stopGame();
    
// add/remove gbColor by the user
    function addBgColor(e) {
        window.getComputedStyle(e.currentTarget).backgroundColor === "rgb(169, 169, 169)" ? e.currentTarget.style.backgroundColor = "rgb(255, 255, 0)" : e.currentTarget.style.backgroundColor = "rgb(169, 169, 169)";
    }
    
// get the current pattern  
    function getCurrentPattern() {
        var collectArray = [];
        for(let i = 0; i < tableRows.length; i++) {
            collectArray.push([]);
            for (let k = 0; k < tableRows[i].children.length; k++) {
                if (window.getComputedStyle(tableRows[i].children[k]).backgroundColor == "rgb(255, 255, 0)") {
                    tableRows[i].children[k].value = true;
                    collectArray[i].push(true);
                } else {
                    tableRows[i].children[k].value = false;
                    collectArray[i].push(false);
                }
            }
        }
        gameState = collectArray;
    }
    
// calculate the new state 
    function nouveauState () {
        var currentState = gameState;
        for(let i = 0; currentState.length > i; i++) {
            for(let k = 0; currentState[i].length > k; k++) {
                let liveNeighbours = 0;
                if (currentState[i - 1] != undefined) {
                    currentState[i - 1][k - 1] ? liveNeighbours++ : null;
                    currentState[i - 1][k] ? liveNeighbours++ : null;
                    currentState[i - 1][k + 1] ? liveNeighbours++ : null;
                }
                if (currentState[i + 1] != undefined) {
                    currentState[i + 1][k - 1] ? liveNeighbours++ : null;
                    currentState[i + 1][k] ? liveNeighbours++ : null;
                    currentState[i + 1][k + 1] ? liveNeighbours++ : null;
                }
                currentState[i][k - 1] ? liveNeighbours++ : null;
                currentState[i][k + 1] ? liveNeighbours++ : null;
                
                if( currentState[i][k] == true && liveNeighbours < 2 || currentState[i][k] == true && liveNeighbours > 3 ) { 
                    table.firstElementChild.children[i].children[k].style.backgroundColor = "rgb(169, 169, 169)";
                } else if (currentState[i][k] === false && liveNeighbours == 3) {
                    table.firstElementChild.children[i].children[k].style.backgroundColor = "rgb(255, 255, 0)";
                } else { }
            }
        }
        getCurrentPattern();
    }
    
// generate random pattern 
    function generateRandomPattern() {
        for(let i = 0; i < tableRows.length; i++) {
            
            for (let k = 0; k < tableRows[i].children.length; k++) {
                if( Math.random() < 0.25) {
                    table.firstElementChild.children[i].children[k].style.backgroundColor = "rgb(255, 255, 0)";
                } else {
                    table.firstElementChild.children[i].children[k].style.backgroundColor = "rgb(169, 169, 169)";
                }
            }
        }
    }
</script>
