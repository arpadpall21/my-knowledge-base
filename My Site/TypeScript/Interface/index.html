Description and Demonstration :

- an interface is basically an object type that describes properties/methods (for the object type)

- unlike type interfaces can be extended (opened) by other interfaces, and can be implemented in Classes 

    interface Interface { 
        readonly propName?: tsType                             // property (readonly = readonly property) (? = optional property)
        methodName?(arg?: tsType): tsType                      // function 
    }
    
    
// extending interfaces -----------------------------------------------
    interface InterfaceA extends InterfaceB { ... }                 // extends interfaceA with InterfaceB (we have to implement InterfaceB propertyes)
    
    
// implementation ---------------------------------------------
    let|const objName: InterfaceA = obj                          // object with InterfaceA type must implement all InterfaceA methods/properties    
        // InterfaceA properties/methods must be implemented ... 
    
    class ClassName implements InterfaceA, ... { ... }               // class must implement all interfaceA, ... method/properties    
        // InterfaceA, ... porperties/methods must be implemented ...
DEMO

type VS interface 

    interface PersonInterface { name:string }
    interface PersonInterface { age:number }    // interfaces 'opened' (extended)
    
    type PersonType = { name:string }
    type PersonType = { age:number }            // -! throws an error (type is not extendable)

Extending interface 

    interface Animal {
        race: string
    }
    
    interface Cat extends Animal {          // extending on creation
        getVoice(): string
    }
    
    const cat : Cat = {
        race: 'cat',
        getVoice: () => 'meoww'
    }
    
// --------------------------------------------------------
    interface Cat {
        race: string 
    }
    interface Cat {                         // 'reopens' (extends) the interface
        getVoice(): string 
    }
    
    const cat : Cat = {
        race: 'cat',
        getVoice: () => 'meoww'
    }

Implementation 

    interface Person {
        name: string
        age: number
        getRace(): string
    }
    
    
// ------------------------------------------------------------------
    const stewen : Person = {               // object must implement all interface properties and methods
        name: 'Steven',
        age: 32,
        getRace: () => 'Hungarian'
    }
    
    
// ------------------------------------------------------------------
    class Stewen implements Person {
        constructor(public name : string,
                    public age: number){
            this.name = name;
            this.age = age;
        }
        
        getRace() { return 'Hungarian' }
    }
