<script>
    var sun = new Image();
    var moon = new Image();
    var earth = new Image();
    function init() {
        sun.src = 'https://mdn.mozillademos.org/files/1456/Canvas_sun.png';
        moon.src = 'https://mdn.mozillademos.org/files/1443/Canvas_moon.png';
        earth.src = 'https://mdn.mozillademos.org/files/1429/Canvas_earth.png';
        window.requestAnimationFrame(draw);
    }

function draw() {
        var ctx = document.getElementById('solarSys').getContext('2d');

        ctx.globalCompositeOperation = 'destination-over';
        ctx.clearRect(0, 0, 300, 300);                          // clear canvas

        ctx.fillStyle = 'rgba(0, 0, 0, 0.4)';
        ctx.strokeStyle = 'rgba(0, 153, 255, 0.4)';
        ctx.save();
        ctx.translate(150, 150);
// Earth
        var time = new Date();
        ctx.rotate(((2 * Math.PI) / 60) * time.getSeconds() + ((2 * Math.PI) / 60000) * time.getMilliseconds());
        ctx.translate(105, 0);                                      // this is genius!! (because the matrix entierly is rotated we move the translate point on the ALREADY rotated matrix)
        ctx.fillRect(0, -12, 40, 24);                           // Shadow
        ctx.drawImage(earth, -12, -12);
// Moon     (the moon is drawn around the earty (matrix is already moved in the center of the eart))
        ctx.save();
        ctx.rotate(((2 * Math.PI) / 6) * time.getSeconds() + ((2 * Math.PI) / 6000) * time.getMilliseconds());
        ctx.translate(0, 28.5);
        ctx.drawImage(moon, -3.5, -3.5);
        ctx.restore();

        ctx.restore();

        ctx.beginPath();
        ctx.arc(150, 150, 105, 0, Math.PI * 2, false);          // Earth orbit
        ctx.stroke();

        ctx.drawImage(sun, 0, 0, 300, 300);

        window.requestAnimationFrame(draw);
    }
init();
</script>

    <canvas id="myClock" style="border:black 7px double; margin:auto; display:block;" width="500" height="500"></canvas>
<script>
    var myClock = function() {
        var ctx = document.getElementById("myClock").getContext("2d");
// these shapes does not need to be redrawn (because they do not ovelap with moving parts)      
        ctx.globalCompositeOperation = "destination-over";
        
    // clock frame   
        ctx.beginPath();
        var radGrad = ctx.createRadialGradient(250, 250, 225, 255, 255, 258);
        radGrad.addColorStop(0, "#1a2f86");
        radGrad.addColorStop(.5, "#535353");
        radGrad.addColorStop(1, "gray");
        ctx.strokeStyle = radGrad;
        ctx.lineWidth = 15;
        ctx.arc(250, 250, 230, 0, Math.PI * 2);
        ctx.stroke();
        
    // background arround the clock's frame
        var img = new Image();                          // I noticed a problem while using "globalCompositeOperation" here, because the "onload" even is asynchronous this event function will use the last value of the "globalCompositeOperation"!
        img.src = "Metal.jpg";
        img.onload = function() {
            ctx.fillStyle = "rgba(0, 0, 0, 0.4)";
            ctx.fillRect(0, 0, 500, 500);
            ctx.drawImage(img, 0, 0, 500, 500);
            ctx.globalCompositeOperation = "source-over";
            drawClockFace();                            // I've forced to call this function here otherwise I cannot change the globalCompositeOperation value without affecting this image (mentioned above why)
        }        
        
    // background clock
        ctx.beginPath();
        ctx.fillStyle = "white";
        ctx.arc(250, 250, 230, 0, Math.PI * 2);
        ctx.fill();
    }
    
    function drawClockFace() {
        var ctx = document.getElementById("myClock").getContext("2d"); 
        var time = new Date();
        ctx.save();
        ctx.translate(250, 250);
        ctx.beginPath();
        ctx.arc(0, 0, 200, 0, Math.PI * 2);
        ctx.fillStyle = "white";
        ctx.fill();
        
    // draw numbers    
        ctx.font = "bold 70px sans-serif";
        ctx.textAlign = "center";
        ctx.textBaseline = "middle";
        ctx.fillStyle = "#2f2088";
        
        var rotation = Math.PI * 2 / 12;
        for(let i = 1; i <= 12; i++) {
            ctx.rotate(Math.PI * 2 / 12);
            ctx.save();
                ctx.fillStyle = "darkgreen";
                ctx.translate(0, -180);
                ctx.rotate(-rotation);
                ctx.fillText(i, 0, 0);
            ctx.restore();
            ctx.save();
                ctx.fillStyle = "darkgreen";
                ctx.font = "30px sans-serif";
                ctx.fillStyle = "darkorange";
                ctx.translate(0, -120);
                ctx.rotate(-rotation);
                var _24hPrint;
                if(i + 12 == 24) {                              // this makes possible to turn 24 to 0
                    _24hPrint = 0;
                } else {
                    _24hPrint = 12 + i;
                }
                ctx.fillText(_24hPrint, 0, 0);
            ctx.restore();
            rotation = rotation + (Math.PI * 2 / 12);           // counter rotation of the face numbers
        }
        
    // draw hour hand
        ctx.save();
        ctx.beginPath();
        ctx.lineWidth = 17;
        if(time.getHours() > 12) {                              // this fancy stuff change the color of the hour hand when it shows the hour above 12 hours
            ctx.strokeStyle = "darkorange";
        } else {
            ctx.strokeStyle = "darkgreen";
        }    
        ctx.lineCap = "round";
        ctx.rotate( ((Math.PI * 2 / 12) * time.getHours()) + (((Math.PI * 2 / 12) / 60) * time.getMinutes()) );   
        ctx.moveTo(0, 0);
        ctx.lineTo(0, -90);
        ctx.stroke();
        ctx.restore();    
            
    // draw minute hand                                         // save the context to prevent the additional rotation to other hands
        ctx.save(); 
        ctx.beginPath();
        ctx.lineWidth = 6;
        ctx.strokeStyle = "black";
        ctx.lineCap = "round";
        ctx.rotate( (Math.PI * 2 / 60) * time.getMinutes() );  
        ctx.moveTo(0, 0);
        ctx.lineTo(0, -150);
        ctx.stroke();
        ctx.restore();
        
    // draw second hand
        ctx.beginPath();
        ctx.lineWidth = 3;
        ctx.strokeStyle = "red";
        ctx.lineCap = "round";
        ctx.rotate( (Math.PI * 2 / 60) * time.getSeconds() );   // this is where the (magic) seconds are drawn
        ctx.moveTo(0, 0);
        ctx.lineTo(0, -185);
        ctx.stroke();
       
    // middle red dot
        ctx.beginPath();
        ctx.shadowColor = "black";
        ctx.shadowBlur = 5;
        ctx.fillStyle = "red";
        ctx.arc(0, 0, 20, 0, Math.PI * 2);
        ctx.fill();
        
        ctx.restore();
        window.setTimeout(drawClockFace, 1000);
    }
    
    myClock();
</script>

Looping Panorama

CODE : 
    <canvas width="800" height="200" id="panorama"></canvas>
<script>
    var img = new Image();

// User Variables - customize these to change the image being scrolled, its
// direction, and the speed.

img.src = 'https://mdn.mozillademos.org/files/4553/Capitan_Meadows,_Yosemite_National_Park.jpg';
var CanvasXSize = 800;
var CanvasYSize = 200;
var speed = 20;                                     // lower is faster
var scale = 1.05;
var y = -4.5;                                       // vertical offset

// Main program
var dx = 0.75;                                      // pixel moved per frame
var imgW;
var imgH;
var x = 0;                                          // canvas VS image width difference (if image is smaller than canvas it stays 0)
var clearX;                                         // dinamic length to be able to clear the canvas
var clearY;
var ctx;

img.onload = function() {
    imgW = img.width * scale;
    imgH = img.height * scale;
    
    if (imgW > CanvasXSize) {                       // image larger than canvas
        x = CanvasXSize - imgW;                     // difference canvas and image width
        clearX = imgW;                  
    } else {                             
        clearX = CanvasXSize;                       // we need to know the dimension before clearing the canvas
    }
    if (imgH > CanvasYSize) {                       // get the referenced crearing size information
        clearY = imgH;
    } else {
        clearY = CanvasYSize;
    }
    
    ctx = document.getElementById('panorama').getContext('2d');
    return setInterval(draw, speed);
}

function draw() {
    ctx.clearRect(0, 0, clearX, clearY);            // clear the canvas with the given dimension
    
    if (imgW <= CanvasXSize) {                      // reset, start from beginning
        if (x > CanvasXSize) {
            x = -imgW + x;
        }
        if (x > 0) {                                // draw additional image1
            ctx.drawImage(img, -imgW + x, y, imgW, imgH);
        }
        if (x - imgW > 0) {                         // draw additional image2
            ctx.drawImage(img, -imgW * 2 + x, y, imgW, imgH);
        }
    }
    
    else {
        if (x > (CanvasXSize)) {
            x = CanvasXSize - imgW;
        }   
        if (x > (CanvasXSize-imgW)) {               // draw aditional image
            ctx.drawImage(img, x - imgW + 1, y, imgW, imgH);
        }
    }
    ctx.drawImage(img, x, y, imgW, imgH);
    x += dx;                                        // amount to move
}
</script>

Ball Physics

- move over the canvas and click to release the ball

CODE : 
    <canvas id="ballPhysics" width="600" height="300" style="border:solid black 1px; background-color:white;"></canvas>
<script>
    (function() {
        var canvas = document.getElementById("ballPhysics");
        var ctx = canvas.getContext("2d");
        var raf;                                                                // control animation
        var running = false;
        
        var ball = {x:100, y:100, vx:4, vy:1, radius:25, color:"blue", 
                        draw: function() {
                            ctx.beginPath();
                            ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                            ctx.closePath();
                            ctx.fillStyle = this.color;
                            ctx.fill(); 
                        }
                   };
    
    function clear() {
        ctx.fillStyle = "rgba(255, 255, 255, .35)";                              // this simple opacity fill creates trailing effect
        ctx.fillRect(0, 0, canvas.width, canvas.height);
    }    
        
    function draw() {
        clear();
        ball.draw();
        ball.x += ball.vx;                                                      // velocity value controls movement
        ball.y += ball.vy;
        ball.vy *= .99;                                                         // by dinamically modifying the horizontal velocity we simulate the gravitaion 
        ball.vy += .25;
        
        if(ball.y + ball.vy > canvas.height || ball.y + ball.vy < 0) {          // these beauties makes sure that the ball never gets out the frame
            ball.vy = -ball.vy;                                                 // velocity value flipped when collision happens
        }
        if(ball.x + ball.vx > canvas.width || ball.x + ball.vx < 0) {
            ball.vx = -ball.vx;
        }
        raf = window.requestAnimationFrame(draw);
    }
        
    canvas.addEventListener("mousemove", function(e) {
        if (!running) {
            clear();
            ball.x = e.offsetX;
            ball.y = e.offsetY;
            ball.draw();
        }
    });
    canvas.addEventListener("click", function(e) {
        if (!running) {
            raf = window.requestAnimationFrame(draw);
            running = true;
        }
    });    
    canvas.addEventListener("mouseout", function(e) {                           // if the mouse leaves the canvas area the animation stops 
        window.cancelAnimationFrame(raf);
        running = false;
    });     
    })();
</script>

Pie Chart

CODE : 
    <canvas id="pieChart" style="border:white 2px dashed; margin:auto; display:block;" width="300" height="300"></canvas>
<script>
    const results = [
        {name: "Satisfied", count: 1043, color: "lightblue"},
        {name: "Neutral", count: 563, color: "lightgreen"},
        {name: "Unsatisfied", count: 510, color: "pink"},
        {name: "No comment", count: 175, color: "silver"}
    ];
    (function(){
        var canvas = document.getElementById('pieChart').getContext("2d");
        var total = results.reduce((sum, {count}) => sum + count, 0);
        var currentAngle = -0.5 * Math.PI;                              // a half PI is 1/4th of the full circle, counted from the start point in a negative value places it at the TOP of the circle  
        for (let result of results) {
            var sliceAngle = (result.count / total) * 2 * Math.PI;      // this returns the radian value for the current pie slice  
            canvas.beginPath();
            canvas.arc(150, 150, 150, currentAngle, currentAngle + sliceAngle);
            currentAngle += sliceAngle;
            canvas.lineTo(150, 150);
            canvas.fillStyle = result.color;
            canvas.fill();
            canvas.stroke();
        }
    })();
</script>
