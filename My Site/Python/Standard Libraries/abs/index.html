Description and Demonstration :

- the abc module allows us to implement abstract base classes described in PEP 3141

- an abstract class is a base class for other subclasses, an abstract method is a method that is declared but has no implementation

- the idea here is having an abstract base class whith abstract methods that all subclasses must implement, this way we can define a scrict base structure for all subclasses (instances) 

    import abc
    
    class AbstractClass(abc.ABC):                  // class extending abc.ABC will become the abstract base class    
        
        @abc.abstractmethod                        // abstract method decorator 
        def abstractMethod():
            // optional business logic...
    
    
    class SubClass(AbstractClass):                 // subclasses deriving from the abstract base class must implement all the abstract base class' methods   
        def abstractMethod():
            // implementation...

    import abc
    
    class AbstractClass(abc.ABC):
        @abc.abstractclassmethod
        def reqMethod1(self):
            pass
    
    inst = AbstractClass()           // -! raises an Error, abstract class cannot be instantiated directly
    
    import abc  
    
    class AbstractClass(abc.ABC):
        @abc.abstractclassmethod
        def reqMethod1(self):
            pass
    
    class TestClass(AbstractClass):   // -! any inheriting subclass must implement the abstract class' abstract method (this raises an Error)
        pass
        
    inst = TestClass()
    
    import abc
    
    class AbstractClass(abc.ABC):
        @abc.abstractclassmethod
        def reqMethod1(self):
            pass
        
    class TestClass(AbstractClass):
        def reqMethod1(self):       // required abstract method implemented by the subclass 
            return 'something'
        
    inst = TestClass()
    print( inst.reqMethod1() )      // -> 'something'
    
    from abc import ABC, abstractclassmethod
    
    class AbcCls(ABC):
        @abstractclassmethod
        def baseMethod(self):
            return 'base value'     // abstract methods can have implementations as well
    
    
    class TestClass(AbcCls):
        def baseMethod(self):
            return super().baseMethod()   // referencing super class method (just like normal)
            
    inst = TestClass()
    print( inst.baseMethod() )      // -> 'base value'
