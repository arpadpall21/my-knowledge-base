Remember This :

- This guide follows the PEP8 style guide

- Consistent style guide is important -> Consisentcy within a project is more important -> Consistency within a module is the most important

General Roules

- always use spaces instead of tabs

- limit all lines up to 79 characters, docstrings and comments up to 72 characters (not recommended but I prefer 120 characters)

- PEP8 doesn't recommend single or double quotes for strings but avoid using escapes in strings (bad -> 'It\'s' good -> "It's"))

- files should be terminated with a single new line character 

// naming conventions --------------------------------------
  SOME_CONSTANT = 'constant value'                          // constants are written with capital letters and words are spearated by _
  
  
  var_name = None                                           // variables and function names are written with snake_case
  def some_func():
      pass
  
  
  import _some_module                                       // module name starting with _ (undersoce) indicates a high level module to a low level module written in C or C++
  
  
  T = TypeVar('val')                                        // types names should be short as possible and written with CamelCase
  SomeType = TypeVar('val2')
  
  
  ClassName:                                                // Class names should be written with CamelCase
      pass
  

  if x is not None:                                         // with None in comparison operations the is is not operator should be used instead of equality operators   
    print('OK')                                             // use is not instead of is ... not
    
  
  if x != None:
      print('OK')
  
  if not x is None:
      print('OK')
  

  def my_func(x):                                           // always use def for functions (lambda is for inline short function solutions)
      return 2 * x
  
  my_func = lambda x : 2 * x
  

  class MyCustomError(Exception):                         // custom Error classes should inherit from the Exception class and suffixed with Error
      pass
  
  class MyCustomError(BaseException):
      pass
  
  try:
      some code                                             // we should be specific what kind of error are we expecting to catch
  except MyCustomError:
      handle_custom_error
  except Exception:
      handle_all_exceptions
  
  try:
      some code 
  except Exception:
      handle_all_exceptions
  

  def larger_than_10(x):                                    // if we have return in a function we should return all cases explicitly
      if x > 10:
          return True
  
      return None
  
  def larger_than_10(x):
      if x > 10:
          return True

Indentations 

  def some_function:                                        // use 4 spaces of indentation 
      some_var = None
  

  def this_is_a_long_function_name(arg_one, arg_two         // aligned with the opening delimiter
                                   arg_three, arg_four
                                   arg_five)
    
  def this_is_a_long_function_name(                         // extra 4 spaces for arguments to distinguish from the body
          arg_one, arg_two,
          arg_three, arg_four,
          arg_five):
    some_var = None
  
  def this_is_a_long_function_name(                         // arguments are not distinguishable from the function body
      arg_one, arg_two,
      arg_three, arg_four):
      some_var = None
  

  foo = this_is_a_long_function_name(                       // hanging indent should add a level
      arg_one, arg_two,
      arg_three, arg_four)
  
  foo = this_is_a_long_function_name(                       // the closing prenthesis (or bracket) can be at the end of the level    
      arg_one, arg_two,
      arg_three, arg_four,
      )
    
  foo = this_is_a_long_function_name(                       // the closing prenthesis (or bracket) can be after the hanging in a new line    
      arg_one, arg_two,
      arg_three, arg_four,
  )
    
  foo = this_is_a_long_function_name(                       // 4 space indentation is optional when using hanging indents 
    arg_one, arg_two,
    arg_three, arg_four)
  
  foo = this_is_a_long_function_name(arg_one, arg_two,      // arguments on the 1st line are forbidden when using hanging indents    
      arg_three, arg_four,
      arg_five)
  

  if (this_is_one_thing                                     // in case of the if multiline statement this is accepted     
      and this_is_another_thing):
      # some comment                                        // a comment might help to distinguish the statement from its code block    
      some_var = None
    
  if (this_is_one_thing                                     // this one is more readable though
          and this_is_another_thing):
      some_var = None
  

  net_income = (job_income                                  // operators should come at the start on the indented line    
                + side_job_income
                - rent
                - other_expenses)
  
  net_income = (job_income +                                // this is less readable 
                side_job_income -
                rent -
                other_expenses)
  

  result = ('this is OK!'                                   // ternary operator can be broken into multiple lines within parentheses
            if condition
            else 'this is wrong!')
  
  result = (
      'this is OK!'
      if condition
      else 'this is wrong!')
  
  result = ('this is OK!'                                   // indentation should start the same level as the first line of the statement
      if condition
      else 'this is wrong!')

Blank Lines 

  import something
  
  
  class Two:                                                // top level classes and function should be surrounded with 2 blank lines
      def __init__(self):
          pass
      
      def some_class_method(self):                          // methods inside a class should be surrounded by 1 blank lines 
          pass
  
  
  def top_level_function():
      pass
  
  
  import something
  
  class Two:                                                // not readable
      def __init__(self):
          pass
      def some_class_method(self):
          pass
  
  def top_level_function():
      pass

White spaces inside expressions and statements 

  if x == y:                                                // avoid writing multiple statements on the same line  
      print('OK')
  print(a); 
  print(b); 
  print(c)
  
  if x == y: 
      do_something()
  else: 
      do_something_else()
  
  if x == y: print('OK')
  print(a); print(b); print(c)
  
  if x == y: do_something()
  else: do_something_else()
  

  some_dict = {'a': 1, 'b': 2}                              // don't use white spaces immediately inside parenthesis or brackets
  some_tuple = (1,)
  
  some_dict = { 'a': 1, 'b': 2 }
  some_tuple = (1, )
  

  some_dict = {'a': 1, 'b': 2}                              // don't use white spaces right before commas, semicolons and colons    
  print(test1); print(test2)
  
  some_dict = {'a' : 1 , 'b' : 2}
  print(test1) ; print(test2)
  

  x = 'a'                                                   // surround operators with a single white space 
  long_var_name = 'b'
  
  res = i + 1
  submitted += 1
  
  a = x*1 + 1                                               // with multiple operators in a statement add white space around the operator with the lowest priority (last operation)    
  c = x*x + y*y
  d = (1+1) * (2+2)
  
  x             = 'a'
  long_var_name = 'b'
  
  res = i+1
  submitted +=1
  
  a = x * 1 + 1
  c = x * x + y * y
  d = (1 + 1) * (2 + 2)
  

  def my_func(a: str) -> str:                               // return type annotation must be surrounded with a single space
      return a
  
  my_func(a=1)                                              // don't use white space arround = when defining default argument value or keyword argument    
  def my_func(a=None):
      pass
  
  my_func(a: str = None):                                   // use single white space around = when combining argument type and default value    
      pass
  
  def my_func(a: str)->str:
      return a
  
  my_func(a = 1)
  def my_func(a = None):
      pass
  
  my_func(a: str=None):
      pass

Tailing Commas 

  my_tuple = (0,)                                           // use parenthesis to create tuple 
  
  my_dict = {                                               // tailing commas are ok when using hanging indent
      'a': 1,
      'b': 2,
  }
  
  my_func(arg_one,                                          // tailing commas are ok when arguments aligned with the opening delimiter    
          arg_two,
          )
  
  my_tuple = 0,
  
  my_dict = {'a': 1, 'b': 2,}                               // tailing commans are not acceped on the same line
  
  my_func(arg_one, arg_two,)
