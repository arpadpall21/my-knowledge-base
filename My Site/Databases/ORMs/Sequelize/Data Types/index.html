Sequelize type MySQL type JS type Description
Boolean
DataTypes.BOOLEAN TINYINT(1) bool - boolean
|
String
DataTypes.STRING(byteLength:nr) VARCHAR(byteLength:nr) str - string (we can optionally set the byte range (Default: 255))
DataTypes.TEXT('tiny|medium|long') TEXT('tiny|medium|long') str - text
|
Number
DataTypes.TINYINT.UNSIGNED TINYINT UNSIGNED nr - tiny integer
DataTypes.SMALLINT.UNSIGNED SMALLINT UNSIGNED nr - small integer
DataTypes.MEDIUMINT.UNSIGNED MEDIUMINT UNSIGNED nr - medium integer
DataTypes.INT.UNSIGNED INTEGER UNSIGNED nr - integer
DataTypes.BIGINT.UNSIGNED BIGINT UNSIGNED nr - big integer (if stored value cannot be represented as nr then str returned)
|
DataTypes.FLOAT.UNSIGNED FLOAT UNSIGNED nr - 32bit float
DataTypes.DOUBLE.UNSIGNED DOUBLE UNSIGNED nr - 64bit float
|
DataTypes.DECIMAL(digits:nr, dec:nr).UNSIGNED DECIMAL(digits:nr, dec:nr) UNSIGNED nr - decimal (Default: DECIMAL(10, 2))
|
Date
DataTypes.DATE DATETIME dateObj - date
DataTypes.DATEONLY DATE 'YYYY-MM-DD' - date (registerd as string)
|
Blob
DataTypes.BLOB('tiny|medium|long') BLOB('tiny|medium|long') buf - binary large object
|

Description and Demonstration :

DEMO (can be pseudocode) 
    import { Sequelize, DataTypes, Model } from 'sequelize';
    
    const sequelize = new Sequelize('mysql://pall:pass@localhost:3306/test_db');
      await sequelize.authenticate();
      console.log('connected!')
    
    class Person extends Model {};
    Person.init(
      {
        str1: DataTypes.STRING,
        str2: DataTypes.STRING(1024),
        
        txt1: DataTypes.TEXT,
        txt3: DataTypes.TEXT('tiny'),
        txt4: DataTypes.TEXT('medium'),
        txt4: DataTypes.TEXT('long'),
        
        tinyInt: DataTypes.TINYINT,
        tinyIntUnsigned: DataTypes.TINYINT.UNSIGNED,      // unsigned 
        smallInt: DataTypes.SMALLINT,
        medInt: DataTypes.MEDIUMINT,
        int: DataTypes.INTEGER,
        bigInt: DataTypes.BIGINT,
        bigInt2: DataTypes.BIGINT,
        
        float: DataTypes.FLOAT,
        double: DataTypes.DOUBLE.UNSIGNED,
        
        dec: DataTypes.DECIMAL(10, 2),
        decUnsigned: DataTypes.DECIMAL(10, 2).UNSIGNED,
        
        datetime: DataTypes.DATE,
        date: DataTypes.DATEONLY,
        
        uuid: { type: DataTypes.UUID, defaultValue: Sequelize.UUIDV4 },
        
        blob: DataTypes.BLOB,
        blob2: DataTypes.BLOB('tiny'),
        blob3: DataTypes.BLOB('medium'),
        blob4: DataTypes.BLOB('long'),
        
        enum: { type: DataTypes.ENUM, values: ['black', 'red'] },
        
        json: DataTypes.JSON,
      },
      { sequelize, modelName: 'Person', timestamps: false }
    );
    
    await Person.sync({ force: true })
    
    
    await Person.create({
      str1: 'str1',
      str2: 'str2',
      
      txt1: 'txt1',
      txt2: 'txt2',
      txt3: 'txt3',
      txt4: 'txt4',
      
      tinyInt: 32,
      tinyIntUnsigned: 255,
      smallInt: 32,
      medInt: 3232,
      int: 94967295,                            // value can be stored because the integer is unsigned
      bigInt: 9223372036854775807n,             // will be returned as string
      bigInt2: 922337,                          // will be returned as number
      
      float: -3245234.32,
      double: 343242423432.324,
      
      dec: -314.3,
      decUnsigned: 324314.3,
      
      datetime: new Date(),
      date: '2024-08-10',
      
      blob: Buffer.from('Hello World!'),
      blob2: Buffer.from('Hello World!'),
      blob3: Buffer.from('Hello World!'),
      blob4: Buffer.from('Hello World!'),
      
      enum: 'red',
      
      json: { p1: 'Hello World', p2: [1, 2, { c1: 'test' }] },
    })
