Description and Demonstration :

- SQLAlchemy maps data types to databases in 2 ways :

- uses CameCase classes (above table) (depending on the database used types may be different)

- uses UPPERCASE classes that represent direct SQL data types 

  from sqlalchemy import camelCaseTypes                      // import camel case types (String, Integer, etc.)
  
  from sqlalchemy.dialects.mysql import upperCaseTypes       // import upper case types (VARCHAR, INTEGER, etc.)

CamelCase data types 

    import datetime
    import enum
    from decimal import Decimal
    from sqlalchemy import (URL, create_engine, Column,
        BigInteger,
        Boolean,
        Date,
        DateTime,
        Double,
        Enum,
        Float,
        Integer,
        Interval,
        LargeBinary,
        Numeric,
        PickleType,
        SmallInteger,
        String,
        Text,
        Time,
        Unicode,
        UnicodeText,
        Uuid,
    )
    from sqlalchemy.ext.declarative import declarative_base
    from sqlalchemy.orm import sessionmaker
    
    Base = declarative_base()
    
    
    class MyEnum(enum.Enum):
        one = 'one'
        two = 'two'
        three = 'three'
    
    
    class TestTable(Base):
        __tablename__ = 'TestTable'
    
        a = Column('BigInteger', BigInteger, primary_key=True)
        b = Column('Boolean', Boolean, default=False)
        c = Column('Date', Date)
        e = Column('DateTime', DateTime)
        f = Column('Double', Double)
        g = Column('Enum', Enum(MyEnum))
        h = Column('Float', Float)
        i = Column('Integer', Integer)
        j = Column('Interval', Interval)
        k = Column('LargeBinary', LargeBinary)
        l = Column('Numeric', Numeric(20, 10))
        m = Column('PickleType', PickleType)
        n = Column('SmallInteger', SmallInteger)
        o = Column('String', String(20))
        p = Column('Text', Text)
        q = Column('Time', Time)
        r = Column('Unicode', Unicode(20))
        s = Column('UnicodeText', UnicodeText)
    
        def __init__(self, a, b, c, e, f, g, h, i, j, k, l, m, n, o, p, q, r, s):
            self.a = a
            self.b = b
            self.c = c
            self.e = e
            self.f = f
            self.g = g
            self.h = h
            self.i = i
            self.j = j
            self.k = k
            self.l = l
            self.m = m
            self.n = n
            self.o = o
            self.p = p
            self.q = q
            self.r = r
            self.s = s
    
    
    url_obj = URL.create(
        'mysql+mysqlconnector',
        username='test_user',
        password='test_password',
        host='localhost',
        port=3306,
        database='sqlalchemy_test'
    )
    engine = create_engine(url_obj, echo=True)
    connection = engine.connect()
    Base.metadata.create_all(bind=engine)
    
    Session = sessionmaker(bind=engine)
    session = Session()
    
    
    session.add(TestTable(
        None,                                          # BigInteger    
        True,                                          # Boolean       
        datetime.date(2023, 5, 26),                    # Date       
        datetime.datetime(2023, 5, 26, 23, 59, 4),     # DateTime       
        23.33,                                         # Double       
        MyEnum.two,                                    # Enum       
        333.3333,                                      # Float       
        884,                                           # Integer       
        datetime.timedelta(1, 2, 4),                   # Interval       
        bytearray(b'lkkjlkjsdflkjdlfjslkfj'),          # LargeBinary   
        Decimal('9999999999.9999999999'),              # Numeric       
        {'a': 1, 'b': [1, False, (3, 2)]},             # PickleType      
        2.2,                                           # SmallInteger       
        'some string ❤',                               # String       
        'another string ❤',                            # Text       
        datetime.time(23, 59, 4),                      # Time     
        'unicode string ❤',                            # Unicode       
        'unicode string ❤'                             # UnicodeText    
    ))
    
    session.commit()

UPPERCASE data types 

    import datetime
    
    from sqlalchemy import URL, create_engine, Column
    from sqlalchemy.dialects.mysql import INTEGER, TINYTEXT, VARCHAR, TIMESTAMP     # MySQL types
    from sqlalchemy.ext.declarative import declarative_base
    from sqlalchemy.orm import sessionmaker
    
    Base = declarative_base()
    
    
    class TestTable2(Base):
        __tablename__ = 'TestTable2'
    
        _int = Column('INT', INTEGER, primary_key=True)
        varchar = Column('VARCHAR', VARCHAR(34))
        tinytext = Column('TINYTEXT', TINYTEXT)
        timestamp = Column('TIMESTAMP', TIMESTAMP)
    
        def __init__(self, _int, varchar, tinytext, timestamp):
            self._int = _int
            self.varchar = varchar
            self.tinytext = tinytext
            self.timestamp = timestamp
    
    
    url_obj = URL.create(
        'mysql+mysqlconnector',
        username='test_user',
        password='test_password',
        host='localhost',
        port=3306,
        database='sqlalchemy_test'
    )
    engine = create_engine(url_obj, echo=True)
    connection = engine.connect()
    Base.metadata.create_all(bind=engine)
    
    Session = sessionmaker(bind=engine)
    session = Session()
    
    session.add(TestTable2(
        None,
        'Hello World!',
        'Hello World!',
        datetime.datetime.now()
    ))
    
    session.commit()
