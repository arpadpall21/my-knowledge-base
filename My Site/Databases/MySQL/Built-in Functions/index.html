Aggregation Grouping
SELECT aggrFn [AS colAlias], col FROM table, ... GROUP BY col, ... [HAVING cond] [ORDER BY col, ... ASC|DESC] [LIMIT nr] - groups rows that have the same values into summary rows
- the (aggregate) function returns a value per grouped records (col)
- (AS) renames the specified column name in the returned result (gives an alias to the column|table)
- (HAVING) filters the returned table (runs after the SELECT statement is executed)
- (ORDER BY) sorts alphabetically the returned records based on column(s) (ASC = ascending (Default) | DESC = descending)
- (LIMIT nr) limits how many records (rows) to returns

// ASCII() / SPACE() ------------------------------------------------------------------
    mysql> SELECT ASCII('x');                                                          // -> 120                    // ASCII value of 'x'
    mysql> SELECT ASCII('X');                                                          // -> 88                     // ASCII value of 'X'
    
    mysql> SELECT SPACE(4);                                                            // -> '    '                 // returns a string with 4 spaces    
    
// STRCMP() ---------------------------------------------------------------------------
    mysql> SELECT STRCMP('cherry', 'strawberry');                                      // -> -1                     // alphabetically 'cherry' is before 'strawberry'    
    mysql> SELECT STRCMP('alpha', 'alba');                                             // -> 1                      // alphabetically 'alpha' is after 'alba'     
    mysql> SELECT STRCMP('prune', 'prune');                                            // -> 0                      // alphabetically both arguments are the same    
    
// CHAR_LENGHT() / CHARACTER_LENGTH() / LENGTH() --------------------------------------
    mysql> SELECT CHAR_LENGTH('Árpád');                                                // -> 5                      // string contains 5 characters 
    mysql> SELECT CHARACTER_LENGTH('Árpád');                                           // -> 5                        // does the same as above  
    mysql> SELECT LENGTH('Árpád');                                                     // -> 7                      // string contains 7 bytes  
    
// CONCAT() / CONCAT_WS() -------------------------------------------------------------
    mysql> SELECT CONCAT('Kicsi', 'Kutya', 'Tarka');                                   // -> 'KicsiKutyaTarka'      // concatenates strings together   
    mysql> SELECT CONCAT_WS(' ', 'Kicsi', 'Kutya', 'Tarka');                           // -> 'Kicsi Kutya Tarka'    // concatenates strings together by inserting a character between each concatenated string    
    
// FIELD() / FIND_IN_SET() / INSTR() / POSITION() / LOCATE() --------------------------
    mysql> SELECT FIELD('ot', 'harom', 'negy', 'ot', 'hat', 'ot');                     // -> 3                      // first occurence of 'ot' in a string list (from 2nd argument (included)) (1 indexed)    
    mysql> SELECT FIND_IN_SET('ot', 'harom,negy,ot,hat,ot');                           // -> 3                      // first occurence of 'ot' in an SQL Set (1 indexed)    
    
    mysql> SELECT INSTR('Kicsi Kutya Tarka', 'a');                                     // -> 11                     // first occurence of 'a' in 'Kicsi Kutya Tarka' (1 indexed)    
    mysql> SELECT POSITION('a' IN 'Kicsi Kutya Tarka');                                // -> 11                     // first occurence of 'a' in 'Kicsi Kutya Tarka' (1 indexed)    
    
    mysql> SELECT LOCATE('a', 'Kicsi Kutya Tarka');                                    // -> 11                     // first occurence of 'a' in 'Kicsi Kutya Tarka' (1 indexed)    
    mysql> SELECT LOCATE('a', 'Kicsi Kutya Tarka', 12);                                // -> 14                     // first occurence of 'a' in 'Kicsi Kutya Tarka' search starts from 12th index (included) (1 indexed)    
    
// FORMAT() / LPAD() / RPAD() ---------------------------------------------------------
    mysql> SELECT FORMAT(5648100.24587, 2);                                            // -> '5,648,100.25'         // returns a formated string (hundreds are separated by commas and contains 2 decimal digits)    
    
    mysql> SELECT LPAD('Yankee', 10, '.');                                             // -> '....Yankee'           // missing left characters are filled with '.' to 10 characters    
    mysql> SELECT RPAD('Yankee', 10, '.');                                             // -> 'Yankee....'           // missing right characters are filled with '.' to 10 characters    
    
    mysql> SELECT LPAD('Yankee', 3, '.');                                              // -> 'Yan'                  // string choped if the length is shorter    
    
// TRIM() / LTRIM() / RTRIM() ---------------------------------------------------------
    mysql> SELECT TRIM('   Muzli   ');                                                 // -> 'Muzli'                // removes leading and trailing spaces   
    mysql> SELECT LTRIM('   Muzli   ');                                                // -> 'Muzli   '             // removes leading spaces    
    mysql> SELECT RTRIM('   Muzli   ');                                                // -> '   Muzli'             // removes trailing spaces    
    
// INSERT() / REPLACE() / REPEAT() / REVERSE() ----------------------------------------
    mysql> SELECT INSERT('Kicsi Kutya Tarka', 7, 5, 'Cica');                           // -> 'Kicsi Cica Tarka'     // replaces 'Kutya' with 'Cica' in the string    
    mysql> SELECT REPLACE('Kaka', 'a', '*');                                           // -> 'K*k*'                 // replaces every 'a' with '*'    
    
    mysql> SELECT REPEAT('.', 5);                                                      // -> '.....'                // repeats 5 times the passed string    
    
    mysql> SELECT REVERSE('abcde');                                                    // -> 'edcba'                // reverses the string   
    
// LCASE() / LOWER() / UCASE() / UPPER() ----------------------------------------------

    mysql> SELECT COUNT(*) FROM someTable;                                             // -> 11                     // counts all table records

    mysql> SELECT COUNT(DISTINCT col1) FROM someTable;                                 // -> 4                      // counts 'col1' columns distinct (one of each) values (excluding NULL)

    mysql> SELECT AVG(col1) FROM someTable;                                            // -> 3                      // average of 'col1' column values (excluding NULL)   
    mysql> SELECT AVG(DISTINCT col1) FROM someTable;                                   // -> 2.5                    // average of 'col1' column distinct (one of each) values (excluding NULL)

    mysql> SELECT SUM(col1) FROM someTable;                                            // -> 30                     // sum of 'col1' column values (excluding NULL)    
    mysql> SELECT SUM(DISTINCT col1) FROM someTable;                                   // -> 10                     // sum of 'col1' column distinct (one of each) values (excluding NULL)

    mysql> SELECT MIN(col1) FROM someTable;                                            // -> 1                      // minimal value of 'col1' column    
    mysql> SELECT MAX(col2) FROM someTable;                                            // -> 'A'                    // alphabetically the closest value of 'col2'   
    
    mysql> SELECT MAX(col1) FROM someTable;                                            // -> 4                      // maximal value of 'col1' column
