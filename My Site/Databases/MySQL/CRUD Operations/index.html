// SELECT -----------------------------------------------------------------------------
    mysql> SELECT * FROM table_1;                                                      // returns all records 
    
    mysql> SELECT col1, col2 FROM table_1;                                             // returns col1 and col2 records 
    mysql> SELECT table_1.col1, table_1.col2 FROM table_1;                               // table.column syntax can be used 
    
  // aliases -------------------------------------------------
    mysql> SELECT col1 AS c1, col2 AS c2 FROM table_1;                                 // in the returned table 'col1' column is named 'c1' and 'col2' column is named as 'c2'
    
    mysql> SELECT table_1.col1, table_2.col1 FROM table_1, table_2                     // columns from multiple tables 
    mysql> SELECT A.col1, B.col1 FROM table_1 AS A, table_2 AS B                       // same as above with table aliases 
    
  // WHERE condition -----------------------------------------
    mysql> SELECT * FROM table_1 WHERE col2 IS NULL;                                   // return all records where 'col2' has NULL values  
  
  // HAVING condition ----------------------------------------
    mysql> SELECT * FROM table_1 HAVING col2 IS NULL;                                  // return all records where 'col2' has NULL values   
                                                                                         // -! visually does the same as above but the HAVING clause runs after the SELECT statement   
    mysql> SELECT * FROM table_1 WHERE col1 > 10 HAVING col2 IS NOT NULL;              // there are 2 filters in this statement, the 'WHERE' filters the SELECT statement and the 'HAVING' filters the returned result table    
                                                                                         // returns all records where 'col1' have higher value than 10 then filters again 'col2' cannot have NULL values    
    
  // order ---------------------------------------------------
    mysql> SELECT * FROM table_1 ORDER BY col1;                                        // the returned talbe's records are ordered (ascending order) according to 'col1' column   
    mysql> SELECT * FROM table_1 ORDER BY col1, col2;                                    // ordered first by 'col1' then by 'col2' 
    
  // limit ---------------------------------------------------
    mysql> SELECT * FROM table_1 LIMIT 4;                                              // the returned table has only 4 records  
    
    
    
// TABLE ------------------------------------------------------------------------------
    mysql> TABLE table_1;                                                              // returns all records of 'table_1'   
    mysql> TABLE table_1 ORDER BY col1 LIMIT 10;                                       // 10 records are returned and ordered alphabetically by 'col1' column   
    
    
    
// SELECT DISTINCT --------------------------------------------------------------------
    mysql> SELECT DISTINCT * FROM table_1;                                             // one of each value is returned if there are multiple same values     
    
    mysql> SLECT DISTINCT col1 FROM table_1;                                           // only one instance of each of the same records are returned in the 'col1' column    
    
    
    
// UNION ------------------------------------------------------------------------------
    mysql> SELECT col1 FROM table_1 UNION SELECT col1 FROM table_2;                    // returns a single column from 'col1' and 'col2' (colums must have the same data types)    
    
    mysql> SELECT col1, col2, col3 FROM table_1                                        // unifies the two select statement into one talbe (the returned table has 3 columns with both tables' records)    
         > UNION                                                                         // 'col1' 'col2' 'col3' must have the same datatypes in both tables    
         > SELECT col1, col2, col3 FROM table_2;                                         // -! the united columns contain only distinct (one of each) values (SELECT DISTINCT rule)    
    
    mysql> SELECT col1, col2, col3 FROM table_1                                        // does the same as above but the united columns contain all data 
         > UNION ALL 
         > SELECT col1, col2, col3 FROM table_2;  
    
    
    
// INSERT INTO ------------------------------------------------------------------------
    mysql> INSERT INTO table_1 VALUES(1, 2, 3, 4);                                     // inserts a value in each columns 
    
    mysql> INSERT INTO table_1(col1, col3) VALUES(21, 45);                             // inserts values in 'col1' and 'col2' columns only 
                                                                                         // other columns will have their default values inserted (NULL in this case)    
        
    mysql> INSERT INTO table_1 SELECT * FROM table_2;                                  // inserts all 'table_2' records to 'table_1' (the 2 table structure must be identical)    
    
    
    
// REPLACE INTO -----------------------------------------------------------------------
    mysql> REPLACE INTO table_1 VALUES(1, 'egy', 'ketto');                             // replaces the primary key 1 record to these values (if the 1 primary key exist, otherwise inserts this record)    
    
    mysql> REPLACE INTO table_1(col1, col2) SELECT col1, col2 FROM table_2;            // inserts records having only 'col1' and 'col2' fields to 'table_1' from 'table_2.col1' and 'table_2.col2'    
                                                                                       // -! replaces primary key or unique index records if there are duplicates in 'table_1'    
    
    
// UPDATE -----------------------------------------------------------------------------
    mysql> UPDATE table_1 SET col4=NULL;                                               // -! sets all values in 'col4' to NULL   
    
    mysql> UPDATE table_1 SET col4=NULL WHERE col4=666;                                // sets the 'col4' value to NULL where 'col4' has a 666 value    
    
    
    
// DELETE FROM ------------------------------------------------------------------------
    mysql> DELETE FROM table_1;                                                        // !! this deletes all the table's content (same as TRUNCATE TABLE)   
    
    mysql> DELETE FROM table_1 WHERE col4 IS NULL;                                     // deletes records where the 'col4' column has NULL values    
    
    
    
// VALUES -----------------------------------------------------------------------------
    mysql> VALUES                                                                      // creates a table with 2 records on the fly 
        ->   ROW(1, 2, 2 + 1),                                                         // values can be returned by an expression   
        ->   ROW('egy', 'ketto', 'harom');                                             // columns can have mixed data types   
    
    
    
// Nested Queries ---------------------------------------------------------------------
    SELECT * FROM (SELECT col1, col2 FROM table_1) AS A;                               // the nested query basically generates a table on which the main query is perforemed (table alias is required)    
    
    SELECT * FROM table_1 WHERE table1.col2=(SELECT col1 FROM table_2)                 // nested query must return a table with one column containing only one value on which the WHERE condition is based     
    
    SELECT * FROM table_1 WHERE table1.col2 IN (SELECT col1 FROM table_2)              // nested query can return a table with one column but multiple values because the IN operator

Wildcards 

    mysql> SELECT * FROM myTable WHERE col_1 = 'a%';                                   // -! the 'a%' is not treated as wildcard because the equal sign is used instead of the LIKE keyword   
    
    mysql> SELECT * FROM myTable WHERE col_1 LIKE 'a%';                                // matches any values in the 'col_1' field which starts with 'a'   
    
    mysql> SELECT * FROM myTable WHERE col_1 LIKE 'a_';                                // matches any values in the 'col_1' field which starts with 'a' and followed by one character    
    
    mysql> SELECT * FROM myTable WHERE col_2 LIKE '%00';                               // matches any values in the 'col_2' field which ends with '00'           // -! works even if the 'col_2' field contains number data types    
    
    mysql> SELECT * FROM myTable WHERE col_3 LIKE '2020-01-%';                         // matches any values in the 'col_2' field which starts with '2020-01-'   // -! works even if the 'col_3' field contains date data types    
    
    mysql> SELECT * FROM myTable WHERE col_1 LIKE '\%%'                                // matches any values in the 'col_1' field which starts with '%'  
    
    mysql> SELECT * FROM myTable WHERE col_1 LIKE '%\_'                                // matches any values in the 'col_1' field which ends with '_'    
    
// NOT LIKE ---------------------------------------------------------------------------
    mysql> SELECT * FROM myTable WHERE col_1 NOT LIKE 'a%'                             // returns all values in 'col_1' which does not start with 'a'  
    
    mysql> SELECT * FROM myTable WHERE col_1 NOT LIKE '%x%'                            // returns all values in 'col_1' which does not contain any 'x' characters

Transactions

- a transaction ensures that a set operations are fully executed or not executed at all (no partial exection) 

    START TRANSACTION;              // starts a transaction
    
    sqlOperation   // -! if any operation fails all operations are rolled back
    ...
    
    ROLLBACK;                       // reverts all operation up to the lastest COMMIT or to START TRANSACTION
    COMMIT;                         // ends the transaction
DEMO 
    START TRANSACTION;
    
    INSERT INTO employee(name, surname, age) values('Doe', 'John', 22);
    INSERT INTO employee(name, surname, age) values('Bob', 'Chucky', 32);
    
    ROLLBACK;
    
    INSERT INTO employee(name, surname, age) values('Alice', 'Deejay', 42);         // only these 3 records are persisted in db
    INSERT INTO employee(name, surname, age) values('Jonathan', 'Livra', 52);
    INSERT INTO employee(name, surname, age) values('Arnold', 'Schwarzii', 62);
    
    COMMIT;
