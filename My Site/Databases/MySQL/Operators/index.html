Comparison Operators

- there's no data type checking 

    val = val                                                                  // returns true if both values are equal (returns NULL if one or both operands are NULL)  
    val <=> val                                                                // returns true if both values are equal (returns false if one operand is NULL | returns true if both operands are NULL) (NULL safe comparison)    
    val LIKE val|pattern                                                       // returns true if both values are equal (returns NULL if one or both operands are NULL)  
                                                                                  // does the same as (=) but also accept patterns on its right side (wildcards can be used in patterns)    
    
    val <>|!= val                                                              // returns true if values are different (returns NULL if one or both operands are NULL)    
    val NOT LIKE val|pattern                                                   // returns true if values are different (returns NULL if one or both operands are NULL)    
                                                                                  // does the same as (!= OR <>) but also accept patterns on its right side (wildcards can be used in patterns)    
    
    
    nr > nr                                                                    // returns true if the left operand is bigger than the right one (returns NULL if one or both operands are NULL)    
    nr >= nr                                                                   // returns true if the left operand is bigger or equal than the right one (returns NULL if one or both operands are NULL)    
    
    
    nr < nr                                                                    // returns true if the left operand is smaller than the right one (returns NULL if one or both operands are NULL)    
    nr <= nr                                                                   // returns true if the left operand is smaller or equal than the right one (returns NULL if one or both operands are NULL)
DEMO 
                                                                                       // results are returned in a table record  
// = / <=> / LIKE (equal) -------------------------------------------------------------
    mysql> SELECT 2 = '2';                                                             // -> true
    mysql> SELECT 2 = NULL;                                                            // -> NULL
    mysql> SELECT NULL = NULL;                                                         // -> NULL
    
    mysql> SELECT 2 <=> '2';                                                           // -> true       // NULL safe equal comparison operator
    mysql> SELECT 2 <=> NULL;                                                          // -> false
    mysql> SELECT NULL <=> NULL;                                                       // -> true
    
    mysql> SELECT 2 LIKE '2';                                                          // -> true       // does the same as (=)
    mysql> SELECT NULL LIKE '2';                                                       // -> NULL
    mysql> SELECT 221 LIKE '2%';                                                       // -> true       // can compare pattern to values 
    mysql> SELECT 'Kicsi Kutya Tarka' LIKE 'K%a';                                      // -> true  
    
// != / <> / NOT LIKE (not equal) -----------------------------------------------------
    mysql> SELECT 'egy' != 'ketto';                                                    // -> true
    mysql> SELECT 'egy' != 'egy';                                                      // -> false
    
    mysql> SELECT 'egy' <> 'ketto';                                                    // -> true
    mysql> SELECT 'egy' <> 'egy';                                                      // -> false
    
    mysql> SELECT 'egy' NOT LIKE 'ketto';                                              // -> true       // does the same as (!= or <>)
    mysql> SELECT 'egy' NOT LIKE 'e%';                                                 // -> false      // can compare pattern to values 
    mysql> SELECT 'egy' NOT LIKE 'k%';                                                 // -> true   
    
// > / >= (greater (or equal) than) ---------------------------------------------------
    mysql> SELECT 10.1 > 9.99;                                                         // -> true
    mysql> SELECT 10.1 > 19.99;                                                        // -> false
    
    mysql> SELECT 10.1 >= 9.99;                                                        // -> true
    mysql> SELECT 10.1 >= 10.1;                                                        // -> true
    
// < / <= (smaller (or equal) than) ---------------------------------------------------
    mysql> SELECT 9.99 < 10.1;                                                         // -> true
    mysql> SELECT 19.99 < 10.1;                                                        // -> false
    
    mysql> SELECT 9.99 <= 10.1;                                                        // -> true
    mysql> SELECT 10.1 <= 10.1;                                                        // -> true

Logical Operators 

    cond AND|&& cond                                                           // returns true if both conditions are true    
    cond OR||| cond                                                            // returns true if one or both conditions are true    
    NOT|! cond                                                                 // returns true if the condition is false     
    exp XOR exp                                                                // returns NULL if either operand is NULL (for none NULL operands it might be true or false)   
    
    col|exp IN (exp, ...)                                                      // returns true if the column or expression value is present in the listed values (shorthand for multiple OR conditions)    
    col|exp IN (selectStatement)                                               // returns true if the column or expression value is present in the listed values what are returned by the nested SELECT statement

        
 - must return a table with one column containing one or multiple values 

    
              
    col|exp NOT IN (exp, ...)                                                  // returns true if the column or expression value is not present in the listed values (shorthand for multiple OR conditions)    
    col|exp NOT IN (selectStatement)                                           // returns true if the column or expression value is not present in the listed values what are returned by the nested SELECT statement

        
 - must return a table with one column containing one or multiple values 

    
  
    
    col|exp BETWEEN val_1 AND val_2                                            // returns true if the column or expression value is between val_1 (included) and val_2 (included) (works for numbers, strings and dates)

        
 - for numbers, between the given number range 

        
 - for strings, between the alphabetical range 

        
 - for dates, between the given time range 

    
    
    
    EXISTS (selectStatement)                                                   // returns true if the nested query returns a table with at least one record

        
 - the EXISTS clause immediately stop testing and returns true once the first value is found in the nested query (for performance reasons) 
    

    NOT EXISTS (selectStatement)                                               // returns true if the nested query returns an empty table

                        
 - the NOT EXISTS clause immediately stop testing and returns false once the first value is found in the nested query (for performance reasons) 
    

    
    col|exp compOp

        
 - any comparison operator (= <> != > >= < or <=) 
 
    
 ANY|SOME (selectStatement)                                 // returns true if one or more of the nested SELECT query values

        
 - must return a table with one column containing one or multiple values 

    
 are compared true against the expression or current column value (col)    
    col|exp compOp

        
 - any comparison operator (= <> != > >= < or <=) 
 
    
 ALL (selectStatement)                                      // returns true if all of the nested SELECT query values

        
 - must return a table with one column containing one or multiple values 

    
 are compared true against the expression or current column value (col)

// AND | OR | NOT | XOR ---------------------------------------------------------------// results are returned in a table record 
    mysql> SELECT 1 AND 1;                                                             // -> true
    mysql> SELECT 1 AND 0;                                                             // -> false 
    
    mysql> SELECT 1 OR 1;                                                              // -> true
    mysql> SELECT 1 OR 0;                                                              // -> true 
    mysql> SELECT 0 OR 0;                                                              // -> false 
    
    mysql> SELECT NOT 1;                                                               // -> false
    mysql> SELECT NOT 0;                                                               // -> true 
    
    mysql> SELECT NULL XOR 'something';                                                // -> NULL  
    mysql> SELECT NULL XOR NULL;                                                       // -> NULL 
    mysql> SELECT 'something' XOR 'something';                                         // -> false      // -! might be true of false if none NULL operand found   
    
// IN() -------------------------------------------------------------------------------
    mysql> SELECT 1 + 1 IN(1, 2, 3, 4, 3 + 5);                                         // -> true       // expression result is present in the list   
    mysql> SELECT * FROM table_1 WHERE col1 IN(3, 4, 5);                               // returns records where column 'col1' contains 3, 4, 5 values    
    
    mysql> SELECT 1 + 1 IN(SELECT col1 FROM table_1);                                  // -> true       // the table_1.col1 contains 2  
    mysql> SELECT * FROM table_1 WHERE col1 IN(SELECT col1 FROM table_2);              // returns records where 'table_1.col1' and 'table_2.col1' have the same values      
    
// BETWEEN ----------------------------------------------------------------------------
    mysql> SELECT 5.21 BETWEEN 5 AND 10;                                               // -> true
    mysql> SELECT 4.21 BETWEEN 5 AND 10;                                               // -> false
    mysql> SELECT 5 BETWEEN 5 AND 10;                                                  // -> true  
    
// EXISTS / NOT EXISTS ----------------------------------------------------------------
    mysql> SELECT EXISTS(SELECT * FROM table_1);                                       // -> true       // table_1 has at least one record  
    mysql> SELECT EXISTS(SELECT * FROM table_2);                                       // -> false      // table_2 has no records    
    
    mysql> SELECT NOT EXISTS(SELECT * FROM table_1);                                   // -> false       // table_1 has at least one record  
    mysql> SELECT NOT EXISTS(SELECT * FROM table_2);                                   // -> true        // table_2 has no records    
    
// ANY / SOME / ALL -------------------------------------------------------------------
    mysql> SELECT 21 = SOME (SELECT col1 FROM Numbers);                                // -> true        // 'Numbers.col1' contains at least one 21 
    mysql> SELECT 21 = ANY (SELECT col1 FROM Numbers);                                 // -> true        // does the same as above   
    
    mysql> SELECT 10 < ALL(SELECT col1 FROM Numbers);                                  // -> true        // all 'Numbers.col1' records are bigger than 10
