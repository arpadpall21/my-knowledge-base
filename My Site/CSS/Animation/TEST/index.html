Fancy Animation with JavaScript setInterval() / clearInterval() technique

Animation technique with JS Test

CODE : 
    -- CODE (HTML) --
    
    <button onclick="myMove()">Click Me</button> 
    <div id ="myContainer">
        <div id ="myAnimation"></div>
    </div>
<script>
    function myMove() {
        var elem = document.getElementById("myAnimation"); 
        var pos = 0;
        var animSystem = setInterval(frame, 10);                // "frame" function fires every "10" milliseconds  
        function frame() {
            if (pos == 170) {
                clearInterval(animSystem);                      // this stops the animation
            } else {
                pos++; 
                elem.style.top = pos + 'px';                    // variable combined with a string
                elem.style.left = pos + 'px'; 
            }
        }
    }
</script>

    -- CODE (CSS) --
    
        #myContainer {                          #myAnimation {
        width: 200px;                               width: 30px;
        height: 200px;                              height: 30px;
        position: relative;                         position: absolute;
        background: yellow;                         background: red;
    }                                           }

solving the restart animation problem when the animation is cotroled by javaScript

- CSS can run an animation only once! so the solution make an infinite replacement script which replaces the old element that has ran the animation already by a new one

CODE : 
    -- CODE (HTML) --
    <button onclick="animSt()">start animation</button>
    <div id="animProblem"></div>
<script>
    var clone;
    
    var animSt = function() {
        var elm = document.getElementById('animProblem');
        clone = elm.cloneNode();
        
        elm.parentNode.replaceChild(clone, elm);
        clone.style.animation = 'animProb 2s ease-out forwards';        
    }
</script>
    
    -- CODE (CSS) --
    #animProblem {
        width: 50px;
        height: 50px;
        background-color: mediumspringgreen;
        border: solid 2px black;
    }
    @keyframes animProb {
        100% { 
            width: 250px;
            height: 250px; 
            transform: rotateZ(1440deg)
        }
    }
