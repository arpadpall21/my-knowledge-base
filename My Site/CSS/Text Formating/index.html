The vertical-align / line-height properties

DEMO 
    .myElement {
        line-height: 1.5;                   // line height is 1 and a half size compared to the set font-size
        line-height: 150%;                      // same effect expressed in percent
        line-heihgt: 21px;                  // line height set to a fixed value
    }

// vertical-align
    .myElement .child {
        vertical-align: 50%;                // element is 1/2 rised compared to the element line-height
        vertical-align: -25px;              // element lowered 25px compared to the base line
        vertical-align: text-top;           // element aligns to top with the tallest element in the line
    }
    .table td {
        vertical-align: bottom;             // table cell content aligned to the bottom
    }

The white-space / tab-size / quotes properties

DEMO 
    .myElement {
        white-space: normal;                // carriage return ignored           /  extra white space collapsed  /  text wraps when it hits the end of the element 
        white-space: nowrap;                // carriage return ignored           /  extra white space collapsed  /  text does not wrap when it hits the end of the element 
        white-space: pre;                   // carriage return creates new line  /  extra white space preserved  /  text does not wrap when it hits the end of the element
        white-space: pre-wrap;              // carriage return creates new line  /  extra white space preserved  /  text wraps when it hits the end of the element
        white-space: pre-line;              // carriage return creates new line  /  extra white space collapsed  /  text wraps when it hits the end of the element 
    }
    
// quotes ----------------------------------------
// -- HTML -- 
    <p class="myElement"> <q>text <q>nested</q> text</q>            // displays "<text ⇉nested⇇ text>"
    
// -- CSS -- 
    .myElement {
        quotes: "<" ">" "\21C9" "\21C7";                            // the "<" ">" are the first level quotation values and the "\21C9" "\21C7" (unicode characters) are the 2nd level quotation values (we could specify more)  
    }
    
// tab-size -------------------------------------- 
    pre {
        tab-size: 16;                                               // tab size set to 16 characters width
        -moztab-size: 16;
        -o-tab-size: 16;
    }
