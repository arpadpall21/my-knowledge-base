// for ...in --------------------------------------------------------------------------
    for (let i in myObject){                                                           // iterating over an object's properties 
        i;                                                                             // -> Name | surename | age      // properties are returned
    }

// for ...of --------------------------------------------------------------------------
    for (let i of myArray) {                                                           // iterating over an Array's items

    for(let i of 'lo 🐴'){                                                            // this loop iterates over any objects or values that has the [Symbol.iterator] implemented (like strings)

The while loop 

// while ----------------------------
    while(cond) {         // if cond evaluates true runs 1 iteration, if evaluates false stops the loop  
        // some code...
    }
            
            
// do ...while ----------------------------
    do {                                   // runs 1 interation once garanteed, then if cond evaluates true runs 1 iteration, if evaluates false stops the loop  
        // some code...
    } while(cond)

The break / continue keywords and labeled statements

- the break and continue statements are mostly used in loops but in JavaScript we can break / continue any code block { }

- the labeled statement can be used with any loop, its use case is mostly break / continue parent loop from nested loops 

// break / continue ------------------------
    someStatement {
        break;                              // breaks out this code block (stop any furhter execution in this code block)
        continue;                           // stops the current iteration 
    }
    
    
// sabeled statement ------------------------
    label: anyLoop {
        break|continue label;              // break / continue the labeled loop
    }
Example :

The break / continue keywords 

    const result = [];
    
    for (let i = 0; true; i++) {
      if (i > 10) {
        break;                            // break if iteration is higher than 10
      }
      if (i % 2 !== 0) {
        continue;                         // skip if cycle odd
      }
      
      result.push(i);
    }
    
    result;                              // -> [ 0, 2, 4, 6, 8, 10 ]

Labeled statement 

    const result = [];
    
    outer: for (let i = 0; i < 10; i++) {
      inner: for (let j = 0; j < 10; j++) {
        if (i >= 2) {
          break outer;                    // break outer loop
        }
        if (j % 2 !== 0) {
          continue inner;                 // skip inner odds
        }
        
        result.push(String(i) + String(j));
      }
    }
    
    result;                             // -> ['00', '02', '04', '06', '08', '10', '12', '14', '16', '18']
