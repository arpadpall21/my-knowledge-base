SharedArrayBuffer
new SharedArrayBuffer(length:nr) - creates a 0 filled SharedArrayBuffer object instance of the specified size in bytes
- the SharedArrayBuffer object represents a fixed length RAW binary data buffer (called 'byte array' = 1 address represents 1 byte)
- it cannot be modified directly (the DataView object can get and set it or use one of the TypedArrays)
- unlike the ArrayBuffer the SharedArrayBuffer shares its memory between threads and cannot be transfered from one thread to another

SharedArrayBuffer

- shared memory (ex: main.js worker_1.js worker_2.js sees the same underlying memory 'SharedArrayBuffer')

- not transferable (ex: the SharedArrayBuffer cannot be transfered from main.js to worker_1.js only shared!)

ArrayBuffer

- cloned memory (ex: main.js worker_1.js worker_2.js gets individual copies 'clones' of the send ArrayBuffer)

- transferable memory (ex: the transfered ArrayBuffer from main.js to worker_1.js is moved so it won't be visible in main.js anymore)

DEMO 
    var buffer = new ArrayBuffer(128);                                                 // we allocated 128 bytes of RAW data (empty = 0 filled)
    var dataView1 = new DataView(buffer);                                              // this data view sees the full ArrayBuffer 
    var dataView2 = new DataView(buffer, 4, 64);                                       // this one views the ArrayBuffer from 4th index (included) 64 bytes   
    
// (ArrayBuffer) isView() / byteLength / slice() --------------------------------------
    ArrayBuffer.isView(dataView2);                                                     // -> true                 // passed object is a DataView or TypedArray 
    buffer.byteLength;                                                                 // -> 128                  // ArrayBuffer contains 128 bytes 
    newBuffer = buffer.slice(32, 64);                                                  // -> ArrayBuffer(32)      // slices 32 bytes from the ArrayBuffer and returns it in a new ArrayBuffer object   
    
// (DataView) buffer / byteLength / byteOffset ----------------------------------------
    dataView2.buffer;                                                                  // -> ArrayBuffer(128)     // returns the referenced ArrayBuffer object (ArrayBuffer which will be read and written)   
    dataView2.byteLength;                                                              // -> 64                   // the byte length of this data view 
    dataView2.byteOffset;                                                              // -> 4                    // this dataView object views the referenced ArrayBuffer from 4th index  
    
// read / write the ArrayBuffer -------------------------------------------------------
    dataView1.setUint8(0, 241);                                                        // writing the 1st 8 bits of the ArrayBuffer as Unsigned Integer value 
    dataView1.getUint8(0);                                                             // -> 241                  // reads the 1st 8 bits of the ArrayBuffer as Unsigned Integer value (returns the set data)   
    dataView1.getInt8(0);                                                              // -> -15                  // the same data is interpreted differently if we read as an 8bit Unsigned Integer  
    
    dataView1.setFloat64(1, 145.2244, true);                                           // writing 64 bits from the 2nd byte of the ArrayBuffer as 64bit Floating point    // -! written as little endianness 
    dataView1.getFloat64(1, true);                                                     // -> 145.2244             // read the written data                                // -! read as little endianness 
    
    dataView2.setBigUint64(0, 6549874654165165n);                                      // writing 64 bits from the 4th byte of the ArrayBuffer as 64bit Big Integer       // -! as BigInt datatype  
    dataView2.getBigUint64(0);                                                         // -> 6549874654165165n    // read the written data                                // -! as BigInt datatype  
    
// CAVEATS ----------------------------------------------------------------------------
    dataView1.setUint16(0, -100);                                                      // writting an invalid value will give unexpected result (Unsigned Integer = 0 to 65535)    
    dataView1.getUint16(0);                                                            // -> 65436                // unexpected value 
    
    dataView1.setUint16(0, 2.999);                                                     // decimal numbers will be floored if set as integers 
    dataView1.getUint16(0);                                                            // -> 2                    // value floored

ArrayBuffer VS SharedArrayBuffer 

// ------------------------------------------------------------------------------------
// main.js ----------------------------------------------------------------------------
    var workers = require('worker_threads');
    
    var worker_1 = new workers.Worker('./worker_1.js');
    
    var shArrBuf = new SharedArrayBuffer(8);                                           // SharedArrayBuffer created 
    new DataView(shArrBuf).setFloat64(0, 1111.1111);                                   // set data to the SharedArrayBuffer
    
    worker_1.postMessage(shArrBuf);                                                    // the posted SharedArrayBuffer memory is shared between threads (NOT cloned)
        
    setTimeout(function(){
        console.log( new DataView(shArrBuf).getFloat64(0) );                           // -> 9999.9999     // -! since the SharedArrayBuffer memory is shared between threads modifications in 'worker_1.js' also visible here   
    }, 1000);
    
// ------------------------------------------------------------------------------------
// worker_1.js ------------------------------------------------------------------------
    var workers = require('worker_threads');
    
    workers.parentPort.on('message', function(buf){
        new DataView(buf).setFloat64(0, 9999.9999);                                    // modifying the received SharedArrayBuffer 
    });
    
// ------------------------------------------------------------------------------------
// main.js ----------------------------------------------------------------------------
    var workers = require('worker_threads');
    
    var worker_1 = new workers.Worker('./worker_1.js');
    
    var shArrBuf = new ArrayBuffer(8);                                                 // ArrayBuffer created 
    new DataView(shArrBuf).setFloat64(0, 1111.1111);                                   // set data to the ArrayBuffer
    
    worker_1.postMessage(shArrBuf);                                                    // the thread receives a clone of this ArrayBuffer (NOT a shared memory)    
    
    setTimeout(function(){
        console.log( new DataView(shArrBuf).getFloat64(0) );                           // -> 1111.1111     // -! since the ArrayBuffer is cloned each thread has a copy of this ArrayBuffer, so the ArrayBuffer in this trhead cannod be modified by another thread     
    }, 1000);

TypedArrays

- TypedArrays are Array-like view of the underlying binary data buffer (the underlying ArrayBuffer)

- there is not directy visible TypedArray constructor, instead there are global properties what implements typed Arrays for specific element types

- TypedArrays use the platform's hardware endianness to get or set the underlying ArrayBuffer

DEMO 
    new Uint8ClampedArray(4);                                                    // -> Uint8ClampedArray[0, 0, 0, 0]    // new TypedArray created with the specified length (empty)  
    
    new Int16Array(new Int16Array([475, 1244, -474]))                            // -> Int16Array[475, 1244, -474]      // new TypedArray created by using another TypedArray  
    
    new Float32Array([415.1, 124.22, 142.03]);                                   // -> Float32[415.1, 124.22, 142.03]   // new TypedArray created by using an iterable object   
    
    new BigUint64Array(new ArrayBuffer(16));                                     // -> BigUint64Array[0n, 0n]           // new TypedArray created by passing an ArrayBuffer   
    
    var uint8Clamped = new Uint8ClampedArray(4);
    var float32 = new Float32Array([142.12, 12.4, 145.2]);
    
// set() / subarray() / other array methods ------------------------------------
    uint8Clamped.set([-1454, 300, 32.8], 1);                                    // set to = [0, 0, 255, 33]             // sets the typedArray values from the 1st index (clamped array = clamps offset values and rounds decimals!)  
    var subArr = uint8Clamped.subarray(1, 3);                                   // -> Uint8ClampedArray[0, 255]         // returns a view of the targeted region
    subArr[1] = 0;                                                              // set TypedArray items through the view   
    uint8Clamped;                                                               // -> Uint8ClampedArray[0, 0, 0, 33]    // original TypeArray is modified 
    
    uint8Clamped.includes(33);                                                  // -> true                              // all Array methods are available on TypedArrays as well!   
    
// BYTES_PER_ELEMENT / name / buffer / byteOffset / byteLength / length --------
    BigUint64Array.BYTES_PER_ELEMENT;                                           // -> 8                                  // constructor builds a typedArray which contains 8 underlying bytes per array item    
    BigUint64Array.name;                                                        // -> 'BigUint64Array'                   // constructor's name 
       
    float32.buffer;                                                             // -> ArrayBuffer(12){}                  // returns the underlying ArrayBuffer object (binary raw data)   
    float32.byteOffset;                                                         // -> 0                                  // typedArray offset relative to the underlying ArrayBuffer    
    
    float32.length;                                                             // -> 3                                  // number of array items 
    float32.byteLength;                                                         // -> 12                                 // number of bytes of the underlying binary data
