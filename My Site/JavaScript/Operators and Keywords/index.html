Remember This :

- the const identifier only keeps primitives unchangable, complex data (objects) still can be changed (ex: myArray.push(4))

- the ... spread operator also spreads nested items/objects

- comparison operators can compare only primitives, therefore 2 objects cannot be compared (aways return false)

- comparison operators compare strings by their Alphabetical Order (numbers are placed before letters!)

Description and Demonstration:

- operators using one value called Unary Operators ( typeof 1 )

- operators using two values called Binary Operators ( 1 + 1 )

- operator using three values called Ternary Operator ( (10 > 2) ? "ok" : "not-ok" ) (only one in JS)

Operator Precedence

- operator precedence describes the order in wich order the operations will be performed in an arithmetic expression (like the * and / have higher precedence than the + and -)

- equal precedence operators are computed form left to right

    var x = 100 + 50 * 3;                                                              // -> 250   // the multiplication have higher precedence
    var x = (100 + 50) * 3;                                                            // -> 450   // precedence changed
    var x = 100 + 50 - 3;                                                              // -> 147   // equal precedence operators are computed from left to right

Bitwise Operators & / | / ^ / ~ / << / >>> / >>

- JavaScript stores numbers as 64 bit double precision floating points, but bitwise operations are performend on 32bit signed integers

- bitwise operations are performend on bit level

DEMO 
// bitwise AND ------------------------------------------------------------------------
    21 & 68;                                                                           // -> 4 
        //               (32bit signed integer binary representation)   
        // 21               -> 00000000000000000000000000010101 
        // 68               -> 00000000000000000000000001000100 
        //  4               -> 00000000000000000000000000000100                        // sets the bits to 1 where both operands have 1 bits 
    
    
// bitwise OR -------------------------------------------------------------------------
    21 | 68;                                                                           // -> 85       
        // 21               -> 00000000000000000000000000010101 
        // 68               -> 00000000000000000000000001000100 
        // 85               -> 00000000000000000000000001010101                        // sets the bits to 1 where operands have 1 bits 
    
    
// bitwise XOR ------------------------------------------------------------------------           
    21 ^ 68;                                                                           // -> 81 
        //               (32bit signed integer binary representation)   
        // 21               -> 00000000000000000000000000010101 
        // 68               -> 00000000000000000000000001000100 
        // 81               -> 00000000000000000000000001010001                        // sets the bits to 1 where operands have 1 bits 
                                                                                       // sets the bits to 0 where both operands have 1 bits 
    
// bitwise NOT ------------------------------------------------------------------------  
    ~21;                                                                               // -> -22 
        //               (32bit signed integer binary representation)   
        //   21             -> 00000000000000000000000000010101  
        //  -22             -> 11111111111111111111111111101010                        // inverses the bits 
        
    ~68;                                                                               // -> -69 
        //               (32bit signed integer binary representation)   
        //   68             -> 00000000000000000000000001000100                                 
        //  -69             -> 11111111111111111111111110111011                        // inverses the bits 
    
    
// bitwise left shift (zero-fill) -----------------------------------------------------
    21 << 1;                                                                           // -> 42
        //               (32bit signed integer binary representation)   
        //                  -> 00000000000000000000000000010101 
        //                  -> 00000000000000000000000000101010                        // fills 0's on the right while shifting    
    
    68 << 3;                                                                           // -> 544
        //               (32bit signed integer binary representation)   
        //                  -> 00000000000000000000000001000100 
        //                  -> 00000000000000000000001000100000                        // fills 0's on the right while shifting    
        
        
// bitwise right shift (zero-fill) ----------------------------------------------------
    21 >>> 1;                                                                          // -> 10
        //               (32bit signed integer binary representation)   
        //                  -> 00000000000000000000000000010101
        //                  -> 00000000000000000000000000001010                        // fills 0's on the left while shifting    
    
    -68 >>> 6;                                                                         // -> 67108862
        //               (32bit signed integer binary representation)   
        //      -68         -> 11111111111111111111111110111100
        // 67108862         -> 00000011111111111111111111111110                        // fills 0's on the left while shifting    
    
    
// bitwise right shift (left-bit-preserve) --------------------------------------------
    2607208886 >> 3;                                                                   // -> -210969802
        //               (32bit signed integer binary representation)   
        //  2607208886      -> 10011011011001101101100110110110  
        //  -210969802      -> 11110011011011001101101100110110                        // the leftmost bit is copied over while shifting    
    
    1801902518 >> 7;                                                                   // -> 14077363
        //               (32bit signed integer binary representation)   
        //  1801902518      -> 01101011011001101101100110110110 
        //    14077363      -> 00000000110101101100110110110011                        // the leftmost bit is copied over while shifting    
        
    -2147483648 >> 5;                                                                  // -> -67108864
        //               (32bit signed integer binary representation)   
        //  -2147483648     -> 10000000000000000000000000000000  
        //  -67108864       -> 11111100000000000000000000000000                        // the leftmost bit is copied over while shifting

Keywords

- JavaScript keywords = actions to be performed, these words are reserverd in JS and cannot be used for names!

The var / let / const keywords

DEMO 
    if (true) {                                 // this is a block scope not a local scope! 
        var x = 10;                             // this { } is considered as a block scope  
        let y = 11;
        const z = 12;
    }
    x;                                          // returns "10", "var" value can be accessed outside the code block
    y; / z;                                     // throws an Error because the "let" and the "const" only available inside the code block                       
    
    var x = 10;
    x = 11;                                     // can be reassigned;
    var x = 12;                                 // can be redeclared;
    
    let y = 20;
    y = 21;                                     // can be reassigned;
    let y = 22;                                 // cannot be redeclared (Throws an Error)
    
    const z = 30;
    z = 30;                                     // cannot be reassigned (Throws an Error)
    const z = 32;                               // cannot be redeclared (Throws an Error)

- the let and the const keywords does not create a global object properties like the var does (JavaScript keeps these identifiers in the DER (declarativ environment record) 

    var x = 10;         
    window.x                                    // -> 10
    x                                           // -> 10
//  ------------------------------------
    let y = 11;
    window.y                                    // -> undefined (not a property of the current scope)
    y                                           // -> 11
//  ------------------------------------
    let z = 12
    window.z                                    // -> undefined (not a property of the current scope)
    z                                           // -> 12

- the var keyword is hoisted (declaration moves to the top) VS the let and const are not hoisted!

    x = 10;
    var x;                                      // "x" is hoisted 
    
    y = 11;
    let y;                                      // "y" is not hoister (Throws an Error)  
    
    z = 11;
    const z;                                    // "z" is not hoister (Throws an Error)

- interestingly we cannot redeclare the let / const or reassign the const but if we use them on an Object or an Array we still can modify them by using some methods 

    const myArray = [];                         // cannot be redeclare or reassign 
    myArray.push("element");                    // add a new element through a method
    
    myArray[0];                                 // -> "element"

- using let let in for loop 

    for(var i = 0; i < 5; i++){                                                        // the var keyword's value changes at each iteration  
        setTimeout(function(){
            i;                                                                         // -> 5 | 5 | 5 | 5 | 5
        }, i * 1000);
    }
    
    for(let i = 0; i < 5; i++){                                                        // -! the let keyword is redeclared at each iteration (let keyword special case when used in a for loop)    
        setTimeout(function(){
            i;                                                                         // -> 0 | 1 | 2 | 3 | 4
        }, i * 1000);
    }

The ... (spread operator)

DEMO

- the ... spread operator mostly used with arrays 

    var myArray = [1, 2, 3];

// used on a function 
    function myFunc(a, b, c) {
        return a + b + c;
    }
    myFunc(...myArray);                         // returns "6" array items applied as function arguments
    
// combined arrays
    var myArray_2 = [a, b, ...myArray, c, d];
    myArray_2;                                  // returns "a,b,1,2,3,c,d" (gives position control)

- used on Objects 

    myObj = {prop_1:"value_1",
             nestedObj:{nestProp:"nestValue"}   // nested object
             };
    
    myObj_2 = {prop_1_Obj2:"value_1_Obj2", ...myObj};
    
// after combining the two objects with the spread operator this looks like this 
    myObj_2 = {prop_1_Obj2:"value_1_Obj2", 
               prop_1:"value_1",
               nestedObj:{nestProp:"nestValue"}
               };

- used on function call (this is amazing) 

    function min(...n) {                        // pased arguments are spreaded here
        arguments[0];                           // -> 1
        arguments[1];                           // -> 2
        arguments[2];                           // -> 3
        n                                       // -> [1, 2, 3]
    }
    min(1, 2, 3);
