The map() / forEach() methods

- the forEach() method has better performance over the map() method because its callback does not return anything (used for side effects only)

DEMO 
    var myArray = [10, 20, 30];
    
// map() ------------------------------------------------------------------------------
    myArray.map(function(val, index) {                                                 // -> [11, 21, 31]      // new Array returned
        val;                                                                               // -> 10 | 21 | 30      // values 
        index;                                                                             // -> 0  | 1  | 2       // indexes 
        return val +=1;                                                                    // received values are computed and returned 
    });
    
    myArray;                                                                           // -> [10, 20, 30]      // original Array is not modified  
    
// forEach() --------------------------------------------------------------------------
    myArray.forEach(function(val, index) {                                             // -> undefined         // -! callback does not return values therefore this method is faster over the 'map()' method   
        val;                                                                               // -> 10 | 21 | 30      // values 
        index;                                                                             // -> 0  | 1  | 2       // indexes        
        return val +=1;                                                                // -! this statement is ignored in this callback   
    });
    
    myArray;                                                                           // -> [10, 20, 30]      // original Array is not modified

    const persons = [
        { name: "Bob", age: 25 },
        { name: "Charlie", age: 25 },
        { name: "David", age: 30 },
        { name: "Alice", age: 30 },
    ]
    
    persons.sort((a, b) => a.age - b.age)
    // output:
    [                                           // the original name order is not changes, that's what stable sort garantees
      { name: 'Bob', age: 25 },
      { name: 'Charlie', age: 25 },
      { name: 'David', age: 30 },
      { name: 'Alice', age: 30 }
    ]

The entries() / keys() methods

DEMO 
    var myArray = [23, "myString", {prop1:"val", prop2:"val"}, [1, 2, 3]];
    
// keys() -----------------------------------------------------------------------------  
    var iterObj = myArray.keys();                                                      // iterable object returned with the Array's keys (indexes)
    iterObj.next();                                                                    // -> {value:0, done:false}             // call items from the iterable Object 
    iterObj.next();                                                                    // -> {value:1, done:false}
    iterObj.next();                                                                    // -> {value:2, done:false}
    iterObj.next();                                                                    // -> {value:3, done:false}
    iterObj.next();                                                                    // -> {value:undefined, done:true}      // end of the iterable object 
    
// entries() -------------------------------------------------------------------------- 
    var iterObj = myArray.entries();                                                   // iterable object returned with the Array's key(index)/value pairs 
    iterObj.next();                                                                    // -> {value:[0, 23], done:false}       // call items from the iterable Object 
    iterObj.next();                                                                    // -> {value:[1, "myString"], done:false}
    iterObj.next();                                                                    // -> {value:[2, {prop1:"val", prop2:"val"}], done:false}
    iterObj.next();                                                                    // -> {value:[3, [1, 2, 3]], done:false}
    iterObj.next();                                                                    // -> {value:undefined, done:true}      // end of the iterable object
