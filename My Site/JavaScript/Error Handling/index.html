The RangeError object 

    var number = 14;
    
    try {                       
        number.toPrecision(500); 
    }
    catch(errorObject){
        errorObject.name                        // -> "RangeError"
        errorObject.message                     // -> "toPrecision() argument must be between 1 and 100"
    }

The ReferenceError object 

    try {
        cap;   
    }
    catch(errorObject){
        errorObject.name                        // -> "ReferenceError"
        errorObject.message                     // -> "cap is not defined"
    }

The SyntaxError object 

    try{
        eval("console{}")                       // we use eval function here, otherwise the "try" statement cannot pass the exeption
        console.{}                              // this would shut down all further code because of the SyntaxError
    }
    catch(errorObject){
        errorObject.name                        // -> "SyntaxError"
        errorObject.message                     // -> "Invalid or unexpected token {"
    }

The TypeError object 

    var one = true;
    try{
        one.getDate(); 
    }
    catch(errorObject){
        errorObject.name                        // -> "TypeError"
        errorObject.message                     // -> "one.getDate is not a function"
    }

The try...catch statement

- the try...catch statement is used try to run some code and cach eventual errors in a scope so the main program doesn't exit 

    try {
        // code to test...
    }
    catch (err) {                  // code block executes if an error is thrown in the try block
                                                    // 2019  65  58  53  79  11.1  10   // the (err) is optional
        // error handling here... 
    }    
    finally {                                       // garanteed to executed regardless of the result of try and catch (even if the error is not handled) 
        // finally code...  
    }
    
    
// valid statemenet combinations ---------------------------
    try...catch
    try...finally
    try...catch...finally
DEMO 
    try {
        bla..bla                                                                       // this throws an error 
    }
    catch (err) {                                                                      // the error is caught and returned here 'err'   
        err;                                                                           // -> ReferenceError 
    }                                                                                  // the "catch" statement block only fires if error is thrown byt the "try" statement
    finally {
        console.log( "garanteed" );                                                    // block is garanteed to fire 
    }                                                                                  // if an error is thrown by the "try" statement and there is not "catch" statement to handle it all further code fails but the "finally" statemenet   
    
    console.log( 'continue' );                                                          // if the thrown error in not caught by the "catch" statement this won't be executed!
