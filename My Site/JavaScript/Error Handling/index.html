Description and Demonstration :

- when JavaScript encounters and error it creates an apropriate Error Object and throws it and stops all further script execution

The Error objects Error / TypeError / ReferenceError / RangeError / SyntaxError / EvalError / URIError / name / message

- we can create custom Error objects and throw them with the throw statement

DEMO 
    var error_ = new Error("myError");          // create a new error object
    var ref_error = ReferenceError("my_RefError");  // create a new referenceError object ("new" keyword can be omitted but not recommended)
        
    ref_error.name                              // -> "ReferenceError"
    ref_error.message                           // -> "my_RefError"     // message passed as argument
    throw ref_error;                            // ReferenceError throw in the console with the message "my_RefError"
    
    var myDate = new Date();                    // all further code is not executed because we have thrown an execption above

- the RangeError is thrown when a number is outside of legal limits 

    var number = 14;
    
    try {                       
        number.toPrecision(500); 
    }
    catch(errorObject){
        errorObject.name                        // -> "RangeError"
        errorObject.message                     // -> "toPrecision() argument must be between 1 and 100"
    }

- the ReferenceError is thrown when a non-existing variable is referenced 

    try {
        cap;   
    }
    catch(errorObject){
        errorObject.name                        // -> "ReferenceError"
        errorObject.message                     // -> "cap is not defined"
    }

- the SyntaxError is thrown when the code is syntactically invalid 

    try{
        eval("console{}")                       // we use eval function here, otherwise the "try" statement cannot pass the exeption
        console.{}                              // this would shut down all further code because of the SyntaxError
    }
    catch(errorObject){
        errorObject.name                        // -> "SyntaxError"
        errorObject.message                     // -> "Invalid or unexpected token {"
    }

- the TypeError is thrown when the value is an unexpected type (like a property is not on the object's prototype chain) 

    var one = true;
    try{
        one.getDate(); 
    }
    catch(errorObject){
        errorObject.name                        // -> "TypeError"
        errorObject.message                     // -> "one.getDate is not a function"
    }

- the URIError is thrown when an illegal character is used in a URI function 

    try{
        decodeURI("%£$*&");          
    }
    catch(errorObject){
        errorObject.name                        // -> "URIError"
        errorObject.message                     // -> "URI malformed"
    }

- the EvalError does NOT throw any eval errors it remained in JavaScript for compatibility

    try {
        eval(blaBla);                           <-- does not throw any EvalError
    } catch(e) {
        e.name                                  // -> "ReferenceError"      // not an EvalError instance
    }

The try...catch statement

- the try...catch statement is used try a piece code, if an error is thrown in the try statemenet, the program won't exit 

                                                                                          1  1  4  5  1  0.10.0     
SYNTAX:     try{                                                                       // tests for runtime exceptions (cannot handle Syntax Errors)    
                // code to test...
            }
            catch(err

                
 - the thrown error 

            
){                                                               // only fires if the try statement throws an exception (if the thrown exception is not handled by it all further code will fail (except finally statement))   
                // code to fire if the code in the try statement throws 
            }    
            finally{                                                                   // garanteed to fire regardless of the result of try and catch (even if the error is not handled) 
                // code garanteed to fire regardless of the result of the try statement  
            }
            
  // valid statemenet combinations ----------------------------------------------------
            try...catch
            try...finally
            try...catch...finally
DEMO 
    try {
        bla..bla                                                                       // this throws an error 
    }
    catch (err) {                                                                      // the error is caught and returned here 'err'   
        err;                                                                           // -> ReferenceError 
    }                                                                                  // the "catch" statement block only fires if error is thrown byt the "try" statement
    finally {
        console.log( "garanteed" );                                                    // block is garanteed to fire 
    }                                                                                  // if an error is thrown by the "try" statement and there is not "catch" statement to handle it all further code fails but the "finally" statemenet   
    
    console.log( 'continue' );                                                          // if the thrown error in not caught by the "catch" statement this won't be executed!

The throw and debugger statements 

                                                                                          1  1  4  5  1  0.10.0     
SYNTAX :    throw exp                                                                  // throws an exception and stops all further script execution (unwinds the stack / raises an exception)     
            
                                                                                          5  1  10  4  5  0.10.0     
            debugger                                                                   // invokes the debugging functionality of the host (if any)    
                                                                                          if the host has a debugger the code is handled accordingly (ex: in the browser sets a breakpoint and opens the console window)   
                                                                                          if the host doesn't have any debugger this statement does nothing
DEMO 
    var myCustom_error = new Error("my error message");
    throw myCustom_error;                                                              // (error object) throws exception "my error message" message printen in console
    throw "myError";                                                                   // (string) throws exception "myError" message printed in console
        // some code...                                                                // -! code not executed after the "throw" statement
    
// Unwinding the Stack -----------------
    function myFunc_A() { return myFunc_B()};
    function myFunc_B() { return myFunc_C()};
    function myFunc_C() { return myFunc_D()};
    function myFunc_D() {
        throw "I'm Raised!";                                                           // exception throwed directly to the caller "myFunc_A()"  -->  "unwinds the stack"
    }                                                                                  // "myFunc_B()" and "myFunc_C()" are ignored at this point 
    
    myFunc_A();                                                                        // exception is throwed here
    
    var myDocument = document;
    debugger;                                                                          // debugger functionality of the browser called here (script paused)
    console.info(typeof myDocument);                                                   // -! not executed
