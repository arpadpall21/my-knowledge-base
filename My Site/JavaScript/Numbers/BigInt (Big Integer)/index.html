Description and Demonstration :

- the BigInt (big integer) is a new data type in javaScript (ES11) which can represent integers in "arbitrary-precision" (unlimited digits (memory limited))

- bigint values are useful if we want to use numbers beyond the Number.MAX_SAFE_INTEGER and Number.MIN_SAFE_INTEGER limits

DEMO 
    8329848858n;                                                                       // big integer value 
    
    BigInt(-874);                                                                      // -> -874n              // number converted to bigint   // very identical to the Number() constructor  
    BigInt('8743');                                                                    // -> 8743n              // string converted to bigint
    BigInt(true);                                                                      // -> 1n                 // boolean converted to bigint
    BigInt(457.1);                                                                     // -! error thrown (decimals cannot be converted to bigint values)
    
// not mixable with Number instances in arithmetic operations -------------------------
    4n + 4;                                                                            // -! error thrown (not mixable with Number instances in arithmetic operations)
    4n + BigInt(4);                                                                    // -> 8n                 // solution is to convert to the same data type 
    
// decimals are ignored ---------------------------------------------------------------
    5 / 3;                                                                             // -> 1.6666666666666667
    5n / 3n;                                                                           // -> 1n                 // bigint result (decimals are ignored)    
    
// using operators with bigint values -------------------------------------------------
    4n % 3n;                                                                           // -> 1n                 // works just like with numbers 
    4n ** 3n;                                                                          // -> 64n
    
    4n >>> 3n;                                                                         // -! error thrown       // these 2 operators cannot be used with bigint values    
    +4n;                                                                               // -! error thrown 
    
// unsupported by Math object ---------------------------------------------------------
    Math.pow(4, 3);                                                                    // -> 64
    Math.pow(4n, 3n);                                                                  // -! error thrown       // Math object does not support bigint values 
    
// comparison mixing ------------------------------------------------------------------
    5n >= 4;                                                                           // -> true               // types can be mixed in comparison   
    
    5n == 5;                                                                           // -> true               // have the same value (Abstract Equality Comparison)    
    5n === 5;                                                                          // -> false              // different types (Same Value Comparison)
    
// conditional statement support ------------------------------------------------------
    7n ? 'evaluated true' : 'evaluated false';                                         // -> 'evaluated true'   // transformed just like numbers 
    0n ? 'evaluated true' : 'evaluated false';                                         // -> 'evaluated true'

    console.log( typeof myBigInt_obj );                     // -> 'object'

    console.log( result );                                  // -> 500n
    console.log( typeof result );                           // -> 'bigint'          // converted to bigInt primitive value 
    
console.log( 'mixing number types -----------------------------------------------' );
    try {
        10n + 1;                                            // -> error thrown (Number and BigInt cannot be mixed)
    } catch(err) {
        console.log( err );
    }
    
    console.log( 10n + BigInt(1) );                         // -> 11n               // solution is to convert on to the other's data type
};
</script>

    console.log( 100n << 2n );                              // -> 400n              // all bitwise operators are supported except the >>> (zero-right shift)

    } catch(err) {
        console.log( err );
    }

    } catch(err) {
        console.log( err );
    }
};  
</script>

The asIntN() / asUintN() Static Methods

DEMO 
// asIntN() ---------------------------------------------------------------------------
    BigInt.asIntN(8, 127n);                                                            // -> -127n      // 127 is the largest 8bit signed integer 
    BigInt.asIntN(8, 130n);                                                            // -> -126n      // out of range values are coerced into 8bit signed integer range  
    
    BigInt.asIntN(8, -128n);                                                           // -> -128n      // 128 is the lowest 8bit signed integer 
    BigInt.asIntN(8, -12456n)                                                          // -> 88n
    
    
    BigInt.asIntN(16, 34000n);                                                         // -> -31536n    // -! the passed bigint is coerced into the 16bit Signed Integer range but the returned value is unexpected   
    
// asUintN() --------------------------------------------------------------------------
    BigInt.asUintN(8, 255n);                                                          // -> 225n        // is the largest 8bit unsigned integer 
    
    BigInt.asUintN(8, 256n);                                                          // -> 0n          // out of range values are coerced into 8bit unsigned integer range

<script>
var asIntN_test = function(){
console.warn( "8bit limited range  --------------------------------------------------------" );
    console.log( BigInt.asIntN(8, -98n) );                  // -> -98n          // limits the passed bigint to an 8bit Signed Integer Value 
    console.log( BigInt.asUintN(8, 245n) );                 // -> 245n          // limits the passed bigint to an 8bit Unsigned Integer Value
    
    console.log( BigInt.asIntN(8, 1000n) );                 // -> -24n          // passing an out of range value will result in an inexpected result 
    console.log( BigInt.asUintN(8, 1000n) );                // -> 232n          // passing an out of range value will result in an inexpected result
    
console.warn( "6bit limited range  --------------------------------------------------------" );
    console.log( BigInt.asIntN(6, 31n) );                   // -> 31n           // 6bit range limits (between these limits the values are returned as passed)
    console.log( BigInt.asIntN(6, -32n) );                  // -> -32n
    
    console.log( BigInt.asIntN(6, 32n) );                   // -> -32n           // -! out of range values are coerced into the 6bit Signed Integer range but the result is unpredictable 
    console.log( BigInt.asIntN(6, -33n) );                  // -> -31n
};
</script>

The toString() / toLocaleString() / valueOf() Methods

DEMO 
// toString() -------------------------------------------------------------------------
    100n.toString()                                                                    // -> "100"              // the 'n' is omitted
    100n.toString(2);                                                                  // -> "1100100"          // binary representation 
    100n.toString(8);                                                                  // -> "144"              // octal representation
    100n.toString(16);                                                                 // -> "64"               // hexadecimal representation
        
// toLocaleString() -------------------------------------------------------------------
    1000000n.toLocaleString('tr-TR');                                                  // -> "1.000.000"        // bigint returned in a string according to the set language    
    1000000n.toLocaleString('en-EN');                                                  // -> "1,000,000"
    1000000n.toLocaleString('fr-FR');                                                  // -> "1 000 000"
    
    1000000n.toLocaleString('sv-SE', {style:'currency', currency:"SEK"});              // -> "1 000 000,00 kr"  // bigint style as currency value 
    
// valueOf() --------------------------------------------------------------------------
    var bigInt_obj = Object(324n);                                                     // bigint wrapped in object 
    bigInt_obj.valueOf();                                                              // -> 324n               // primitive value returned of the wrapper object
