Description and Demonstration :

- JavaScript stores numbers as double precision floating point numbers (64 bits) 

    nr                     // number literal
    Number(val:any)        // converts the passed value into number (not recommended)
    
    100_000_000                             // 2021 numeric separator
                                            //  83  78  69  83  14  12 
    
    
// notations ---------------------------------------------
    nre+nr  |  nre-nr               // scientific exponent notation (the number after the e basically tells where the decimal dot is compared to its current position)
    
    0xhex...                          // hexadecimal notation (0 - f)
    
    0nr...                          // octal notation (0 - 7) (deprecated since ES6 throws an error in strict mode)
    0onr...  |  0Onr...       // 6 octal notation (0 - 7)
                                                                              //  51  34  38  12  10  4 
    
    
// keywords ------------------------------------------------
    NaN                           // not a number (number value indicating that an expression result is not a number)
    
    Infinity                      // Positive Infinity
    -Infinity                     // Negative Infinity
DEMO 
// Integers ---------------------------------------------------------------------------
    4512654                                                                            // integers are safe to use to a certain limit   
    Number.MAX_SAFE_INTEGER;                                                           // -> 9007199254740991         // largest safe integer   
    Number.MIN_SAFE_INTEGER;                                                           // -> -9007199254740991        // smallest safe integer   
    
    56432165795465497643;                                                              // -> 56432165795465495000     // !! beyond the safe limits numbers are not accurate because of the double precision mechanism     
    
// Numbers with decimals --------------------------------------------------------------
    1547.45;                                                                           // -! decimal are NOT 100% accurate because of floating point nature careful! 
    0.2 + 0.1;                                                                         // -> 0.30000000000000004      // !! NOT accurate 
    
// exponent notations -----------------------------------------------------------------
    143e+3;                                                                            // -> 143000                   // scientific exponent notation  
    143e-3;                                                                            // -> 0.143                    // scientific exponent notation  
    
// hexadecimal notations --------------------------------------------------------------
    0xa4f;                                                                             // -> 2639                     // hexadecimal notation 
    
// octal notations --------------------------------------------------------------------
    0o5117;                                                                            // -> 2639                     // octal notations (new in ES6)    
    05117;                                                                             // -> 2639                     // -! octal notations deprecated since ES6
    
    074                                                                                // -> 60                       // treated as octal notation 
    094                                                                                // -> 94                       // !! treated as decimal number (this is why this notation format is deprecated)  
    
// NaN --------------------------------------------------------------------------------
    Number('string');                                                                  // -> NaN                      // value cannot be converted to a number  
    typeof NaN;                                                                        // -> 'number'                 // primitive value 
    
// Infinity / -Infinity ---------------------------------------------------------------
    10**100;                                                                           // -> Infinity                 // returned value is larger than any other number 
    10 - 10e+400;                                                                      // -> -Infinity                // returned value is smaller than any other number
    
    typeof Infinity                                                                    // -> 'number'                 // primitive value    
    typeof -Infinity                                                                   // -> 'number'                 // primitive value    
    
    1 / Infinity;                                                                      // -> 0                        // Infinity behaves as numbers on arithmetic operations

CODE : 
    <button onmouseup="testNaN()"> RUN TEST (see the result in console) </button>
<script>
var testNaN = function(){
console.warn( 'TEST-1 ---------------------------------------------------------------' );
    console.log( window.isNaN(2) );                           // -> false            
    console.log( Number.isNaN(2) );                           // -> false             
    
console.warn( 'TEST-2 ---------------------------------------------------------------' );
    console.log( window.isNaN('2') );                         // -> false            
    console.log( Number.isNaN('2') );                         // -> false             
    
console.warn( 'TEST-3 ---------------------------------------------------------------' );
    console.log( window.isNaN('deux') );                      // -> true        // passed argument when converted to a number has no numeric value  
    console.log( Number.isNaN('deux') );                      // -> false       // passed argument not a number type in the first place (so automatically returned false)    
    
console.warn( 'TEST-3 ---------------------------------------------------------------' );
    console.log( window.isNaN(NaN) );                         // -> true        // passed argument is a number type and its value is not a number  
    console.log( Number.isNaN(NaN) );                         // -> false       // passed argument is a number type and its value is not a number  
};
</script>

var toStr = function(){
    var myNr = 241;
        console.log( myNr.toString() );                                 // -> "21"          // simple string conversion 
        console.log( myNr.toString(2) );                                // -> "11110001"    // binary value 
        console.log( myNr.toString(8) );                                // -> "361"         // octal value
        console.log( myNr.toString(16) );                               // -> "f1"          // hexadecimal value
        console.log( myNr.toString(21) );                               // -> "ba"          // base 21 value
        
console.warn( 'reconvert ------------------------------------------------------' );
    var octal = 0o361;
    var hex = 0xf1;
        console.log( octal.toString() );                                // -> 241           // octal to decimal 
        console.log( hex.toString() );                                  // -> 241           // hexadecimal to decimal 
};
</script>

The parseFloat() / parseInt() methods

Example :

CODE : 
    <p id="boolean"></p>
    <p id="float"></p>
    <p id="str1"></p>
    <p id="str2"></p>
    <p id="str3"></p>
<script>
    var a = true;
    var b = false;
    var c = " 13.65"
    var d = "21.65 04 1985"
    var e = "32 years old"
    var f = "I am 32"
    
    document.getElementById('boolean').innerHTML = "\"false\" / \"true\" || ".fontcolor('white') + 
        "Number() = " + Number(a) + " / " +  Number(b) + " || " +                   // booleans work only with "Number()"!
        "parseFloat() = " + parseFloat(a) + " / " + parseFloat(b) + " || " +
        "parseInt() = " + parseInt(a) + " / " + parseInt(b);    
    
    document.getElementById('float').innerHTML = "\"13.65\" || ".fontcolor('white') +
        "Number() = " + Number(c) + " || " +
        "parseFloat() = " + parseFloat(c) + " || " +
        "parseInt() = " + parseInt(c);                                              // returns a whole number (DOES NOT PROCES)
    
    document.getElementById('str1').innerHTML = "\"21.65 04 1985\" || ".fontcolor('white') +
        "Number() = " + Number(d) + " || " +                                        // cannot process multiple numbers
        "parseFloat() = " + parseFloat(d) + " || " +                                // returns only the first number
        "parseInt() = " + parseInt(d) + " || ";                                     // returns only the first number as a whole number   
    
    document.getElementById('str2').innerHTML = "\"32 years old\" || ".fontcolor('white') +
        "Number() = " + Number(e) + " || " +                                        // cannot process numbers and strings in the same string
        "parseFloat() = " + parseFloat(e) + " || " +
        "parseInt() = " + parseInt(e);
    
    document.getElementById('str3').innerHTML = "\"I am 32\" || ".fontcolor('white') +
        "Number() = " + Number(f) + " || " +                                      
        "parseFloat() = " + parseFloat(f) + " || " +
        "parseInt() = " + parseInt(f);
</script>
