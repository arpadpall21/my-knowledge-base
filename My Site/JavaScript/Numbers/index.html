Remember This :

- avoid starting numbers with 0 because they might be interpreted as octal notation, (in strict mode an error will be thrown)

- the 32 bit Single Precision floating point is accurate about 7 significant digits and the 64 bit Double Precision floating point is accurate about 16 significant digits, but they can represent far larger/smaller numbers by the cost of accuracy!

Description and Demonstration :

- JavaScript numbers are stored as Double Precision floating point numbers (Binary64) or as BigInt values

- the highest safe integer is 9.007.199.254.740.991 and the lowest is -9.007.199.254.740.991, beyond these Integer limits Number instances are not accurate!

DEMO 
// Integers ---------------------------------------------------------------------------
    4512654                                                                            // integers are safe to use to a certain limit   
    Number.MAX_SAFE_INTEGER;                                                           // -> 9007199254740991         // largest safe integer   
    Number.MIN_SAFE_INTEGER;                                                           // -> -9007199254740991        // smallest safe integer   
    
    56432165795465497643;                                                              // -> 56432165795465495000     // !! beyond the safe limits numbers are not accurate because of the double precision mechanism     
    
// Numbers with decimals --------------------------------------------------------------
    1547.45;                                                                           // -! decimal are NOT 100% accurate because of floating point nature careful! 
    0.2 + 0.1;                                                                         // -> 0.30000000000000004      // !! NOT accurate

The toPrecision() / toFixed() / toExponential() Methods

DEMO 
// toPrecision() ----------------------------------------------------------------------
    var myNr = 21.5428;
    myNr.toPrecision();                                                                // -> "21.5428"     // original length (no argument)
    myNr.toPrecision(1);                                                               // -> "2e+1"        // integer expressed with exponential
    myNr.toPrecision(2);                                                               // -> "22"          // rounds up! (rounds decimals if necessary)  
    myNr.toPrecision(4);                                                               // -> "21.54"     
    myNr.toPrecision(8);                                                               // -> "21.542800"   // adds some 0s
    
// toFixed() --------------------------------------------------------------------------
    var myNr = 21.5528;
    myNr.toFixed();                                                                    // -> "21.5428"     // original length (no argument)
    myNr.toFixed(0);                                                                   // -> "22"          // -! rounds up!
    myNr.toFixed(2);                                                                   // -> "21.56        // -! rounds up!
    myNr.toFixed(5);                                                                   // -> "21.55280"    // adds 0's 
    
// toExponential() --------------------------------------------------------------------
    var myNr = 2.12345;
    myNr.toExponential();                                                              // -> "2.12345e+0"   // original length (no argument)
    myNr.toExponential(2);                                                             // -> "2.12e+0"      // 2 decimal precision 
    myNr.toExponential(4);                                                             // -> "2.1234e+0"    // 4 decimal precision
    
    var myNr2 = 0.2123456;
    myNr2.toExponential();                                                             // -> "2.12345e-1"   // original length (no argument)
    myNr2.toExponential(2);                                                            // -> "2.12e-1"      // 2 decimal precision 
    myNr2.toExponential(4);                                                            // -> "2.1234e-1"    // 4 decimal precision
    
    var myNr3 = 2123.456;
    myNr3.toExponential();                                                             // -> "2.12345e+3"   // original length (no argument)
    myNr3.toExponential(2);                                                            // -> "2.12e+3"      // 2 decimal precision 
    myNr3.toExponential(4);                                                            // -> "2.1234e+3"    // 4 decimal precision

The parseFloat() / parseInt() Methods

DEMO 
    window.parseInt === Number.parseInt;                                               // -> true           // methods are the same (same objects)  
    window.parseFloat === Number.parseFloat;                                           // -> true           // methods are the same (same objects)  
        
    window.parseFloat("13.3 feet")                                                     // -> 13.3           // string parsed left-to-right as many digits as possible    
    window.parseInt("13.3 feet")                                                       // -> 13             // string parsed left-to-right as many digits as possible    
    
    Number.parseFloat("It's 21")                                                       // -> NaN            // string does not stat with a number therefore no parse is possible    
    Number.parseInt("It's 21")                                                         // -> NaN            // string does not stat with a number therefore no parse is possible    
    
// interprete values in a set base ----------------------------------------------------
    Number.parseInt("10", 2);                                                          // -> 2              // 10 binary value = 2 decimal 
    Number.parseInt("17", 8);                                                          // -> 15             // 17 octal value = 15 decimal 
    Number.parseInt("a", 16);                                                          // -> 10             // a hexadecimal value = 10 decimal
    Number.parseInt("j", 32);                                                          // -> 19             // j base32 value = 19 decimal
    
    Number.parseInt("21", 2);                                                          // -> NaN            // there is no 2 in binary base 
    Number.parseInt('101211', 2);                                                      // -> 5              // -! only the first 3 digits are parsed, after the digit 2 everything is ignored
