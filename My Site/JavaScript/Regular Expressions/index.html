Remember This :

- double backslashes are required if we want to use special characters in regular expression created with RegExp constructor string (new RegExp("\\b.", "i"))

- regular expession patterns are not designed to negate search, the sort of way around is using look aheads/look behinds or [^] negate brackets

Description and Demonstration :

- a regular experssion is a sequence of characters forming a search pattern which can be used for text search

- a regular expression searches a string for a match in the sequence, once a match is found the result is returned and the search is paused, a new search will start from this paused index (lastIndex)

- there are serveral str methods supporiting regex (.search(), .match(), .replace(), etc... )

Flags (Modifiers) i / m / g / y / u / s

Example: 
// -- i / g -----------------------------------------------
    var txt = "The JavaScript is the true wonder of the word";
    var myRegExp = RegExp("the", "ig");
    
     myRegExp.exec(txt);                                   // -> ["The", index:0]       // case-insensitive match 
     myRegExp.exec(txt);                                   // -> ["the", index:18]      // the "g" flag ensures that the next search will start from end of this one   
        myRegExp.lastIndex;                                // -> 21                     // "myRegExp" search position right now 
     myRegExp.exec(txt);                                   // -> ["the", index:37]        
     myRegExp.exec(txt);                                   // -> null                   // no more match (if we keep calling the "exec" method the search restarts)  
    
// -- m ---------------------------------------------------   
    var txt = "Is this\nis a Kis";
              
    txt.match(/^is/mig);                                   // -> ["Is", "is"]           // matches at the beginning of each line
    txt.match(/is$/mig);                                   // -> ["is", "is"]           // matches at the end of each line
    
//  -- y --------------------------------------------------
    var txt = "bla Bla";
    var stickyExp = /b/iy;
    
    stickyExp.exec(txt);                                   // -> null                   // because the "y" sticky flag restricts the search to index 0      // -! does not search any further   
    stickyExp.lastIndex = 1;                               // search index set to 1
    stickyExp.exec(txt);                                   // -> ["b"]                  // match at index 1   
    
//  -- u --------------------------------------------------
    /ðŸŽ{3}/.test("ðŸŽðŸŽðŸŽ");                              // -> false                   // because the ðŸŽ is composed of two bytes, so this string is basically composed of 6 bytes 
    /ðŸŽ{3}/u.test("ðŸŽðŸŽðŸŽ");                             // -> true                    // the "u" flag handles unicode code points instead of character per byte  
    
// -- s ---------------------------------------------------
    RegExp(/./).test("\n");                                // -> false                  // because the "." metacharacter by default does not matches any new line and line terminator character   
    RegExp(/./, "s").test("\n");                           // -> true                   // the "s" (dotAll) flag makes the "." metacharacter to match ANY character (including new line and line terminator)

// brackets -----------------------------------------------
    var txt = "I count numbers 12345678";
    
    txt.match(/[cnu]/g);                                   // -> ["c", "u", "n", "n", "u"]          // matches any individual characters
    txt.match(/[^cnu]/g);                                  // -> ["I", " ", "o", "t", " ", "m", "b", "e", "r", "s", " ", "1", "2", "3", "4", "5", "6", "7", "8"]  
    txt.match(/[3-9]/g);                                   // -> ["3", "4", "5", "6", "7", "8"]     // matches the character range 
    txt.match(/[^3-9]/g);                                  // -> ["I", " ", "c", "o", "u", "n", "t", " ", "n", "u", "m", "b", "e", "r", "s", " ", "1", "2"]
    
// parentheses --------------------------------------------
    var santaClaus = "hohohohooooohoho";
    var animals = "41 pigs and 3 dogs and 1 cat"
    
    santaClaus.match(/(ho)+/g);                            // -> ["hohohoho", "hoho"]               // matches the pattern 
    animals.match(/\d+ (pig|dog|cat)s?/g)                  // -> ["41 pigs", "3 dogs", "1 cat"]     // the space can be followed by any of the enclosed patterns

    var kutyus = "Kicsi kutya";
    RegExp('^k', 'ig').test(kutyus);                       // -> true                               // string starts with a "k" character
    RegExp('s$').test(kutyus);                             // -> false
    RegExp('a$').test(kutyus);                             // -> true                               // string ends with an "a" character

Metacharacters . / \w / \W / \d / \D / \s / \S / \b / \B / \0 / \n / \f / \r / \t / \v / \u / \u{} / \x / \nr / \p{} / \P{}

Example: 
    let text = "ECMA-262 is the official name of JavaScript";    
        
// -- . ---------------------------------------------------
    text.match(/E......../g);                              // -> ["ECMA_262 "]          // matches any character except new line and line terminator 
    
// -- \w \W -----------------------------------------------
    text.match(RegExp('^\\w+', 'g'));                      // -> ["ECMA_262"]           // matches (a-z|A-Z|0-9|_)      // -! double backslashes are required with RegExp constructor string   
    text.match(RegExp('\\W', 'g'));                        // -> [" ", " ", " ", " ", " ", " "]                         // -! double backslashes are required with RegExp constructor string   
    
// -- \d \D -----------------------------------------------
    text.match(/\d/g);                                     // -> ["2", "6", "2"]        // matches (0-9)
    text.match(/\D+/g);                                    // -> ["ECMA_", " is the official name of JavaScript"]
    
// -- \s \S -----------------------------------------------
    text.match(/\s/g);                                     // -> [" ", " ", " ", " ", " ", " "] // matches any white space character 
    text.match(/\S+\s/);                                   // -> ["ECMA_262 "]   
    
// -- \b \B -----------------------------------------------
    text.match(/\bo\w+/g);                                 // -> ["official", "of"]     // matches the beginnin of the word (each word in the string) 
    text.match(/.+\d\b/);                                  // -> ["ECMA_262"]           // matches the end of the word (each word in the stirng)        // matches words which end with a digit 
    text.match(/\Ba/g);                                    // -> ["a", "a", "a", "a"]   // matches in the word (each word in the string)                // matches any "a" character in the word (start and end characters are excluded!)   
    
// -- \uHexHexHexHex \u{Hex...} \xHexHexHex NrNrNr --------
    text.match(/\u0061/g);                                 // -> ["a", "a", "a", "a"]   // (hexadecimal) matches Unicode characters   (4 digits are required)
    text.match(/\x61/g);                                   // -> ["a", "a", "a", "a"]   // (hexadecimal) matches Latin characters     (limited to 2 digits)
    text.match(/\141/g);                                   // -> ["a", "a", "a", "a"]   // (octal) matches Latin characters           (limited to 3 digits)
    
// -- \uHexHexHexHex VS \u{Hex...} ------
    /\u1F600/u.test('ðŸ˜€');                                // -> false       // -! the \u is limited to 4 hex digits (so cannot handle this situaiton)
    /\u{1F600}/u.test('ðŸ˜€');                              // -> true        // the \u{} accept more than 4 hex digits
    
// -- \p{} \P{} -------------------------------------------
    /\p{Lu}/u.test("Lu");                                  // -> true                   // letter uppercase             // -! "u" flag is required
    /\p{Sc}/u.test("$");                                   // -> true                   // symbol currency 
    /\P{Sc}/u.test("4");                                   // -> true                   // not a symbol currency

Quantifiers * / + / ? / {}

- quantifiers specify how many instances of a character or a group must be present

DEMO : 
    var txt = "c co coo coool";
    
    txt.match(/co+/g);                                                                 // -> ['co', 'coo', 'cooo']        // matches one or more times
    txt.match(/co*/g);                                                                 // -> ['c', 'co', 'coo', 'cooo']   // matches zero or more times
    txt.match(/co?/g);                                                                 // -> ['c', 'co', 'co', 'co']      // matches zero or once 
                 
    txt.match(/co+?/g);                                                                // -> ['co', 'co', 'co']           // matches once exactly (NON-GREEDY)
    txt.match(/co*?/g);                                                                // -> ['c', 'c', 'c', 'c']         // matches as few times as possible (NON-GREEDY I don't see that useful)   
    txt.match(/co??/g);                                                                // -> ['c', 'c', 'c', 'c']         // matches as few times as possible (NON-GREEDY I don't see that useful)   
            
    txt.match(/co{2}/g;                                                                // -> ['coo', 'coo']               // matches 2 times exactly
    txt.match(/co{2,}/g);                                                              // -> ['coo', 'cooo']              // matches 2 times minimum
    txt.match(/co{1,3}/g;                                                              // -> ['co', 'coo', 'cooo']        // matches from 1 to 3 times

Look Ahead / Look Behind (?=) / (?!) / (?<=) / (?<!)

DEMO 
// lookahead ----------------------------------------------
    '3 cows and 1dog'.match(RegExp('\\d(?=\\w)'));         // -> ['1']       // positive lookahead matches a digit character followed by a word character 
    '3 cows and 1dog'.match(RegExp('\\d(?!\\w)'));         // -> ['3']       // negative lookahead matches a digit character NOT followed by a word character
        
    "Arpad P".match(RegExp(/Arpad(?= P)/));                // -> ['Arpad']   // matches "Arpad" if followed by " P"
    "Kutya Farka".match(RegExp(/Kutya(?! Fule)/));         // -> ['Kutya']   // matches "Kutya" if NOT followed by " Fule"
        
// lookbehind ---------------------------------------------
    '3 cows and 1dog'.match(/(?<=\d)\w/);                  // -> ['d']       // positive lookbehind matches a word character if it's preceded by a digit character
    '3 cows and 1dog'.match(/(?<!\d)\w/);                  // -> ['3']       // negative lookbehind matches a word character if it's NOT preceded by a digit character
                                                                              
    "B Bircsy".match(/(?<=B )Bircsy/)                      // -> ['Bircsy']  // matches "Bircsy" if it's preceded by "B "
    "Farkas Kutya".match(/(?<!Bernathegyi )Kutya/);        // -> ['Kutya']   // matches "Kutya" if it's NOT preceded by "Bernathegyi ""

// lastIndex ----------------------------------------------
    var myMaci = "Mici Maci";
    var myRegExp = RegExp("i", "g");                       // the "g" (global) flag allows us to use the lastIndex property in order to set the search index position   
    
    myRegExp.lastIndex = 2;                                // we set the search index position for the regex object at index 2
    myRegExp.exec(myMaci);                                 // -> ['i']      // character matches at index 3
        myRegExp.lastIndex;                                // -> 4          // next search star index 
    myRegExp.exec(myMaci);                                 // -> ['i']
        myRegExp.lastIndex;                                // -> 9
    myRegExp.exec(myMaci);                                 // -> null       // no more match found in the string 
        myRegExp.lastIndex;                                // -> 0          // reset regex object index position

The $1-$9 / input|$_ / lastMatch|$& / lastPattern|$_ / leftContext|$` / rightContext|$' properties

DEMO 
    var testRegEx = /\w(\d)(\d)(\d)\w/g;                   // the 'g' flag allow us to continue the next match from the previous search index
    
// first match --------------------------------------------
    testRegEx.exec('aaa b123b c456c d789d');               // -> ["b123b", "1", "2", "3"]               
    
    RegExp.input;                                          // -> 'aaa b123b c456c d789d'                // original string against we passed the test  
    RegExp.lastMatch;                                      // -> 'b123b'                                // current match
    
    RegExp.$1                                              // -> 1                                      // references the first regex group of the current match (first regex parentheses)
    RegExp.$2                                              // -> 2                                      // references the second regex group of the current match (second regex parentheses)
    RegExp.lastParen                                       // -> 3                                      // references the last regex group of the current match (last regex parentheses)
    
    RegExp.leftContext                                     // -> 'aaa '                                 // left part of the original string relative to the current match (excluded)
    RegExp.rightContext                                    // -> ' c456c d789d'                         // right part of the original string relative to the current match (excluded)
    
// second match (using different syntax) ------------------
    testRegEx.exec('aaa b123b c456c d789d');               // -> ["c456c", "4", "5", "6"]
    
    RegExp.$_;                                             // -> 'aaa b123b c456c d789d'                // -! this syntax differs from the others
    RegExp['$&'];                                          // -> 'c456c'                
    
    RegExp.$1;                                             // -> 4                                      // no alternative syntax
    RegExp.$2;                                             // -> 5                                      // no alternative syntax
    RegExp["$+"];                                          // -> 6
    
    RegExp["$`"];                                          // -> 'aaa b123b '
    RegExp["$'"];                                          // -> ' d789d'
    
// third test ---------------------------------------------
    testRegEx.exec('other string');                        // -> null                                   // -! there is no match here
    
    RegExp.$_;                                             // -> 'aaa b123b c456c d789d'                // -! because there is no match these properties won't be updated whith the last test so the old data is returned

Methods toString() / test() / exec()

DEMO 
// toString() / test() ----------------------------------------------------------------
    var myRegEx = RegExp('[A-Z].+?', 'gms');                                           // Block letter followed by one character 
        
    myRegEx.toString();                                                                // -> '/[A-Z].+?/gms'
        
    myRegEx.test("she loves Teslas");                                                  // -> true
    myRegEx.lastIndex;                                                                 // -> 12
    myRegEx.test("she loves Teslas");                                                  // -> false                      // no match found after index 12 (included)   
    
// exec() -----------------------------------------------------------------------------
    var txt = "4dogs and 9ducks animals";
    var digitAndChar = new RegExp(/\d(\w)(\w)(\w)/, "g");                              // matches one digit followed by 3 word characters 
        
    digitAndChar.exec(txt);                                                            // -> ["4dog", "d", "o", "g"]    // 2+ arguments are the regexp group matches   
        digitAndChar.lastIndex;                                                        // -> 4
    digitAndChar.exec(txt);                                                            // -> ["9duc", "d", "u", "c"]
        digitAndChar.lastIndex;                                                        // -> 14
    digitAndChar.exec(txt);                                                            // -> null                       // no more match     
        digitAndChar.lastIndex;                                                        // -> 0                          // cycle repeats if we keep calling the 'exec' method
