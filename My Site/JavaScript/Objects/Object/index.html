The Object.create() method 

    var mercedess = {color:"white", country:"Germany"};
    
    var audi = Object.create(mercedess);                                               // -> {}                 // the "audi" is an empty object because it inherits properties from 'mercedess' which will be its prototype   
    audi.color;                                                                        // -> "white"            // inherited propertiey 
    
    mercedess.hasOwnProperty("color");                                                 // -> true               // native property 
    audi.hasOwnProperty("color")                                                       // -> false              // not a native property 
    
    mercedess.isPrototypeOf(audi)                                                      // -> true               // the "mercedes" object is the prototype of the "audi" object   
    
    var bmw = Object.create(mercedess, {wd4:{value:true}});                            // defines an own property when creating an instance 
    Object.getOwnPropertyDescriptor(bmw, "wd4");                                       // -> {configurable:false, enumerable:false, value:true, writable:false}     
    bmw.hasOwnProperty("wd4");                                                         // -> ture               // native property 
    
// property descriptors ---------------------------------------------------------------
    var audi = Object.create(mercedess, {topSpeed:{                                    // -! these are the default property definiers 
        value: undefined,                                                              // undefined value 
        writable: false ,                                                              // cannot be modified  
        enumerable: false,                                                             // not shown in when the parent object is iterated through   
        configurable: false                                                            // cannot be deleted with delete  
    }})
    
    audi;                                                                              // -> {topSpeed:undefined}
    
    audi.topSpeed = '300 km/h'                                                         // not working because the property is not writable  
    delete audi.topSpeed                                                               // not working because the property is not configurable 
    
    audi;                                                                              // -> {topSpeed:undefined}
    
    for(i in audi){ i }                                                                // 'color'  |  'counter' // the 'topSpeed' property is not sowh because it is not enumerable    
    
// defining getter and setter property ------------------------------------------------
    var audi = Object.create(mercedess, {topSpeed:{
        get: function(){ return 'fast enough' },                                       // runs this function when getting this property 
        set: function(val){ console.log('attempt to set ' + val) }                     // runs this function when setting this property   
    }})
    
    console.log( audi.topSpeed );                                                      // -> 'fast enough'
    audi.topSpeed = 21;                                                                // -> 'attempt to set 21'
    
// Object without prototype -----------------------------------------------------------
    var lada = Object.create(null);                                                    // -> {}                 // object without prototype
    lada.__proto__;                                                                    // -> undefined          // this object has no prototype

The Object.assign() method 

    var mercedess = {country:"Germany"};
    var suzuki = {country:"Japan"};
    var dacia = {country:"Romania"}

    obj;                                                                               // -> {prop1:"newVal", prop2:"valB"}
    
    Object.isExtensible(obj);                                                          // -> false (new properties cannot be added)   
    Object.isSealed(obj);                                                              // -> true

        prop1:{
            writable:false}
    });
    
    Object.getOwnPropertyDescriptors(obj)                                              // -> {prop1:{configurable:true, enumerable:true, value:21, writable:false}, prop2:{configurable:true, enumerable:true, value:22, writable:true}}

                                                                                       // gets the targeted property ('prop1') descriptors

The is() / getOwnPropertyNames / hasOwnProperty() / keys() / values() / entries() methods

DEMO 
// is() -------------------------------------------------------------------------------
    Object.is(2, '2');                                                                 // -> false 
    Object.is(2, 2)                                                                    // -> true
    
    Object.is({ a: 'someVal' }, { a: 'someVal' })                                      // -> false              // different objects 
    
    const obj1 = { a: 'someVal' }
    const obj2 = obj1
    Object.is(obj1, obj2)                                                              // -> true               // same objects (variables point to the same memory location)
    
    
// getOwnPropertyNames() / hasOwnProperty() -------------------------------------------
    var objA = {prop1:21};
    var objB = Object.create(objA, {prop2:{value:22}, prop3:{value:23}});
    
    Object.getOwnPropertyNames(objA);                                                  // -> ['prop1']          // the object's native enumerable and non-enumerable properties are returned    
    Object.getOwnPropertyNames(objB);                                                  // -> ['prop2', 'prop1']
    
    objB.hasOwnProperty('prop1');                                                      // -> false              // not native property 
    objB.hasOwnProperty('prop2');                                                      // -> true               // native property 
    
    
// keys() / values() / entries() ------------------------------------------------------
    var objA = {prop1:21};
    var objB = Object.create(objA, {prop2:{value:22}, prop3:{value:23}});
    
    var objA = {prop1:21};
    var objB = Object.create(objA, {
        prop2:{value:22},                                                              // 'prop2' property is not enumerable so it is not shown in these tools 
        prop3:{value:23, enumerable:true},
        prop4:{value:24, enumerable:true}
    });
    
    Object.keys(objB);                                                                 // -> ['prop3', 'prop4']
    Object.values(objB);                                                               // -> [23, 24]
    Object.entries(objB);                                                              // -> [['prop3', 23], ['prop4', 23]]
