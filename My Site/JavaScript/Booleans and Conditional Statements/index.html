Booleans Boolean()

- JavaScript Boolean value is eigther true or false (logical value)

- in JavaScript everything with a "real value" can be evaluated as true otherwise false Data Types [link]

DEMO 
    Boolean(5 == 5);                                    // -> true
    Boolean(9 < 8);                                     // -> false
    Boolean();                                          // -> false             // expression omitted returns false 
    
    new Boolean(3 > 1);                                 // -> {true}            // -! careful this is an object (bad practice to use it!)
    typeof new Boolean();                               // -> object            
    
// value evaluation ------------------------------------
    Boolean(100)                                        // -> true
    Boolean(3.14)                                       // -> true
    Boolean(42 + 54 - 1)                                // -> true
    Boolean("false")                                    // -> true              // -! this is a string careful!
    
    Boolean(0)                                          // -> false             // zero     
    Boolean("")                                         // -> false             // empty string 
    Boolean(undefined)                                  // -> false             
    Boolean(null)                                       // -> false             
    Boolean(NaN)                                        // -> false

Conditional Statements

- conditional statements are used to execute a piece of code based on different conditions

The if...else conditional statement 

                                                                                          1  1  3  3  1  0.10 
SYNTAX :    if (exp){                                                                  // if the expression can be evaluated as true the code inside the curly brackets {} is executed and no further "if" statement will fire 
                some code... 
            } else if (exp){                                                           // if the above if statement fails JS performs another check here (we can have as many else if statements as we like)   
                some code...               
            } else {                                                                   // if the if and all else if statements fail JS fires this as default code block   
                some code...                
            }
            
  // ----------------------------------------------------------------------------------
            if(exp) some code...;                                                      // valid if statement
DEMO 
    var x = Math.random();
    
    if(x < 0.3) {                                       // checks if the value is lower than 0.3    // -! if this condition is true the code is executed in the block and all further "if" statements are stopped   
        console.log("value lower than 0.3");
    } else if(x < 0.6) {                                // same goes here if the expression is true the code block is executed and all "if" statements are stopped 
        console.log("value is from 0.3 to 0.6");
    } else if(x < 0.9) {
        console.log("value is from 0.6 to 0.9");
    } else {                                            // default statement (if all fails this code block kicks in)
        console.log("value is bigger than 0.9");
    }
    
// if statement alone ----------------------------------
    if(0.5 < Math.random()) console.log("value is lower than 0.5");     // if the the expression is evaluated true the code after the "if" statement will be fired (if false nothing happens)

The switch conditional statement 

                                                                                          1  1  4  4  1  0.10 
SYNTAX:     switch (arg){
                case val|exp :                                                         // if arg equals val the this case block is executed
                    some code... 
                    break;                                                             // to stop further switch checks
                ...                                                                    // we can have as many case statements as we like    
                default :                                                              // default case 
                    some code...
            }
DEMO 
    var x = (Math.random() * 10).toFixed();             // this returns a random number between 0-9 as a string 
    
    switch(x) {
        case "0" :                                      // matches to one specific value 
            console.log("value 0");
            break;                                      // -! the "break" statement is required (without it the the code block is not stoped and the default value is yeilded always) 
        case "1" :                                      // matches to multiple values 
        case "2" : 
        case "3" :
        case "4" :
            console.log("value is between 1-4");
            break;
        default :                                       // if not match found the default code kicks in (or if the "break" statement is omitted above)
            console.log("value is greater than 4")
    }

The (conditional) ternary operator ?

- the ? operator (ternary = 3 values operator) is the only operator in JS which accepts 3 values 

                                                                                          1  1  4  3  1  0.10 
SYNTAX:     exp ? truthyExp : falsyExp                                                 // if the exp is evaluated true returns truthyExp, otherwise falsyExp    
    
  // ----------------------------------------------------------------------------------
            exp ? truthyExp:falsyExp                                                   // this is how we can chain multiple ternary expressions   
                ? truthyExp:falsyExp 
                ? truthyExp:falsyExp
DEMO 
    var x = Math.random();
    
    x < 0.5 ? console.log("value is lower than 0.5") : console.log("value is greater than 0.5");
        
// chaining --------------------------------------------
    x < 0.3 ? console.log("value is lower than 0.3") : x < 0.6          // chains through the conditions untill one of them is evaluated true or untill the end of the chain 
            ? console.log("value is between 0.3-0.6") : x < 0.9 
            ? console.log("value is between 0.6-0.9") : console.log("value is greater than 0.9");
