Function declaration VS definition

- in JavaScript declarations are hoisted but definitions are not 

// function declaration -----------
    add(1, 2);                        // -> 3   // function declaration is hoisted so we can call the function before defining it
    
    function add(a, b) {              // function declaration
      return a + b;
    }
    
    
// function definition -------------
    add(1, 2);                        // -> ReferenceError    // defined functions are not hoisted 
    
    const add = function(a, b) {      // function declaration
      return a + b;
    }

Closures

- The closure is a JavaScript specific feature that allows a function to access variables from its outer lexical scope even after that outer function has finished executing 

    function storage(list) {
      const storage = list;
      
      function getStorage() {     // returns the upper function's variable
        return storage;
      }
      
      return getStorage;
    }
    
    const getStoredValues = storage([1, 2, 3]);
    getStoredValues();            // -> [1, 2, 3]

Arrow Function 6 45 22 32 12 10 4.0.0

- arrow functions are nameles functions having nicer syntax than normal functions

- in arrow functions the arguments keywords is not available and there's no this binding (this means that the this keyword has the value that is bound before the arrow function)

    const fn = (arg1, arg2, ...) => {executedCode}  
    const fn = function(arg1, arg2, ...) {executedCode}            // same as above
    
    const fn = (arg1, arg2, ...) => returnedCode                   // if the curly brackets are omitted the code is returned after the arrow
    const fn = function(arg1, arg2, ...) {return returnedCode}     // same as above
    
    const fnId = arg1 => returnedCode                    // if we pass only 1 argument the parentheses are optional
    const fnID = function(arg1) {return returnedCode}    // same as above
    
    const fnId = () => returnedCode                                       // if no argument is passed the parentheses are required
    const fnID = function() {return returnedCode}                         // same as above
Example: 
    var arrFn1 = (a, b) => { 
        var x = "The result is " + a + b;  
        return x;
    }  
    arrFn1(5, 7);                                                                      // -> "The result is 57"        
        
// arrow function without curly brackets ----------------------------------------------
    var arrFn2 = (a, b) => "The result is " + a + b;                                   // the entire code after the arrow is returned 
    arrFn2(5, 7);                                                                      // -> "The result is 57"        
        
// arrow function without parentheses -------------------------------------------------
    var arrFn2 = a => "The result is " + a ;                                           // we can omit the parentheses if we pass only 1 argument 
    arrFn2(5);                                                                         // -> "The result is 5"
    
// arrow function without arguments ---------------------------------------------------
    var arrFn2 = () => "No passed value here";                                         // parentheses are required if no argument is passed 
    arrFn2();                                                                          // -> "No passed value here"
    
// no function declaration possible ---------------------------------------------------
    function myFn() {return "blaBla" }                                                 // function declaration 
    myFn() => "blaBal";                                                                // -! no function declaration possible with arrow functions (no hoisting)

Generator function* 6 39 26 26 13 10 4.0.0

- a generator is basicaly a play | pause interface in JavaScript

- calling next() triggers the fucntion and return the next yield value then the execution pauses 

    function* genFn(arg, ...) {             // when called returns a the generator without invoking the function's body
        nextArg = yield exp    // when the next() is called 2 important things happen: 1) returns the exp result / 2) the argument value passed in next(arg) is returned by yield
        
        yield* iterableObj                      // delegates the iteration cycle to the passed iterable object
        return exp                              // exp result returned as last iterable object value 
    }
    
    
    var generator = genFn();                                                       // creates the generator object    
    
    genObj.next(arg)                  // gets the next generater value (optionally we can pass a value to the currenty paused yeield) 
    
    genObj.throw(err:any)                 // throw the passed err on the current yield  
    genObj.return(val)                                                         // ends the generator object and returns the passed val as the generator's last value 
                                                             50  38  37  13  10  6.0.0 
    
    
// using for ...of loop to get generator members ---------------
    for (const member of generator) {
        member
    }
Example: 
    function* genNumber() {
        yield 1;
        yield* [2, 3];                    // delegates the iteration cycle
        yield 4;
        
        return 'end'
    };
    
  // get next member ----------------
    const numberGenerator = genNumber();
    
    numberGenerator.next();                 // -> { value: 1, done: false }
    numberGenerator.next();                 // -> { value: 2, done: false }
    numberGenerator.next();                 // -> { value: 3, done: false }
    numberGenerator.next();                 // -> { value: 4, done: false }
    numberGenerator.next();                 // -> { value: 'end', done: true }
    
    
  // return() -----------------------
    const numberGenerator2 = genNumber();
    
    numberGenerator2.next();                        // -> { value: 1, done: false }
    numberGenerator2.return('generator killed');       // -> { value: 'generator killed', done: false }
    
    
  // throw() -----------------------
    const numberGenerator3 = genNumber();
    
    numberGenerator3.next();                          // -> { value: 1, done: false }
    numberGenerator3.throw('some error');           // -> { value: 1, done: false }
    numberGenerator3.next();                        // -> { value: 1, done: false }
    
    
  // get member with for loop ------
    const numberGenerator4 = genNumber();
    
    for (const number of numberGenerator4) {
      number;                                     // -> { value: 1, done: false } / ...
    }

Passing argument to next() 

    function* counterGenerator(start = 0) {
      let count = start;
    
    while (true) {
      const increment = yield count;        // the passed value in next() is returned here by yield (so the increment variable holds the passed value)
    
        if (increment !== undefined) {
          count += increment;
        }
      }
    }
    
    
    const counter = counterGenerator(5);
    
    counter.next();                               // { value: 5, done: false }
    counter.next(2);                              // { value: 7, done: false }
    counter.next(3);                              // { value: 10, done: false }
    counter.next();                               // { value: 10, done: false } // No increment, so it stays the same
    counter.next(5);                              // { value: 15, done: false }
    counter.next(-2);                             // { value: 13, done: false } // Decrementing the count

The call() / apply() / bind() methods

DEMO 
    var audi = {
        mark:"audi", 
        type:"A4", 
        year:2010, 
        allData:function(a, b) {return this.mark + " " + this.type + " " + this.year + a}
    };
    
    var mercedess = {
        mark:"mercedess", type:
        "class C", year:
        2014
    };
    
// call() / apply() ------------------------------------------------------------------- 

    audi.allData.call(mercedess, " is a nice", " car");                                // -> "mercedess class C 2014 is a nice car"    // audi.allData method called on 'mercedess' object 
    audi.allData.apply(mercedess, [" is a nice, " car"]);                              // -> "mercedess class C 2014 is a nice car"    // does the same (uses different syntax)
    
// bind() -----------------------------------------------------------------------------
    var myObj = {p1:21};
    
    var globFn = function(){
        return this;
    }
    
    globFn();                                                                          // -> Window         // references the global object (as expected)
    
    var bindTest = window.globFn.bind(myObj);                                          // the 'window.globFn' 'this' keyword is bound to the 'myObj' object and returned a new function 
    
    bindTest();                                                                        // -> 'myObj'        // the 'this' keyword is bound to the 'myObj' object when the new function is called
