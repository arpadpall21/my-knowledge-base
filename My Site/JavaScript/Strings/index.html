Encoding

- JavaScrit use UTF-16 encoding which means characters are encoded by using 2 bytes (this allows (256*256) 65536 Code Units)

- when it comes to encode characters that are encoded higher than 65536 Unicode points (chinese characters or emojis) JavaScript surrogate pair (uses 4 bytes)

- the length property counts the number of 2bytes sequences in the string VS the for ...of loop gets Unicode characters at each iteration 

// 2 byte characters ------------------------------------------------------------------
    let myString = "st";             
        myString.length;                                                               // -> 2 
        for (let i of myString) {i};                                                   // -> 's' | 't'      // loop ran twice
    
// 4 byte characters ------------------------------------------------------------------
    let horseShoe = "🐴👟";         
        horseShoe.length;                                                              // -> 4              // -! length property counts how many 2byte block the string contains    
        for (let i of horseShoe) {i};                                                  // -> '🐴' | '👟'   // -! (loop ran twice) for of loop counts the number of unicode characters

The indexOf() / lastIndexOf() / search() / match() / matchAll() methods

DEMO 
    var x = "the text is the beast the fast";
    
// indexOf() --------------------------------------------------
    x.indexOf('the');                                          // -> 0              // references the first 'the' word's first letter index
    x.indexOf('the', 2);                                       // -> 12             // search begins from 2nd index   
    x.indexOf('you');                                          // -> -1             // no match 
    
// lastIndexOf() ----------------------------------------------
    x.lastIndexOf('the');                                      // -> 22             // references the last 'the' word's first letter index 
    x.lastIndexOf('the', 20);                                  // -> 12             // search starts from index 20 to the beginning of the string 
    x.lastIndexOf('The');                                      // -> -1             // no match 
    
// search() ---------------------------------------------------
    x.search('the');                                           // -> 0              // references the first 'the' word's first letter index
    x.search(/\b\we/);                                         // -> 4              // references the first match (regular expression offers more search power)   
    
// match() ----------------------------------------------------
    x.match(/the/);                                            // -> ['the']
    x.match(/the/g);                                           // -> ['the', 'the', 'the']  // regex offers more powerful search 
    
// matchAll() -------------------------------------------------
    var matches = x.matchAll();                                // -> RegExpStringIperator   // iterator object containing all matches 
    
    matches.next();                                            // -> {value:['the'], done:false}    // get data from the iterable object 
    matches.next();                                            // -> {value:['the'], done:false}
    matches.next();                                            // -> {value:['the'], done:false}
    matches.next();                                            // -> {value:undefined, done:true}

    txt.substring(-21,-11);                                                            // -> ''                       // -! negative indexes are treated as 0

The charAt() / charCodeAt() / codePointAt() / fromCharCode() / fromCodePoint() methods

DEMO 
    var txt = "Big 🌹";                                                                // string is composed of 12 bytes but 5 characters (the 🌹 character is encoded in 4 bytes)
    
// charAt() / charCodeAt() / codePointAt() --------------------------------------------
    txt.charAt(0);                                                                     // -> 'B'
    txt.charAt(2);                                                                     // -> 'g'
    txt.charAt(4);                                                                     // -> �       // because the 🌹 character is composed of 4bytes this method reads the only the first half which is meaningless   
    
    txt.charCodeAt(0);                                                                 // -> 66       // decimal Unicode Point of 'B'
    txt.charCodeAt(2);                                                                 // -> 103      // decimal Unicode Point of 'g'
    txt.charCodeAt(4);                                                                 // -> 55356    // -! this references the 1st half (1st 2 bytes out of 4 bytes) of the 🌹 character only, 
    txt.charCodeAt(5);                                                                 // -> 57145    // this is the 2nd half of the 🌹 character   
    
    txt.charPointAt(0);                                                                // -> 66       
    txt.charPointAt(2);                                                                // -> 103      
    txt.codePointAt(4);                                                                // -> 127801   // this is the correct decimal reference of the 🌹 character (we reference unicode points here)  
    
// fromCharCode() / fromCodePoint() ---------------------------------------------------
    String.fromCharCode(114);                                                          // -> 'r'      // 114 decimal unicode point references the 'r' letter 
    String.fromCharCode(0x2b, 0xb6, 0xb1b);                                            // -> '+¶ଛ'    // unicode point sequence returned in a string    
    String.fromCharCode(127801);                                                       // -> ''      // -! method cannot return characters encoded higher than 65535 in unicode 
    
    String.fromCodePoint(114);                                                         // -> 'r'     
    String.fromCodePoint(0x2b, 0xb6, 0xb1b);                                           // -> '+¶ଛ'
    String.fromCodePoint(127801);                                                      // -> '🌹'      // this method reads characters composed by 2 or 4 byte blocks (one or two UTF-16 unicode point)

    var str = 'I like 🐴es';

    str.concat(' ', 'a ', 'lot');                                                      // -> 'I like 🐴es a lot'        // strings are concatenated together and returned 
        
// split() ----------------------------------------------------------------------------
    str.split(' ');                                                                    // -> ['I', 'like', '🐴es']      
    str.split(' ', 2);                                                                 // -> ['I', 'like']               // split up to 2 pieces 
    str.split(/\s/);                                                                   // -> ['I', 'like', '🐴es']      // using regular expression 
    
// repeat() ---------------------------------------------------------------------------
    str.repeat(2);                                                                     // -> 'I like 🐴esI like 🐴es'   // repeated the number of times 
    str;                                                                               // -> 'I like 🐴es'              // -! none of the above methods modify the original string

    'arbre'.localeCompare('branche');                                                  // -> -1                          // 'branche' alphabetically comes after 'arbre' 
    'table'.localeCompare('chaise');                                                   // -> 1                           // 'chaise' alphabetically comes before 'table'
    'nape'.localeCompare('nape');                                                      // -> 0                           // 'nape' and 'nape' are equals 
    
// normalize() ------------------------------------------------------------------------
  // -! both letters 'canonically' (visually) are equals but encoded differently 
    var letterA = '\u00EA'                                                             // unicode point for the letter 'ê'
    var letterB = '\u0065\u0302';                                                      // the two unicode point compostion will result in the letter 'ê' 
    
    letterA === letterB;                                                               // -> false                       // canonically equal but encoded differently so they cannot be equal    
    letterA.normalize() === letterB.normalize();                                       // -> true                        // normalize makes sure that visually (canonically) equal letters will be compared as same letters
