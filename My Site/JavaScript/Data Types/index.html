Description and Demonstration:

- in JavaScript there are primitive & complex data types (primitive data is immutable VS complex data is multable) 

// primitive data types (immutable) --------------
    str                        // (string) represents a textual data
    nr                         // (number) numeric data (64bit double precision floating point)
    bigInt                     // 2022 (bitInteger) artibrary large integer  67  68  54  79  14  10.4 
    bool                       // (boolean) logical data with 2 possible values true or false
    undefined                                   // value without any real data
    null                                        // null value
    sym                        // 6 (symbol) unique object key   38  36  25  12  9  0.12 
    
// complex data types (mutable) --------------
    obj                        // anything which is not a primitie is considered compex data
    fn
    ...
    
    typeof val                 // return the type of val in a string (for null returns 'object' -> this is a bug in JavaScript)
    
// primitive data types are immutable ---------
    let a = 21;
    let b = a;                  // because primitives are immutable b has different value than a
    
    a;                          // -> 21
    b;                          // -> 21
    
    b = b + 1;                  // changing b won't change a because these are different values
    
    a;                          // -> 21
    b;                          // -> 22
    
// complex data types are mutable ---------
    let a = { x: 21 };
    let b = a;                  // because complex data types are mutable b has the same value as a (so b references the same value)
    
    a;                          // -> { x: 21 }
    b;                          // -> { x: 21 }
    
    b.x = b.x + 1;              // changing b also changes a because both references the same value
    
    a;                          // -> { x: 22 }
    b;                          // -> { x: 22 }

DEMO 
// primitive data ----------------
    typeof "myString"                    // -> "string"
    typeof 21                            // -> "number"
    typeof true                          // -> "boolean"
    typeof undefined                     // -> "undefined" 
    typeof null                          // -> "object"       // !! this is a bug in JavaScript it should be "null"
    typeof Symbol                        // -> "symbol"       // ES6
    typeof 21n                           // -> "bigint"       // ES11
    
// complex data ----------------------
    typeof function myFunction(){}       // -> "function"     // functions are complex datas but returned as "function"
    typeof {}                            // -> "object"       // any other data will be returned as "object"
    typeof []                            // -> "object"
