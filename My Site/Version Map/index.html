<script>
    var ancestorElement = document.querySelector(".frame");
    var parentElement = document.getElementsByClassName("parentBrowserEng")[0];
    var headerElement = document.getElementsByClassName("frame")[0].firstElementChild;
    var currentWidth;
    var currentHeight;
    var currentZoomLevel = {xZoom:currentWidth, yZoom:currentHeight}
    
    function resizeBody(event) {
        currentWidth = window.innerWidth - 35;
        currentHeight = window.innerHeight - 70;
    // sizing the map container    
        parentElement.style.width = currentWidth + "px";
        parentElement.style.height = currentHeight + "px";
    // sizing the parent ancestor element     
        ancestorElement.style.width = currentWidth + "px";
        ancestorElement.style.height = currentHeight + "px";
    // sizing the header element 
        headerElement.style.width = currentWidth - 13 + "px";
    }
// this asures that the header element scrolls allong with the parent element
    parentElement.onscroll = (e) => {
        headerElement.scrollLeft = e.target.scrollLeft;        
    }
    
// move the image along with the mouse ----------------------------------------
    var frameXpos, mouseXpos, frameYpos, mouseXpos;
    
// grab    
    function grab(event) {
        event.preventDefault();
        frameXpos = parentElement.scrollLeft;
        mouseXpos = event.offsetX;
        frameYpos = parentElement.scrollTop;
        mouseYpos = event.offsetY;
        parentElement.addEventListener("mousemove", mouseMove);
    }
// release     
    function release(event) {
        event.preventDefault();
        parentElement.removeEventListener("mousemove", mouseMove);
    }
// ensuring that the mousemouve is desabled when the mouse leaves the frame
    parentElement.addEventListener("mouseleave", () => parentElement.removeEventListener("mousemove", mouseMove));
      
// mouse move     
    function mouseMove(event) {
    // horizontal calculation    
        var currentXMovedDistance = event.offsetX - mouseXpos;           // this calculates the distance form the point where the mouse was clicked last time and the current position
        frameXpos = frameXpos - currentXMovedDistance;
        parentElement.scrollLeft = frameXpos ;
    // vertical calculation
        var currentYMovedDistance = event.offsetY - mouseYpos;
        frameYpos = frameYpos - currentYMovedDistance;
        parentElement.scrollTop = frameYpos;
    }
// prevent mouse scrolls the page when over the element 
    parentElement.addEventListener("wheel", (e) => {
        e.preventDefault();
        e.deltaY > 0 ? window.scrollBy({top:53, left:0, behavior:"auto"}) : window.scrollBy({top:-53, left:0, behavior:"auto"})
    });
</script>
