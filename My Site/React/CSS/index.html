Description and Demonstration :

- React natively supports SASS css preprocessor, the only thing needed is to install the sass library in the project (npm install sass)

Using css|sass|scss files

Global Styles 

    import 'filePath.css|.sass|.scss';             // global styles (styles are affecting all the application's Components, event the HTML document's elements))
  
    function Component() {
      return (
        <htmlTag id="id"> </htmlTag>               // id referenced in filePath.css|.sass|.scss
        <htmlTag className="class"> </htmlTag>     // class referenced in filePath.css|.sass|.scss (className used instead of class)
      )
    }

Module Styles 

// Component --------------------------------------------------------------
    import style from 'filePath.module.css|.sass|.scss';   // IDs and Classes

      
 - any other element stylings (like div, body, etc...) are still available in the Global Application 

    
 are available on the style object, they are used on a specific element (styles are scoped on the element)
    
    function Component() {
      return (
        <htmlTag id={style.id}> </htmlTag>                 // id style applies only to this component    
        <htmlTag className={style.class}> </htmlTag>       // class style applies only to this component    
      )
    }
    
    
// .css|.sass|.scss file --------------------------------------------------
    #id { ... }
    
    .class { ... }

    import ReactDOM from 'react-dom';
    
    import ChildComponent from './child1.js';
    import ChildComponent2 from './child2.js';
    
    class TestComponent extends React.Component{
        render(){
            let style_1 = {color:'red', backgroundColor:'purple', textDecoration:'underline'}
            return(
                <div>
                    <p className="myClass"> Element - 1 </p>                           // 'myClass' from './myStyle.module.css' is not available here or ensewhere    
                    <p id="myId"> Element - 2 </p>                                     // 'myId' from './myStyle.module.css' is not available here or elsewhere 
                    <ChildComponent />
                    <ChildComponent2 />
                </div>
            )
        }
    }
    
    ReactDOM.render(, document.getElementById('testElement'));
    
// Child Component file (child1.js) ---------------------------------------------------
// ------------------------------------------------------------------------------------
    import React from 'react';
    
    import myStyles from './myStyle.module.css';                                       // -! imported stylesheet ID and Class are only available in this module (other element styles are still available in the whole HTML document)   
    
    class ChildComponent extends React.Component{
        render(){
            return (
                <div>
                    <p className='{myStyles.myClass}> Child Element-1 </p>             // class from './myStyle.css'
                    <p id={myStyles.myId} > Child Element-2 </p>                       // id from './myStyle.css'
                </div>
            )
        }
    }
    
    export default ChildComponent;
