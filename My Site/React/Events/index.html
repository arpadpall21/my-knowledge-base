    const ReactComponent = () => {
      function handlerEvent(e:eventObj) { }           // the convention is to name the event object as e (ex: function handleClick(e) {})
      
      return <element htmlEvent={handlerEvent} />     // the convertion is to prefix event handler reference with handle (ex: handleClick)  
                                                      // never call handlerEvent (will be triggered when the event fires)
    }

Passing event handlers as props

- when passing event handlers as props the convetion is to prefix with on and name it some custom name 

  function Component({ onCustomEventName }) { }
DEMO 
// Parent Component -----------------------------------------------------------------
    import { useState } from 'react';
    import ChildComponent from './ChildComponent';
    
    const App = () => {
      function onCustomMessage() {
        alert('Hello World!');
      }
      
      return <ChildComponent onCustomMessage={onCustomMessage} />;
    }
    
    export default App;
    
    
    
// Child Component ------------------------------------------------------------------
    const ChildComponent = ({ onCustomMessage }) => {
      return <button onClick={onCustomMessage}> Print Custom Message </button>
    }
    
    export default ChildComponent;
