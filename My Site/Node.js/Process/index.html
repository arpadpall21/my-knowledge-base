The unhandledRejection / rejectionHandled events

DEMO 
// unhandledRejection / rejectionHandled --------------------------
    process.on('unhandledRejection', function(reason, promise){                        // listens to a Promise which does not have an error handler attached to it   
        reason;                                                                        // -> 'rejected'          // rejected promise's reason (data)  
        promise.catch(() => {});                                                       // -> <Promise>           // the promise object in question      // -! we attach an error handler to it here  
    });
    process.on('rejectionHandled', function(promise){                                  // this basically listens to the 'unhandledRejection' event handler to attach (fix) the missing promise error handler  
        promise;                                                                       // -> <Promise>           // the promise object in question  
    });
    
    new Promise(function(res, rej){ rej('rejected!') });                               // rejected promise does not have any error handler attached to it (method which handles the rejection)
