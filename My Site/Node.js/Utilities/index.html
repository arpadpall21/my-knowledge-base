The inherits() / isDeepStrictEqual() methods

DEMO 
// inherits() -------------------------------------------------------------------------
    const util = require('util');
        
    function Const1() { }                                                              // (ES6 class supported as well)
    function Const2() { }
        Const2.prototype.testProp = 'testVal';                                         // prototype property added to the constructor   
        
    util.inherits(Const1, Const2);                                                     // Const2 becomes superClass (superConstructor) of Const1
    
    var testObj = new Const1();
    testObj.testProp;                                                                  // -> 'testVal'          // object inherits the 'testProp' prototype property from the super constructor (Const2)    
    
// isDeepStrictEqual() ----------------------------------------------------------------
    var obj_A = {propA:415, probB:{subPropA:441, subPropB:[342, 543]}}
    var obj_B = {propA:415, probB:{subPropA:441, subPropB:[342, 543]}};
    
    util.isDeepStrictEqual('975', '975');                                              // -> true                           // values are strictly equal (same value same type )
    util.isDeepStrictEqual(obj_A, obj_B);                                              // -> true                           // objects are strictly equal

The callbackify() / promisify() methods

DEMO 
const util = require('util');
const fs = require('fs');
     
// callbackify() ----------------------------------------------------------------------
    async function testBol(bol) {                                                      // returns a resolved or rejected promise based on the passed argument (can be any function that returns a promise)   
        if( bol ) return Promise.resolve('boolean value is true');
        return Promise.reject('boolean value is false');                               // -! passing null as rejected value would be wrapped in an Error object and thrown in calbackified version   
    }
    
    var testBolCalbacked = util.callbackify(testBol);                                  // basically returns a function with an 'error-first style' callback as its last argument of the above async function    
    
    testBolCalbacked(true, (err, val) => {                                             // first argument is passed to 'testBol' function directly  // -! the asynchronous callback function argumets values are based on the returned promise by the 'testBol' async function    
        err;                                                                           // -> null                                                  // would return 'boolean value is false' if the 'testBol' async function returned a rejected promise    
        val;                                                                           // -> 'boolean value is true'
    });     
    
// promisify() ------------------------------------------------------------------------
    var readdirPromise = util.promisify(fs.readdir);                                   // makes a promise based API of the fs.readdir built-in method        
        
    readdirPromise('./')                                                               // -> Promise                                               // promisify function returns a promise  
        .then(res => console.log(res))                                                 // -> ['index.html', 'main.js', 'styles.css']               // consuming the promise   
        .catch(res => console.log(res));

    const util = require('util');
    
// resolved async --------------------------------------------------------------------
    async function resAsync(){                                                         // returns a resolved promise with the value of 'resolved value'
        return 'resolved value';    
    }
    
    var callbacked1 = util.callbackify(resAsync);                                      // returns an asynchronous function with an 'error-first style' callback as last argument  
        
    callbacked1(function(err, val){                                                    // the first argument of the callback 'err' is the rejected promise value 
        console.log( '----- resolved callback -----' );
        console.log( err );                                                            // -> null
        console.log( val );                                                            // -> 'resolved value'
    });
    
// rejected async --------------------------------------------------------------------
    async function rejAsync(){                                                         // returns a rejected promise with the value 'rejected value'
        throw 'rejected value'
    }
    
    var callbacked2 = util.callbackify(rejAsync);                                      // returns an asynchronous function with an 'error-first style' callback as last argument  
    
    callbacked2(function(err, val){                                                    // the second argument of the callback 'val' is the resolved promise value 
        console.log( '----- rejected callback -----' );
        console.log( err );                                                            // -> 'rejected value'
        console.log( val );                                                            // -> undefined
    });
    
// callbackified function arguments --------------------------------------------------
    async function checkCallback(a, b) {                                               // the final callbackified function passes its arguments here 
        if (a + b < 10 ) {
            return Promise.resolve('less than 10')
        } else {
            return Promise.reject('bigger than 10');
        }
    }
    
    var check10 = util.callbackify(checkCallback);                                     // the returned calbackified function will have 3 arguments (the last is always the callback)   
    
    check10(3, 5, function(err, val){                                                  // calling with less than 10 as sum of the passed first two arguments 
        console.log( '----- callbackified function arguments (less than 10) -----' );
        console.log( err );                                                            // -> null
        console.log( val );                                                            // -> 'less than 10'
    });
    
    check10(5, 5, function(err, val){                                                  // calling with more or equal than 10 as sum of the passed first two arguments 
        console.log( '----- callbackified function arguments (more than 10) -----' );
        console.log( err );                                                            // -> 'bigger than 10'
        console.log( val );                                                            // -> undefined
    });
    
// null as rejected promise value ----------------------------------------------------
    var rejCallback = function(){
        return Promise.reject(null);                                                   // null as rejected promise value 
    }
    
    var tester = util.callbackify(rejCallback);
    
    tester((err, val) => {
        console.log( '----- null as rejected promie value -----' );
        console.log( err );                                                            // -> null is wrapped in an Error object

    async function addAgrs(a, b) {

            throw new Error('passed argument is not a number');
        } else {
            return Promise.resolve(a + b);
        }
