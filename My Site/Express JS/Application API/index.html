- express 

    const express = require('express');
    const app = express();
    
// text() / json() / raw() / urlencoded() ---------------------------------------------
    app.use('/', 
        express.text(),                                                                // parses the response body to string (if the response header is 'Content-Type:text/plain')    
        express.json(),                                                                // parses the response body to JSON (if the response header is 'Content-Type:application/json')    
        express.raw(),                                                                 // parses the response body to Buffer (if the response header is 'Content-Type:application/octet-stream')     
        express.urlencoded({extended:false}),                                          // parses the response body to JavaScript data (if the response header is 'Content-Type:application/x-www-form-urlencoded')     
        function(req, res, next){
            req.body;                                                                  // -> the parsed response body populated here  
            res.end('OK!');
        }
    );
    
// static() ---------------------------------------------------------------------------
    app.use('/', 
        express.static('/srv/someSrv'),                                                // server automatially respond with static files from the '/srv/someSrv' directory   
    );
    
// Router() ---------------------------------------------------------------------------
    const router = express.Router();                                                   // router created 
    router.use('/sub', (req, res, next) => res.end('OK'));                             // router deals with the routed traffict  
    
    app.use('/', router);                                                              // a router passed as middleware   
    
    
    app.listen(1000, ()=> console.log('Expressjs is listening!'));

- app 

    const express = require('express');
    const app = express();
        
// locals / get() / set() / enable() / enabled() / disable() / disabled() -------------
    app.locals;                                                                        // -> {'x-powered-by':true, etag:'weak', ...}    // contains the app settings and the app's local variables    
    
    app.set('localVar', 21);                                                           // setting a local variable (we could set a setting too)
    app.get('localVar');                                                               // -> 21                                         // getting the value of the specified local variable or app setting    
    
    app.enable('strict routing');                                                      // enables the app's strict routing setting   
    app.enabled('strict routing');                                                     // -> true                                       // setting enabled   
    
    app.disable('strict routing');                                                     // disables the app's stricto routing setting    
    app.disabled('strict routing');                                                    // -> true                                       // setting disabled   
    
// mountpath / mount (event) / path() -------------------------------------------------
    const app2 = express();
    const app3 = express();
    
    app3.on('mount', function(parent){                                                 // tirggers when the app3 is mounted in another app as middleware    
        parent;                                                                        // -> app2 object (the direct parent of this app)
    });
    
    app.use('/subDir', app2);                                                          // app2 becomes the sub-app of app    
    app2.use('/childDir', app3);                                                       // app3 becomes the sub-app of app2    
    
    app.mountpath;                                                                     // -> '/'                                         // the direct parent path on where this app is mounted    
    app2.mountpath;                                                                    // -> '/subDir'                                   // the direct parent path on where this app is mounted    
    app3.mountpath;                                                                    // -> '/childDir'                                 // the direct parent path on where this app is mounted    
                        
    app.path();                                                                        // -> ''                                          // empty string because this app has no parent 
    app2.path();                                                                       // -> '/subDir'                                   // canonical path (full mountpath from the top-level parent)    
    app3.path();                                                                       // -> '/subDir/childDir'                          // canonical path (full mountpath from the top-level parent)    
    
// param() ----------------------------------------------------------------------------
    app.use('/:user/time:hour.:minute', (req, res, next) => res.send('OK'));           // the requested URL is 'localhost:1000/pall/time09:27'    
    
    app.param(['user', 'hour', 'minute'], function(req, res, next, id){                // callback triggers 3 times    
        id;                                                                            // -> 'pall'  |  '09'  |  '27'   
        next();                                                                        // -! callback acts as middleware when triggered (next() calls the nest param or middleware in the stack)    
    })
    
// get() / post() / put() / delete() / all() ------------------------------------------
    app.all('/', (req, res, next) => res.send('OK') );                                 // triggered when any request type requests exactly the root '/'   
                                                                                         // ex: '/' 
    app.get('/user', (req, res, next) => res.send('GET OK') );                         // triggers when the exact '/use' path is GET requested

    const express = require('express');
    const app = express();
    const app2 = express();
    const app3 = express();
    
// mount (event) / mountpath ----------------------------------------------------------
    app.on('mount', function(parent){                                                  // never triggers this app is not mounted on another app  
    })
    
    app2.on('mount', function(parent){                                                 // triggers when app2 is mounted on a parent app  
        console.log( app2.mountpath );                                                 // -> '/sub'                    // the path on which this app is mounted (the parent app path)
    })
    
    app3.on('mount', function(parent){                                                 // triggers when app2 is mounted on a parent app  
        console.log( app3.mountpath );                                                 // -> 'nest/moreNest'           // the path on which this app is mounted (the parent app path)
    })
    
    app.use('/sub', app2);                                                             // app2 is mounted on app 
    app2.use('/nest/moreNest', app3);                                                  // app3 is mounted on app2
    
// path() -----------------------------------------------------------------------------
    console.log( app.path() );                                                         // -> ''                        // top-level app doesn't return anything 
    console.log( app2.path() );                                                        // -> '/sub'                    // the canonical path on which this app is mounted   
    console.log( app3.path() );                                                        // -> '/sub/nest/moreNest'      // the canonical path on which this app is mounted   
    
    app.listen(1000, ()=> console.log('Expressjs is listening!'));
