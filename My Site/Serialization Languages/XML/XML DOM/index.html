namespace aware properties in XML
DOM1 (no namespace awareness) DOM2 (namespace awareness) Description
XMLDocument.createAttribute("attrKey") XMLDocument.createAttributeNS("nameSpURI", attrKey") creates an attribute node
XMLDocument.createElement("tagName") XMLDocument.createElementNS("nameSpURI", "tagName") creates an element
Node.getAttributeNode("attrKey") Node.getAttributeNodeNS("nameSpURI", "attrKey") returns the attribute node of the targeted element
Node.getAttribute("attrKey") Node.getAttributeNS("nameSpURI", "attrKey") returns the attribute value of the targeted attribute
XMLDocument.getElementsByTagName("tagName") XMLDocument.getElementsByTagNameNS("nameSpURI", "tagName") returns an HTMLCollection of the targeted elements
NamedNodeMap.getNamedItem("attrKey") NamedNodeMap.getNamedItemNS("nameSpURI", "attrKey") returns the targeted attribute node form a NamedNodeMap (attribute collection)
Node.hasAttribute("attrKey") Node.hasAttributeNS("nameSpURI", "attrKey") returns a boolean value indicating if the element has the targeted attribute
Node.removeAttribute("attrKey") Node.removeAttributeNS("nameSpURI", "attrKey") removes the targeted attribute
NamedNodeMap.removeNamedItem("attrKey") NamedNodeMap.removeNamedItemNS("nameSpURI", "attrKey") removes the targeted attribute from a NamedNodeMap (attribute collection)
Node.setAttribute("attrKey", value) Node.setAttributeNS("nameSpURI", "attrKey", value) sets an attribute to the element
Node.setAttributeNode(attrNode) Node.setAttributeNodeNS("attrNode) sets an attribute node to the element
NamedNodeMap.setNamedItem(attrNode) NamedNodeMap.setNamedItemNS(attrNode) sets an attribute node to the NamedNodeMap (attribute collection)

Description and Demonstration :

- XML DOM is a standard for accessing XML documents and modify XML elements in the XML DOM tree

- in XML DOM everything is a node / document(root) node / element node / attribute node / text node / comment node

- both XMLDocument and HTMLDocument constructors inherit form the Document constructor, therefore XML DOM navigation is almost the same

XML DOM namespace awareness

- because XML namespaces were created after the DOM1, DOM1 properties do not preserve namespaces, in DOM2 this problem is addressed by adding NS postfixes to properties (see table above) 

SYNTAX :    DOM1                                                   DOM2           
            XMLDocument.createAttribute("attrKey")                 XMLDocument.createAttributeNS("nameSpaceURI", attrKey")         // creates an attribute node
            XMLDocument.createElement("tagName")                   XMLDocument.createElementNS("nameSpaceURI", "tagName")          // creates an element                     
            Node.getAttributeNode("attrKey")                       Node.getAttributeNodeNS("nameSpaceURI", "attrKey")              // returns the attribute node of the targeted element
            Node.getAttribute("attrKey")                           Node.getAttributeNS("nameSpaceURI", "attrKey")                  // returns the attribute value of the targeted attribute
            XMLDocument.getElementsByTagName("tagName")            XMLDocument.getElementsByTagNameNS("nameSpaceURI", "tagName")   // returns an HTMLCollection of the targeted elements
            NamedNodeMap.getNamedItem("attrKey")                   NamedNodeMap.getNamedItemNS("nameSpaceURI", "attrKey")          // returns the targeted attribute node form a NamedNodeMap (attribute collection)    
            Node.hasAttribute("attrKey")                           Node.hasAttributeNS("nameSpaceURI", "attrKey")                  // returns a boolean value indicating if the element has the targeted attribute
            Node.removeAttribute("attrKey")                        Node.removeAttributeNS("nameSpaceURI", "attrKey")               // removes the targeted attribute
            NamedNodeMap.removeNamedItem("attrKey")                NamedNodeMap.removeNamedItemNS("nameSpaceURI", "attrKey")       // removes the targeted attribute from a NamedNodeMap (attribute collection) 
            Node.setAttribute("attrKey", value)                    Node.setAttributeNS("nameSpaceURI", "attrKey", value)           // sets an attribute to the element
            Node.setAttributeNode(attrNode)                        Node.setAttributeNodeNS("attrNode)                              // sets an attribute node to the element
            NamedNodeMap.setNamedItem(attrNode)                    NamedNodeMap.setNamedItemNS(attrNode)                           // sets an attribute node to the NamedNodeMap (attribute collection) 
    
    let myXmlCode =                                                             // XML source code
        "<?xml version='1.0' encoding='UTF-8'?>" +
        "<keyboard xmlns='default_ns' xmlns:ez='prefixed_ns'>" +                // a default and a prefixed namespace declared
            "<mark origin='us'>Logitec</mark>" +
            "<e:button_nr>101</e:button_nr>" +
            "<wired e:wireType='normal'>yes</wired>" +
        "</keyboard>";
    
    let myXmlDoc = new DOMParser().parseFromString(myXmlCode, "text/xml");      // string parsed to an XML document
    
// element methods --------------------------
    myXmlDoc.getElementsByTagNameNS("default_ns", "mark");                      // HTMLCollection returned 
    myXmlDoc.createElementNS("prefixed_ns", "ez:color");                        // -> <ez:color xmlns:ez="prefixed_ns"/>      // prefixed element created
    
// attribute methods ------------------------
    var myAttribute = myXmlDoc.createAttributeNS("prefixed_ns", "e:mode");      // -> e:mode=""                               // attribute node created
    myAttribute.nodeValue = "added_value";
    myXmlDoc.getElementsByTagNameNS("default_ns", "keyboard")[0].setAttributeNodeNS(myAttribute);                             // attribute node set no namespace URI here because the Attribute Node is created with namespace URI already    
    myXmlDoc.getElementsByTagNameNS("default_ns", "keyboard")[0].setAttributeNS("prefixed_ns", "e:mode", "added_value");      // this does the same as the code above (creates and sets a namespace aware attribute to an element)          
    
    myXmlDoc.getElementsByTagNameNS("default_ns", "wired")[0].getAttributeNodeNS("prefixed_ns", "wireType");    // -> e:wireType='normal'   // attribute node returned
    myXmlDoc.getElementsByTagNameNS("default_ns", "wired")[0].getAttributeNS("prefixed_ns", "wireType");        // -> "normal"              // attribute value returned
    
    myXmlDoc.getElementsByTagNameNS("default_ns", "wired")[0].hasAttributeNS("prefixed_ns", "wireType");        // -> ture
    myXmlDoc.getElementsByTagNameNS("default_ns", "wired")[0].removeAttributeNS("prefixed_ns", "wireType");     // removes the attribute (does not return it)
    
    let myNamedNodeMap = myXmlDoc.getElementsByTagNameNS("default_ns", "wired")[0].attributes;
    myNamedNodeMap.getNamedItemNS("prefixed_ns", "wireType")                                                    // -> e:wireType='normal'   // attribute node returned
    myNamedNodeMap.removeNamedItemNS("prefixed_ns", "wireType");                                                // -> e:wireType='normal'   // removes the targeted attribute node and returns it
    
    let attrNew = myXmlDoc.createAttributeNS("default_ns", "e:fabrication");                                    
    attrNew.nodeValue = "china";
    myNamedNodeMap.setNamedItemNS(attrNew);                                                                     // set the the created attribute Node to the namedNodeMap

    var xmlNs_Test = `<?xml version="1.0" encoding="UTF-8"?>
<person xmlns="defaultNS" xmlns:a="prefNs">
    <name>Gergo</name>
    <a:surName>Kicsi</a:surName>
    <age a:adult="true">+32</age>
</person>`;
    var xmlNS = new DOMParser().parseFromString(xmlNs_Test, "text/xml");
    
    var element = xmlNS.createElementNS("prefNs", "a:profession");              // creat a new element with the prefixed namespace
    var txtNode = xmlNS.createTextNode("programer");
    element.appendChild(txtNode);
    element.setAttributeNS("prefNs", "a:validity", "valid");                    // add a prefixed attribute
    xmlNS.getElementsByTagName("person")[0].appendChild(element);
    
// get namespace aware element and attribute    
    var ageAttr_ns = xmlNS.getElementsByTagName("age")[0].getAttributeNodeNS("prefNs", "adult");
    var nameElement_ns = xmlNS.getElementsByTagNameNS("defaultNS", "name")[0];
    
// print result
    document.getElementById("awareNS").textContent = new XMLSerializer().serializeToString(xmlNS);      // xml code printed
    
    document.getElementById("NS_print").innerHTML = 
        "Namespace aware attribute returned = " + ageAttr_ns + "<br>" +
        "Namespace aware element returned = " + nameElement_ns.outerHTML + "(check the returned result in console)";
console.warn("NS awareness");
    console.log(nameElement_ns.outerHTML);
