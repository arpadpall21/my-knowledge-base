DEMO

- this page's head section 

    <!DOCTYPE html>
    <head>
        <title> Basics and Elements </title>
// meta --------- 
        <meta charset="utf-8">                                                         // document encoding specified
        <meta name="author" content="Pall Arpad">                                      // author of the document
        <meta name="keywords" content="HTML basics, head elements">                    // keywords for search engines
        <meta http-equiv="refresh" content="120">                                      // -! this page refreshes every 2 minutes
    
// link ---------        
        <link rel="stylesheet" href="stylesPages.css">                                 // link to the stylesheet 
        <link rel="icon" href="favicon.png" sizes="any">                               // tab icon set ("sizes" has no effect)
    
// base --------- not in the headd section but it is tested [2018-12-18] ---        
        <base href="../../../.." target="_blank" >                                     // sets the root for all links in this page (all links will be opened in a new window)
        <link rel="stylesheet" href="stylesPages.css">                                 // target stylesheet (this is basically "../../../../stylesPages.css")
        
// style --------   
        <style>                                                                        // css in the document's head
            .head-style {
                background-color: blueviolet;
                color: white;
                border-bottom: solid 7px green;
            }
        </style>
        
// script -------        
        <script>                                                                       // script in the document's head 
            alert("alerted!");
        </script>
        
// noscript -----        
        <noscript>Browser does not support JavaScript!</noscript>                      // fallback if case the is disabled    
    </head>
    <body>
    
        <p class="head-style"> paragraph styled in the head </p>
        <script>                                                                       // script in the body
            document.getElementsByClassName("head-style")[0].textContent = "text Updated by Script!";
        </script>
        <noscript>Browser does not support JavaScript!</noscript>                      // fallback if case the is disabled    
    </body>
