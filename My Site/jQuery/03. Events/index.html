the on() / off() methods

- the on() method attaches one or multiple handlers for the selected element and children elements

- the off() method removes events added with the on() method or the set listener added by jQuery 

SYNTAX :    jQ_obj.on("event event ..", delegateToChildElement, {dataName: "message"}, function) 
            jQ_obj.on({event:function, event:function, ...})                                  // event map (cannot be combined with other arguments must be used separetely)
                - "event event .." : events to attach to the element
                - delegateToChildElement : children element to delegate the handler
                - {dataName: "dataMessage"} : message to pass to the event object (readed as event.data.dataName)
                - function : function to fire when event is trigered
                - {event:function, event:function, ...} : event map to pass to the element (advantage is that each event can be assigned to a different function, must be used separately)
                
            jQ_obj.off("event event ..", delegateToChildElement, function(eventObj))
            jQ_obj.off({event:function, event:function, ...})                                 // event map (cannot be combined with other arguments must be used separetely)
                - "event event .." : events to remove
                - delegateToChildElement : children element to delegate the handler (must match with the on() method)
                - function(eventObj) : function to remove (must be referenced function!)
                - {event:function, event:function, ...} : event map to remove events (must be referenced function!)

- demo on() method 

//  --------------- attach multiple event Handlers to the same function    
    <p> Hover on Me </p>
<script>
    $(document).ready(function(){
        $("p:contains(Hover on Me)").on("click contextmenu", function(){        // both events will triger the function
            $(this).css("color", "green");
        });    
    });
</script> 
//  --------------- attach event:function pairs (event map) (if you use event map you cannot combine with any other arguments!) 
    <p> Hover on Me </p>
<script>
    $(document).ready(function(){                                               // each event has its own functions
        $("p:contains(Hover on Me)").on({                  
            mouseenter:function(){                                              // element will change color when mouse enters 
                $(this).css('color', "yellow");
            },
            mouseleave:function(){                                              // element will change color when mouse leaves 
                $(this).css("color", "black");
            }
        })                                
    });
</script>    
//  --------------- passing message in the "event" object
    <p id="add_event_data"> Hover on Me </p>                                
<script>
    $(document).ready(function(){                                               // when mouse enters text will change to "Hover on Me (Ta Daa!)"                   
        $("#add_event_data").on("mouseenter", {addTxt: " (Ta Daa!)"},           // add a message to the event
            function(event){                                                    // event object must be passed in order to update it
                document.getElementById("add_event_data").innerHTML = 
                " Hover on Me " + event.data.addTxt;                            // reading the message
        })                            
    });
</script>
//  --------------- delegate event to the child element     
    <p id="delegate_to_child"> Hover on <span>Me</span> </p>                                
<script>
    $(document).ready(function(){                                               // only the "Me" part of the string will change!
        $("#delegate_to_child").on("mouseenter", "span", function(){            // function is delegated to the "span" child element
            $(this).css("color", "green");                                      
        })                    
    });
</script>

    <p id="demo"> Click // RightClick </p>
        <button> Remove Click Effect </button>
        <button> Remove RihgtClick Effect </button>

<script>
    function mouseClick(){                                                      // referenced functions must be used
        $(this).css("color", "white");  
    }
    $(document).ready(function(){
        $("#demo").on("click contextmenu", mouseClick);                         // adding a function to two events 
    // -----------          
        $("#demo + button").click(function(){                                   // remove "click" efect only 
            $("#demo").off("click", mouseClick);
        });
    // ----------- 
        $("#demo + button + button").click(function(){                          // remove "dblclick" effect only
            $("#demo").off("contextmenu", mouseClick);
        });
    // -----------
        $("#demo + button + button + button").click(function(){                 // remove both effects
            $("#demo").off("click contextmenu", mouseClick);    
        });
    });
</script>
//  --------------- remove effects by usint event map
    <p id="demo"> Click // RightClick </p>
        <button> Remove Click Effect </button>
        <button> Remove RihgtClick Effect </button>
        <button> Remove both Effects </button>
<script>
    function mouseClick(){                                                      // referenced functions must be used 
        $(this).css("color", "white");  
    }
    function mouseLeftClick(){
        $(this).css("color", "red");
    }
    $(document).ready(function(){
        $("#demo").on({click:mouseClick, contextmenu:mouseLeftClick});          // adding evetns with event map        
    // -----------     
        $("#demo + button").click(function(){
            $("#demo").off({click:mouseClick});                                 // remove "click" effect and its function only 
        })
    // -----------     
        $("#demo + button + button").click(function(){
            $("#demo").off({contextmenu:mouseLeftClick});                       // remove "contextmenu" effect and its function only 
        })
    // -----------     
        $("#demo + button + button + button").click(function(){
            $("#demo").off({click:mouseClick, contextmenu:mouseLeftClick});     // remove both effects                       
        })
    });
</script>

the one() method

- the one() method adds one or more events to the element, but tirggers only once per element

SYNTAX :        jQ_obj.one("event event ..", {dataName: "message"}, function) 
                    - "event event .." : events to attach to the element
                    - {dataName: "message"} : message to pass to the event object (readed as event.data.dataName)
                    - function : function to fire when event is trigered
    
    <p id="demo"> Click to Increase font size </p>
<script>
    $(document).ready(function(){
        $("#demo").one("mousedown mouseup", function(){                 // function runs once per event only!
            $(this).css("fontSize", "+=1px");
        }); 
    });
</script>

the trigger() / triggerHandler() methods

- the trigger() method triggers jQuery event listener (can trigger only jQuery listeners!)

- the triggerHandler() method tirggers only the jQuery event (does NOT triggers the element's default event (like submit))

- the trigger() and triggerHandler() methods trigger only on the first element if more elements are selected 

SYNTAX :        jQ_obj.trigger(event/eventObj, arg/[arg1, arg2, ...] )
                jQ_obj.triggerHandler(event/eventObj, arg/[arg1, arg2, ...] )
                    - event/eventObj : event to trigger
                    - arg/[arg1, arg2, ...] : arguments to pass to the event Handler (if you pass only one the Array is not required)

- the trigger() method 

    <p> Clicked </p>
<script>
    $(document).ready(function(){
        $("p:contains(Clicked)").click(function(event, color, fontSize, textDecoration){            // set an event with arguments to pass
            $(this).css({"color":color,"fontSize":fontSize,"textDecoration":textDecoration});
        });
        $("p:contains(Clicked)").trigger("click", ["white", "1.4em", "underline"]);                 // trigger the event and passing argument values             
    });
</script>

- the triggerHandler() method 

    <input id="focus_test" type="text">         <!-- the default behaviour of this field is to get focus when clicked -->
<script>
    $(document).ready(function(){
        $("#focus_test").click(function(){
            $(this).css("backgroundColor", "yellow");
        });
        $("#focus_test").triggerHandler("click");           // triggers the click event of the element but only the Query event is fired!  
    });                                                     // the "input" element doesn't get focus only the jQuery event is executed
</script>
