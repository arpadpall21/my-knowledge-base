- noConflict() method

- the main jQuery object has 2 default identifiers, $ and jQuery

- the noConflict() method releases the main jQuery object default identifies ($ and jQuery) or sets a new one (if new id set the old once are overriden) 

SYNTAX :    $/jQuery.noConflict(release_all_jQuery_identifier_(bool))           // removes the $ or jQuery jQuery object identifier
            var new_jQuery_identifier = $.noConflict()                          // set new jQuery object id
                - release_all_jQuery_identifier_(bool) : if set true releases both ($ and jQuery) identifiers
    
// only 1 "$.noConflict()" declaration can be on a page!
    $.noConflict();                                         // this releases the "$" identifier only (jQuery still available)
    $.noConflict(ture);                                     // this releases both "$" and "jQuery" identifiers 
    var newIdentifier$ = $.noConflict(true);                // create custom identifier ("newIdentifier$" is the only available jQuery identifier from this point) 
    
// recommended practice ---------------------
    $.noConflict();
    jQuery(document).ready(function($){                     // "$" is passed as argument
        $("#demo").css("color", "red");                     // the "$" is available only inside the "ready()" function
    });

- data() / removeData() method

- the data() method gets or sets an arbitary data to the element (can get the HTML5 data attribute's value (get only))

- the removeData() method removes the targeted key (or all data if no argument set!) added by the data() method 

SYNTAX :    jQ_obj.data("key", "val")   
            jQ_obj.data({key:val, key:val, ...})

            jQ_obj.removeData("key")
            jQ_obj.removeData("[key, key, ...]")

- the data() method doesn't work in XML documents and in IE, cannot be added to <object> / <embed> / <applet> elements because of plugIns conflict 

    < id="demo" data-my-data="value_A"> myParagraph </p>
<script>
    jQuery(document).ready(function($){
    // set data
        $("#demo").data("myData_A", "value_1")              // sets a data to an internal jQuery object
        $("#demo").data({myData_B:"value_2", myData_C:"value_3"});      // setting multiple values
            var myObj = {myProp:"value"};
        $("#demo").data(myObj);                             // set ready JS object
    // delete data
        $("#demo").removeData("myData_C");                  // removes a data added by the "data()" method 
        $("#demo").removeData();                            // removes all data added by the "data()" method
    // get data
        $("#demo").data("myData");                          // returns "value_A" returns the vaulue of the "data-" attribute
        $("#demo").data("myData_A");                        // returns "value_1", data only retrievable by "data()" method 
        document.getElementById("demo").myData_A;           // returns "undefined" because the property is no directly set to the element
        $("#demo").data("myData_C");                        // returns "undefined" data removed with "removeData()" method               
    });
</script>

- index() method

- the index() method returns the index position of the jQuery object out of the passed argument "jQuery Selector", if no argument passed it returns the index position of the jQuery object relative to its siblings (if no match found -1 is returned)

SYNTAX :    jQ_obj.index(jQuery_selector)
    
    <div>
        <p class="demoLm"> egy </p>
        <p class="demoLm" id="this"> ketto </p>
        <p class="demoLm"> harom </p>
    </div>    
<script>
    jQuery(document).ready(function($){
        $("#this").index(".demoLm");                        // -> 1             // selected element is the 2nd out of the passed element collection ".demoLm"  
        $("#this").index("p");                              // -> 24            // selected element is the 25th out of all "p" elements of the page 
        $("p:contains(harom)").index();                     // -> 2             // selected element is the 3rd relative to its siblings (no argument)
    });
</script>

- $.param() method

- the $.param() method serializes an object into a string suitable for a URL query string or an AJAX request (works only with objects with key/value pair structure) 

SYNTAX :    $.param(obj, useShallow_serialize(bool))
                - useShallow_serialize(bool) : if it is set to "true" deep obects won't be serialized
    
    jQuery(document).ready(function($){
    // object (also nested obj) ----------                   
        $.param( {property1:"value_A, property2:value_B, property_3:{deep_1:"valueX"}} );
                                                            // -> "property_1=value_A&property_2=value_B&property_3%5Bdeep_1%5D=valueX"
    // jQuery object ---------------------
        $.param( $({property1:"value_A, property2:value_B, property_3:{deep_1:"valueX"}})[0] );      <-- call the first obj from the stack
                                                            // -> "property_1=value_A&property_2=value_B&property_3%5Bdeep_1%5D=valueX"
    // Array -----------------------------
        $.param( [1, false, "kicsi_kakas"] );               <-- array fails because it is not a key/value format 
                                                            // -> "undefined=&undefined=&undefined="
    // Array as value --------------------
        $.param( {prop_A:["string", true, 21]} );           <-- this works because every array value has a key 
                                                            // -> "prop_A%5B%5D=string&prop_A%5B%5D=true&prop_A%5B%5D=21"
    });

- normal VS traditional "shallow" serialize 

    jQuery(function($){
        var myObj = {nestObj1:{par1:1, par2:2}, nestObj2:{par1:1, par2:[1, 2]}}; 
        
    // serializes deep objects as well ---- 
        $.param(myObj);                                     // -> nestObj1%5Bpar1%5D=1&nestObj1%5Bpar2%5D=2&nestObj2%5Bpar1%5D=1&nestObj2%5Bpar2%5D%5B%5D=1&nestObj2%5Bpar2%5D%5B%5D=2     
    // "traditional" way does NOT serializes deep objects (shallow serialize)    
        $.param(myObj, true);                               // -> nestObj1=%5Bobject%20Object%5D&nestObj2=%5Bobject%20Object%5D
    });

- toArray() / makeArray() / $.isPlainObject() methods

- the toArray() method returns all elements from a jQuery object into a javaScript Array

- the makeArray() method converts all Array-like object into a real javaScript Array

- the $.isPlainObject() method checks if an object is a plain Object (created with {} or new Object) 

SYNTAX :    jQ_obj.toArray()
            $.makeArray(obj)
            $.isPlainObject(obj)
    
    jQuery(function($){
    // toArray -------------
        $("p").toArray();                                   // -> [p, p, ..]        // returns all paragraph elements of this page in a JS Array
    // makeArray -----------
        $.makeArray($("p"));                                // -> [p, p, ..]        // returns all paragraph elements of this page in a JS Array
    // isPlainObject -------    
        $.isPlainObject({prop:"value"});                    // -> true
        $.isPlainObject(new Object());                      // -> true
        $.isPlainObject(document);                          // -> false             // not a plain object
        $.isPlainObject($('p')[0]);                         // -> false             // not a plain object
    });

- extend() method

- the extend() method merges two or more objects together into the first object (the next right obj properties override the previous ones, "undefined" property values are ignored)

SYNTAX :    $.extend(preserveVal(bool), targObj, srcObj, srcObj, ...)
                - preserveVal(bool) : preserves nested values when possible ("false" is not supported!) 
    
    jQuery(document).ready(function($){
        var myObj_A = {p1:"val_1", p2:{dP1:"deepVal_1", dP2:"deepVal_2"}, p3:true};
        var myObj_B = {p1:"val_A", p2:{dP1:"deepVal_A"}};
        var myObj_C = {p1:null, p2:undefined, p3:null, p4:undefined};
        
        $.extend(myObj_A, myObj_B);                         // -> {p1:"val_A", p2:{dP1:"deepVal_A"}, p3:true}                   // p2 value is overriden entierly  
        $.extend(true, myObj_A, myObj_B);                   // -> {p1:"val_A", p2:{dP1:"deepVal_A", dP2:"deepVal_2"}, p3:true}  // p2 value is preserved when possible
        $.extend(myObj_A, myObj_B, myObj_C);                // -> {p1:null, p2:{dP1:"deepVal_A"}, p3:null}                      // "undefined" value is ignored
    });
