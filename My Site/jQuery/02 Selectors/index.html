Selector Description
$("element:animated", context) selects elements what is animated with jQuery (only during the animation!)
$(":button", context) selects <input type="button"> element
$(":checkbox", context) selects <input type="checkbox"> element
$(":checked", context) selects checked <input> element
$("element:contains(string)", context) selects ellements which contain the specified string
$("CSS_selector", context).action()
$("element[attribute!=value]", context) 		CSS selector selects HTML elements
selects an element which doesn't has the specified attribute value
$(":disabled", context) selects disabled <input> element
$("element:empty", context) selects elements what are empty
$(":enabled", context) selects enabled <input> element
$("element:eq(index)", context) selects an element with a specific index number
$("element:even", context) selects each element with an even index number (0, 2, 4)
$(":file", context) selects <input type="file"> element
$("element:first", context) same as :first-child selector
$("element:first-child", context) selects the first child element of parent element
$("element:first-of-type", context) selects the first child element (specified type) of parent element
$("element:focus", context) selects elements that have focus
$("element:gt(index)", context) selects all emements higher than index (excluded)
$("element:has(jQuery_selector)", context) selects elements that contain the jQuery_selector match
$(":header", context) selects all the headers
$("element:hidden", context) selects elements display:none // form element type="hidden" // width and height set to 0
$(":image", context) selects <input type="image"> element
$(":input", context) selects <input> elements
$("element:lang(language)", context) selects the elements of the specified language attribute set
$("element:last", context) same as :last-child selector
$("element:last-child", context) selects the last child element of parent element
$("element:last-of-type", context) selects the last child element (specified type) of parent element
$("element:lt(index)", context) selects all emements lower than index (excluded)
$("element:not(jQuery_selector)", context) selects all emements except jQuery_selector
$("element:nth-child(n|even|odd|formula)", context) selects the index|even|odd|formula element of parent element
$("element:nth-last-child(n|even|odd|formula)", context) selects the index|even|odd|formula element of parent element counted from the end
$("element:nth-last-of-type(index|even|odd|formula)", context) selects the n|even|odd|formula element (specified type) of parent element counted from the end counted from the end
$("element:nth-of-type(index|even|odd|formula)", context) selects the n|even|odd|formula element (specified type) of parent element counted from the end
$("element:odd", context) selects each element with an odd index number (1, 3, 5)
$("element:only-child", context) selects the the only child element of the parent element
$("element:only-of-type", context) selects the the only child type element of the parent element
$("element:parent", context) selects the parent element
$(":password", context) selects <input type="password"> element
$(":radio", context) selects <input type="radio"> element
$(":reset", context) selects <input type="reset"> element
$(":root", context) selects the root <html> element
$(":selected", context) selects selected <input> element
$(":submit", context) selects <input type="submit"> element
$(":text", context) selects <input type="text"> element
$("element:visible", context) selects elements what are not display:none // form element type="hidden" // width and height set to 0

- :first / :last / :even / :odd / :eq() / :gt() / :lt() / :not()

SYNTAX :    $("element:first", context)             // same as :first-child selector
            $("element:last", context)              // same as :last-child selector
            $("element:even", context)              // selects each element with an even index number (0, 2, 4)  
            $("element:odd", context)               // selects each element with an odd index number (1, 3, 5)  
            
            $("element:eq(index)", context)         // selects an element with a specific index number
            $("element:gt(index)", context)         // selects all emements higher than index (excluded)
            $("element:lt(index)", context)>        // selects all emements lower than index (excluded)
            $("element:not(jQuery_selector)", context)  // selects all emements except jQuery_selector
    
    <ol>
        <li>Item_1</li>
        <li>Item_2</li>
        <li>Item_3</li>
        <li id="demo">Item_4</li>
        <li>Item_5 </li>
    </ol>
<script>
    $(document).ready(function(){
        $("ol li:first");                           // returns "<li>Item_1</li>" the first child element
        $("ol li:last");                            // returns "<li>Item_5</li>" the last child element
        $("ol li:even");                            // returns "<li>Item_1</li>, <li>Item_3</li>, <li>Item_5</li>" all even index elements (interestingly they are odd)
        $("ol li:odd");                             // returns "<li>Item_2</li>, <li>Item_4</li>" all odd index element      
        $("ol li:eq(1)");                           // returns "<li>Item_2</li>" the element with index "1"
        $("ol li:gt(2)");                           // returns "<li>Item_4</li>, <li>Item_5</li>" all element with higher index than "2" (excluded)
        $("ol li:lt(2)");                           // returns "<li>Item_1</li>, <li>Item_2</li>" all element with lower index than "2" (excluded)
        $("ol li:not('#demo')");                    // returns "<li>Item_1</li>, <li>Item_2</li>, <li>Item_3</li>, <li>Item_5</li>" all elements except the targeted one
    });
</script>

- :animated / :contains() / :has() / :hidden / :visible

SYNTAX :    $("element:animated", context)                  // selects elements what is animated with jQuery (only during the animation!)  
            $("element:contains(string)", context)          // selects ellements which contain the specified string
            $("element:has(jQuery_selector)", context)      // selects elements that contain the jQuery_selector match
            $("element:hidden", context)                    // selects elements display:none // form element type="hidden" // width and height set to 0
            $("element:visible", context)                   // selects elements what are not display:none // form element type="hidden" // width and height set to 0
    
// -------------------  jQuery UI added to this page in order to animate string type values

    <p id="anim_A" style="position:relative;"> jQuery <span>Animated_A</span> </p>
    <span id="anim_B"> jQuery Animated_B </span>
<script>
    $(document).ready(function(){
// animate elements
        $("#anim_A").mouseover(function(){
            $(this).animate({"color":"white"}, 1000);
        });
        
        $("#anim_B").mouseover(function(){
            $(this).slideUp(1000);
        });
    // :animate()
        $("p:animated");                            // returns the "#anim_A" paragraph while the animation is running
        $(":animated");                             // returns the "#anim_A, #anim_B" elements wihte the animation is runing (slideUp is a jQuery animation technicaly)
    // :contains()    
        $("p:contains(jQuery Animated_A)");         // returns the "#anim_A" paragraph element (if you don't specify the "p" element 
                                                        //the "html", "body", ... and all elements will be returned which contains the "jQuery Animated_A"!)
    // :has()    
        $("p:has('span')");                         // returns the "#anim_A" paragraph element because it "has" the "span" element 
    // :visible()
        $("p:visible");                             // returns all visible "p" elements on this page  
    // :hidden()
        $("*:hidden");                              // returns all hidden elements on this page "head, meta, script, ..." ("#anim_B" element is returned after the animation is completed!)  
    });
</script>

jQuery context selector

- the opitonal jQuery context selector (second selector argument) refers to a parent element in which the primary selection will be performed 

SYNTAX :    $(jQuery_selector, context)             // selection range is restricted under the set context (context / element) if context selector is set   
    
    <div id="demo_parent">
        <p> demoParagraph_A </p>
        <p> demoParagraph_B </p>
    </div>
<script>
    $(document).ready(function(){
        $("p:first", "demo_parent").css("color", "red);   // selection is reduced only to the "demo_parent" div children ("demoParagraph_A" selected) 
    });
</script>
