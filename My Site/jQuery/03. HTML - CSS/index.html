    $(document).ready(function(){

        var element = document.getElementById('demo_child_B');              // existing JS object

        var txt = document.createTextNode(" / child_E / "); 
        newElement.appendChild(txt);

            $($element).prependTo("#demo_parent");                              // does the same

            $(element).appendTo("#demo_parent");                                // does the same 
        
        $("#demo_parent").after("<span> / child_D / </span>");              // insert newly created jQuery object after 
            $("<span> / child_D / </span>").insertAfter("#demo_parent")         // does the same 
        
        $("#demo_parent").before(newElement);                               // insert newly created JS object before 
            $(newElement).insertBefore("#demo_parent");                         // does the same 
    }); 
</script>

- remove() / detach() / empty() / removeAttr()

- the remove() method removes and returns the element and its children from DOM, also removes all events and jQuery data attached to the element (element can be reused but its event and jQuery data will be lost)

- the empty() method removes and returns all the elements children but not the element itself

- the removeAttr() method removes and returns the element's attribute

            jQ_obj.empty()
            jQ_obj.removeAttr(attributeName)

SYNTAX :    jQ_obj.prop("prop/attrHTML")

- width() / innerWidth() / outerWidth() / height() / innerHeight() / outerHeight()

- the width() method gets or sets the element's content width in pixels (by default) (padding/border/margin excluded)

- the innerWidth() method gets or sets the element's content + padding width in pixels (by default) (border/margin excluded)

- the outerWidth() method gets or sets the element's content + padding + border [+ margin] width in pixels (by default)

- the height() method gets or sets the element's content height in pixels (by default) (padding/border/margin excluded)

- the innerHeight() method gets or sets the element's content + padding height in pixels (by default) (border/margin excluded)

- the outerHeight() method gets or sets the element's content + padding + border [+ margin] height in pixels (by default) 

SYNTAX :    jQ_obj.width(val) 
            jQ_obj.width(fn(index, val))

            jQ_obj.innerWidth(val) 
            jQ_obj.innerWidth(fn(index, val))
            
            jQ_obj.outerWidth(val, includeMargin(bol))
            jQ_obj.outerWidth(fn(index, val))
            
            jQ_obj.height(val) 
            jQ_obj.height(fn(index, val))

            jQ_obj.innerHeight(val) 
            jQ_obj.innerHeight(fn(index, val))
            
            jQ_obj.outerHeight(val, includeMargin(bol))
            jQ_obj.outerHeight(fn(index, val))

<script>
    $(document).ready(function(){
    // get elements widths
        $("#demo").width();                                                 // returns the width of the content of the element
        $("#demo").innerWidth();                                            // returns the width of the content + padding of the element
        $("#demo").outerWidth();                                            // returns the width of the content + padding + border of the element
        $("#demo").outerWidth(true);                                        // returns the width of the content + padding + border + margin of the element
    // set elements widths
        $("#demo").width(50);                                               // if number set jQuery assumes it is a px! (50px) (set the width of content)
        $("#demo").innerWidth("50px");                                      // this does the same declaration (set the width of content + padding)
        $("#demo").outerWidth("50%");                                       // string value specified in "%" (set the width of the content + padding + border)
        $("#demo").outerWidth("60em");                                      // (set the width of the content + padding + border + margin)
        
    // using function (the a = currentIndex always)
    // get elements heights
        $("#demo").height(function(a, b){                                   /* b = the height of the content of the element */ });             
        $("#demo").innerHeight(function(a, b){                              /* b = the height of the content + padding of the element */  });
        $("#demo").outerHeight(function(a, b){                              /* b = the height of the content + padding + border of the element */  });
        $("#demo").outerHeight(function(a, b){                              /* b = the height of the content + padding + border + margin of the element */  }, true);
    // set elements widths
        $("#demo").height(function(a, b){return b = 40});                   // if number set jQuery assumes it is a px! (40px) (set the height of content)
        $("#demo").innerHeight(function(a, b){return b = "40px"});          // this does the same declaration (set the height of content + padding)
        $("#demo").outerHeight(function(a, b){return b = "10%"});           // string value (works only if parent height set because the "%" is a relative value!)
        $("#demo").outerHeight(function(a, b){return b = "5em"}, true);     // (set the height of the content + padding + border + margin)
    });
</script>
