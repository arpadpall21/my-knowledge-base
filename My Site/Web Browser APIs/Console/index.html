Methods Description
console.log(obj|message) - prints the passed message in the console
console.info(obj|message) - prints the passed informational message in the console (same as console.log())
console.assert(exp|message) - prints the error in the console if the passed expression is evaluated "false" (otherwise does nothing)
console.warn(obj|message) - prints the passed message as a warning in console
console.error(obj|message) - prints the passed message as an error in console
console.clear() - clears the console and prints the message "console was cleared"
|
console.trace(label:str) - outputs the stack trace to the console (prints the call path)
|
console.time(label:str) - start the console execution timer counter
console.timeLog(label:str) - log (prints) the current console execution timer value previously started by the time() method 71 62 60 79 13
console.timeEnd(label:str) - ends the console execution timer counter an prints the result in the console
|
console.count(label:str) - prints a counter in the console the number of times it is called
console.countReset(label:str) - resets the console.count() counter (with the corresponding label if any) 68 62 55 79 13
|
console.group(label:str) - creates a message group in the console
console.groupCollapsed(label:str) - creates a collapsed message group in the console (same as group() but collapsed)
console.groupEnd(label:str) - ends the message group in the console
- group that started by the group() or groupCollapsed() method
|
console.dir(obj) - displays the passed object in "Object View" in the console window
console.dirxml(obj) - displays the passed object in "Interactive Markup View" if possible, otherwise in object view (does the same as log [TESTED 2019-02-09])
console.table(obj|arr, colName:arr

colName:str - only print the specified table column

...

) 		- prints a table in the console which visually represents the passed object
- the optional 2nd argument (colOnly:arr) can restrict which columns to print

Description and Demonstration :

- the console object provides access to the browser's debugging console

DEMO 
    var car_1 = {mark:"audi", type:"A4", releaseDate:2010, break:true};
    var car_2 = {mark:"mercedess", type:"C", releaseDate:2014, break:false};
        
// log() / info() / assert() / warn() / error() / clear() ---------------------- 
    console.log("Ta Daa");                      // -> "Ta Daa"          // printed in console window
    console.info(car_2);                        // object printed in console window
    console.assert(x);                          // -> "ReferenceError: x is not defined"    // printed in console window (because the passed value is converted to "false")
    console.assert(car_2);                      // does nothing because the value is converted to "true"
    console.warn("myWarning");                  // -> "myWarning"        // printed in console window as warning message
    console.error("myError");                   // -> "myError"          // printed in console window as error message
    console.clear()                             // all previous messages are printed from the console window and "console was cleared message" printed 
    
// trace() ---------------------------------------------------------------------
    childFunc();
    function childFunc(){
        console.trace();                        // -> "childFunc   @ index.html:689"        // printed in console window (trace and prints the call path through the call stack)   
    }                                           // -> "(anonymous) @ index.html:686"
    
// time() / timeLog() / timeEnd() ----------------------------------------------
    console.time("Loop Test");                  // timer starts from here 
    for (let i = 0; i < 1e6; i++) {
        if (i == 5e5) console.timeLog("Loop Test");  // returns the timer value at half loop (Chrome / Mozialla only [TESTED 2019-02-09])
    }
    console.timeEnd("Loop Test");               // timer finished and execution time returned
    
// count() / countReset() -------------------------------------------------------
    for (let i = 0; i < 6; i++) {
         console.count("myCount");              // -> "myCount: 1" / "myCount: 2" / "myCount: 3" / "myCount: 1" / "myCount: 2" / "myCount: 3"   // printed in the console window 
        if(i == 2) console.countReset("myCount");   // reset the counter here 
    }
    
// group() / groupCollapsed() / groupEnd() -------------------------------------
    console.group("g1");                        // creates a group in the console window 
        < some code >
    console.groupEnd("g1");                     // ends the group in the console window (if omitted all the code after the "g1" will be grouped)
    console.group("g2");                        // creates a collapsed group in the console window (same as "group()" method but this is automatically collapsed)
        < some code >
    console.groupEnd("g2");                     // ends the group in the console window (if omitted all the code after the "g1" will be grouped)
    
// dir() / dirxml() / table() --------------------------------------------------
    console.dir(car_1);                         // displays the object in "object view" in the console window 
    console.dirxml(document.body);              // displays the object in "interactive markup" in the console window (does the same as "log" [TESTED 2019-02-09])
    
    console.table(car_1);                       // displays the object content in a nice table in the console window 
    console.table([car_1, car_2], ["mark", "type"]);    // displays the objects' content in a nice table // -! but only their "mark" and "type" properties

time() / timeLog() / timeEnd() TEST (old fashion execution time measure TEST included)

- see debugger window to compare the two results

CODE : 
    <p id="measureTime"></p>
<script>
    function measureExecTime() {
    // - - - - first test
        console.time("main")                    // main time starts here (this will measure the whole execution time)
        console.time("firstTimer");             // returns the result in console window
        var start_A = new Date().getTime();     // traditional execution time measure technique 
            var container_A = 0;
            for (let i = 0; i < 1e6; i++) {
                container_A += i;
            }
        var end_A = new Date().getTime();
        console.timeEnd("firstTimer");
        
    // - - - - second test
        console.timeLog("main")                  // the main timer returns the result at this point (Chrome / Mozilla / Edge Only! [TESTED 2019-02-09])
        console.time("secondTimer");             // returns the result in console window
        var start_B = new Date().getTime();      // traditional execution time measure technique 
            var container_A = 0;
            for (let i = 0; i < 1e7; i++) {
                container_A += i;
            }
        var end_B = new Date().getTime();
        console.timeEnd("secondTimer");
        console.timeEnd("main");                 // main timer ends here
        
    // print                                     print the old fashion result on the page
        document.getElementById('measureTime').innerHTML = 
            "First Loop timer = " + (end_A - start_A) + "ms" + "<br>" + 
            "Second Loop timer = " + (end_B - start_B) + "ms";
    }
</script>

group() / groupCollapsed() / groupEnd() TEST

CODE : 
<button onmouseup="webGroup()"> create a message group in console </button>
<script>
    function webGroup() {
        console.group('group-1');                               // console message group starst from here
        console.log('blaBla');
        console.warn('blaBla');
        console.error('blaBla');
        console.groupEnd("group-1");                            // console message group ends here (if no "groupEnd()" specified group will extend to the end)

        console.log('blaBla');
        console.warn('blaBla');
        console.error('blaBla');
        console.groupEnd('group-2');                            // console message group ends here (if no "groupEnd()" specified group will extend to the end)
    }
</script>
