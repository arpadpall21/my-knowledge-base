    function dragStart_Text(ev) {                                // fires when element drag starts                   
        var itemDrop = '<p draggable="true" id="dragMe" ondragstart="dragStart_Text(event)"> Drag This Text </p>';
        ev.dataTransfer.setData("text", itemDrop);                                      // the ev is the drag event

    }

// JavaScript -------------------------------------------------------------------------
    <script>
        var item = document.getElementById('grabMe');
        var box2 = document.getElementById('box2');
        
        item.addEventListener('dragstart', function(ev){
            var id = ev.target.id;
            ev.dataTransfer.setData('text', id);                                       // data set in the dataTransfer object 
            
            let img1 = new Image()
            img1.src = "sprite.jpg";
            
            event.dataTransfer.setDragImage(img1, 200, 200);                           // image displayew whild dragging (grabbed 200px from the top and left)

        });
        
        box2.addEventListener('dragover', (ev)=>{ev.preventDefault()});
        box2.addEventListener('drop', function(ev){
            var id = ev.dataTransfer.getData('text');                                  // getting the set data 
            ev.currentTarget.appendChild( document.getElementById(id) );
            
            ev.dataTransfer.types;                                                     // -> ['text/plain']
            ev.dataTransfer.items[0];                                                  // -> {kind:'string', type:'text/plain'}
        });

        item.addEventListener('dragstart', function(ev){
            ev.dataTransfer.effectAllowed = 'move';                                    // this element is allowed to move only 
            ev.dataTransfer.setData('text', ev.target.id);
        });
        
        
        box2.addEventListener('dragenter', (ev)=>{ev.preventDefault()});
        box2.addEventListener('dragover', function(ev){
            ev.preventDefault();
            ev.dataTransfer.dropEffect = 'move';                                       // in this element we can only move dragged items   
        });
        
        box2.addEventListener('drop', function(ev){
            ev.currentTarget.appendChild( document.getElementById(ev.dataTransfer.getData('text')) );
        });
        
        
        box3.addEventListener('dragenter', (ev)=>{ev.preventDefault()});
        box3.addEventListener('dragover', function(ev){
            ev.preventDefault();
            ev.dataTransfer.dropEffect = 'copy';                                       // in this element we can only copy dragged items (so the dragged item cannot be dropped here)    
        });
        
        box3.addEventListener('drop', function(ev){
            ev.dataTransfer.dropEffect = 'copy';
            ev.currentTarget.appendChild( document.getElementById(ev.dataTransfer.getData('text')) );
        });
    </script>

    var fileDrop = function(event) {
        event.preventDefault();                                              // prevent default behaviour
        var collector = "";
        
        for(let i = 0; i < event.dataTransfer.files.length; i++) {           // looping through FileList ojbect's content and getting information
            collector += "File Name = " + event.dataTransfer.files[i].name + 
                " / Size = " + event.dataTransfer.files[i].size +

        }
        event.target.innerHTML = collector;
    }
</script>
