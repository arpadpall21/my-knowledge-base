new FileReader() - creates and returns a fileReader object which asynchronously can reads file or blob objects
|
fileReader.readAsText(file|blob) - asynchronously reads the content of file|blob as text
fileReader.readAsArrayBuffer(file|blob) - asynchronously reads the content of file|blob as arrayBuffer
fileReader.readAsDataURL(file|blob) - asynchronously reads the content of file|blob as dataUrl
fileReader.result - contains the content of a read operation (usually called on onload event)
|
fileReader.abort() - aborts a reading operation (only during reading operation)
fileReader.error - if error occures returns a DOMExeption
|
fileReader.readyState - returns the ready state (0 = EMPTY | 1 = LOADING | 2 = DONE)
fileReader.DONE - returns 2 indicating the end of the reading operation
fileReader.EMPTY - returns 0 indicating no data loaded
fileReader.LOADING - returns 1 indicating data reading
|

|

progress - repetedly fires during the reading operation
|

- BLOB 

<script>
    var myBlob = new Blob(['my ', 'test ', 'BLOB',], {type:'text/plain'});
    var myBlob_2 = myBlob.slice(3, 7);                                                 // creates a new blob of the sliced part of 'myBlob'    
    
    myBlob.size;                                                                       // -> 21
    myBlob.type;                                                                       // -> 'text/plain'
    
// text() -----------------------------------------------------------------------------
    myBlob.text()                                                                      // returns a Promise that reads the Blob as text  
        .then(data => data)                                                            // -> 'my text BLOB'
        .catch(err => err)
    
    myBlob_2.text()
        .then(data => data)                                                            // -> 'text'
        .catch(err => err)
    
// arrayBuffer() ----------------------------------------------------------------------
    myBlob.arrayBuffer()                                                               // returns a Promise that reads the Blob as ArrayBuffer 
        .then(data => data)                                                            // -> ArrayBuffer
        .catch(err => err)
    
// stream() ---------------------------------------------------------------------------
    myBlob.stream();                                                                   // -> ReadableStream      // returns a readable stream which when readed returns the blob's content    
</script>

DEMO

- FileReader() 

    <form id="test">
        <input type="file">
        <input type="submit">
    </form>
<script>
    var fileReaderObj = new FileReader();                                              // create the file-reader object 
    
    document.getElementById("test").elements[0].onchange = function() {
        var droppedElement = document.getElementById("test").elements[0].files[0]; 
        
// readAsArrayBuffer -----------------------------------------------------------------
        fileReaderObj.readAsArrayBuffer(droppedElement);                               // read the dropped file as ArrayBuffer nothing returned here
        fileReaderObj.onload = function(e) {                                           // we must use the "load" event to makes sure the file is successfuly loaded before start reading it   
            fileReaderObj.result;                                                      // -> ArrayBuffer (the "result" property holds the returned value of the file-reader) 
        }
        
// readAsBinaryString -----------------------------------------------------------------
        fileReaderObj.readAsBinaryString(droppedElement);                              // read the dropped file as BinaryString nothing returned here
        fileReaderObj.onload = function(e) {                                           // we must use the "load" event to makes sure the file is successfuly loaded before start reading it   
            fileReaderObj.result;                                                      // -> string (the "result" property holds the returned value of the file-reader) 
        }
        
// readAsDataURL ----------------------------------------------------------------------
        fileReaderObj.readAsDataURL(droppedElement);                                   // read the dropped file as readAsDataURL nothing returned here
        fileReaderObj.onload = function(e) {                                           // we must use the "load" event to makes sure the file is successfuly loaded before start reading it   
            fileReaderObj.result;                                                      // -> dataURL string (the "result" property holds the returned value of the file-reader) 
        }
        
// readAsText -------------------------------------------------------------------------
        fileReaderObj.readAsText(droppedElement);                                      // read the dropped file as readAsText nothing returned here
        fileReaderObj.onload = function(e) {                                           // we must use the "load" event to makes sure the file is successfuly loaded before start reading it   
            fileReaderObj.result;                                                      // -> string (the "result" property holds the returned value of the file-reader) 
        }
    }
    
// abort() / onabort ------------------------------------------------------------------
    fileReaderObj.abort();                                                             // aborts the reading operation (works only during readign operation)
    fileReaderObj.onabort = function(e) { };                                           // fires when the reading operation is aborted
        
// error / onerror --------------------------------------------------------------------
    fileReaderObj.onerror = function(e) {                                              // fires when an error occurs during the file reading operation 
        fileReaderObj.error                                                            // returns the DOMExeption object with the corresponding error
    }
        
// readyState / onprogress ------------------------------------------------------------
    fileReaderObj.onprogress = function(e) { }                                         // fires repeatedly during the reading operation 
    fileReaderObj.readyState;                                                          // returns the reading state (0 = no data loaded | 1 = reading | 2 = reading completed)
        
// onloadstart / onloadend ------------------------------------------------------------
    fileReaderObj.onloadstart = function(e) { }                                        // fires at the beginning of the reading operation (first event to fire)
    fileReaderObj.onloadend = function(e) { }                                          // fires at the end of the reading operation (even if the reading fails)
</script>

    <fieldset>
    <legend> file upload reading TEST </legend>
    <form id="testForm">
        <input type="file"><br>
        <input type="submit">
    </form>
    </fieldset>
    <p id="statusState">Status : <span style="font-weight:bold;"></span></p>
    <p id="workProgress">Progress : <span style="font-weight:bold;">0</span>%</p>
    <button id="abortBtn" style="background-color:indianred;" onmouseup="abortReading()"> Abort </button><br>
    <button onmouseup="read_as_ArrayBuffer()"> Read as ArrayBuffer </button>
    <button onmouseup="read_as_BinaryString()"> Read as BinaryString </button>
    <button onmouseup="read_as_DataURL()"> Read as DataURL </button>
    <button onmouseup="read_as_ReadAsText()"> Read as Text </button>
<script>
// get the dropped file
    var uploadedFile;
    var readMyFile = new FileReader();                                  // file reader object created
    document.forms['testForm'].elements[0].onchange = function(e) {
        uploadedFile = e.target.files[0];
        requestAnimationFrame(monitorUploadStatus);                     // call requestAnimationFrame in order to display the upload status
    }
    
// monitor and print the upload status
    var printElement = document.getElementById("statusState").firstElementChild;
    var monitorUploadStatus = function() {
        switch(readMyFile.readyState) {
            case 0 :
                printElement.style.color = "green";
                printElement.textContent = "File selected ready to read";
            break;
            case 1 : 
                printElement.style.color = "yellow";
                printElement.textContent = "LOADING...";
            break;
            case 2 :
                printElement.style.color = "green";
                printElement.textContent = "DONE";
        }
        requestAnimationFrame(monitorUploadStatus);
    }
    
// ready the selected file 
    var read_as_ArrayBuffer = function() {
        readMyFile.readAsArrayBuffer(uploadedFile);
        readMyFile.onload = function() { console.log(readMyFile.result)};      // the "onload" event waits till the file is fully bean read
    }
    var read_as_BinaryString = function() {
        readMyFile.readAsBinaryString(uploadedFile);
        readMyFile.onload = function() { console.log(readMyFile.result)};      // once the file is read the returned result is stored in the "result" property
    }
    var read_as_DataURL = function() {
        readMyFile.readAsDataURL(uploadedFile);
        readMyFile.onload = function() { console.log(readMyFile.result)};
    }
    var read_as_ReadAsText = function() {
        readMyFile.readAsText(uploadedFile);
        readMyFile.onload = function() { console.log(readMyFile.result)};
    }
    
// abort file reading 
    var abortReading = function() {
        readMyFile.abort();                                                     // abort file reading
    }
    readMyFile.onabort = function(e) {                                          // fires if file reading is aborted (not directly when you hit the "abort" button)
        var abortElm = document.createElement("span");
        abortElm.style.color = "red";
        abortElm.style.fontWeight = "bold";
        abortElm.textContent = " File Reading Aborted!";
        document.getElementById("abortBtn").parentElement.insertBefore(abortElm, document.getElementById("abortBtn").nextElementSibling);
    } 

// error handling (tested with a winrar compressed file)
    readMyFile.onerror = function(e) {
        var abortElm = document.createElement("span");
        abortElm.style.color = "red";
        abortElm.style.fontWeight = "bold";
        abortElm.textContent = " " + readMyFile.error.name;                     // error name returned
        document.getElementById("abortBtn").parentElement.insertBefore(abortElm, document.getElementById("abortBtn").nextElementSibling);
    }
    
// handling reading progress
    readMyFile.onprogress = function(e) { 
        var total = e.total;
        var currentWork = e.loaded;
        document.getElementById("workProgress").firstElementChild.textContent = Math.round(currentWork / total * 100);
    };
    
    readMyFile.onloadstart = function(e) { console.warn('"onloadstart" event triggered'); };   // first progres event to fire
    readMyFile.onloadend = function(e) { console.warn('"onloadend" event triggered'); };       // fires at the end of the progress (even if progress fails)
</script>
