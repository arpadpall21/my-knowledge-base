    var counter = document.getElementById('count').innerHTML;           // gets the element ;
    
    function startCounter() {
        var counterInit = setInterval(counterFunc, 1000);
        function counterFunc() {
            counter ++;
            document.getElementById('count').innerHTML = counter;       // sets the element;
        }
        document.querySelector('#stopCounter').onmousedown =            // pauses the counter (need to use add Event technique because the "counterInit" is unreachable from the global scope)     
            function() {clearInterval(counterInit);};   
    }
    
    // document.querySelector('#stopCounter + button').addEvenListener("mousedown", function(){ counter = 0;});
    document.getElementById('reinitCounter').addEventListener("mousedown", stopCounter);
    
    function stopCounter() {
        document.getElementById('count').innerHTML = 0;                 // reinitializes the counter but doesn't sotps it!! 
    }
</script>

set and delete Cookies

(Tested with EDGE on local site)

CODE : 
    <button onclick="generateCookie()"> Generate a cookies </button> 
    <button onclick="modifyCookie()"> Modify my Cookies </button> <span id="docCookie"></span>
<script>
    var plus30m = new Date(new Date().getTime() + 60 * 1000).toUTCString();
    var plus24h = new Date(new Date().getTime() + 24 * 60 * 60 * 1000).toUTCString();
    
    function generateCookie() {
        document.cookie = "Bircsy=Motor";                                               // this will be deleted as soon the browser closed
        document.cookie = "1MCookie=Value; expires=" + plus30m;                         // this will be deleted in 1 minute
        document.cookie = "24HCookie=Value; expires=" + plus24h;                        // this will be in 24 hours
    }
    function modifyCookie() {
        document.cookie = "Bircsy=";                                                    // delete the cookie value
        document.cookie = "24HCookie=Value; expires=Sun, 11 Mar 2018 07:21:12 GMT";     // delete the 24HCookie
    }
    
    document.getElementById('docCookie').innerHTML = "Document's Cookies : " + document.cookie;
</script>
