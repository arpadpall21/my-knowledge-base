FormData API

- the FormData API represents an HTML form's input/value pairs as an object key/value pairs 

SYNTAX:     formData                                                                   // represents an HTML form's input/values as an object (key/value pair structure) (iterable object)    
            
            
            new FormData(formElement)                                                  // (no argument) creates and returns an empty formData object 
                                                                                          (argument) maps the the passed form element and return its input/value data in a formData object    
            
            
            formData.append(key:str, val:str|blob, fileName:str

                
 - the file name when blob passed as value 

            
)                      // adds a new key/value pair to the formData, if the key exist appends (add) the passed value to it

                
 - formData entries do not hold multiple values, instead have multiple entries 

                
EX: 

                
 - stored data structure is {A:1, A:2} instead of {A:[1, 2]} 

            
    
            formData.set(key:str, val:str|blob, fileName:str

                
 - the file name when blob passed as value 

            
)                         // adds a new key/value pair to the formData, if the key exist its value is overwritten by the passed one    
            
            
            formData.get(key:str)                                                      // returns the first value associated with the key    
            formData.getAll(key:str)                                                   // returns all values in an Array associated with the key    
            
            
            formData.has(key:str)                                                      // returns true if key exist, otherwise false
            
            
            formData.delete(key:str)                                                   // delete the specified key with its value  
            
            
            formData.keys()                                                            // creates and returns an iterator object containing the formData keys    
            formData.values()                                                          // creates and returns an iterator object containing the formData values    
            formData.entries()                                                         // creates and returns an iterator object containing the formData entries (key/values)

- creating FormData object 

    <form id="testForm">
        <input>
        <input>
        <input type="submit">
    </form>
<script>
// create FormData object -------------------------------------------------------------
    var formElement = document.getElementById('testF');
    
    formElement.addEventListener('formdata', function(ev){
        ev.formData;                                                                   // -> FormData        // constructed when the form submits (holds the submitted input/value as key/values)    
    });
        
    new FormData();                                                                    // -> FormData        // creates a new empty formData object 
    
    new FormData(formElement);                                                         // -> FormData        // maps the form input/values and creates a formElement object based on it 
</script>

- FormData methods and properties 

<script>
    var formDt = new FormData();
    
// append() / set() -------------------------------------------------------------------
    formDt.append('key1', 1);                                                          // new key/value pair added    
    formDt.append('key1', 2);                                                          // value appended to the key 
    
    formDt.set('key2', 11);                                                            // new key/value pair added    
    formDt.set('key2', 12);                                                            // 'key2' value is replaced  
    
    for(i of formDt){ i };                                                             // -> ['key1', '1']  |  ['key1', '2']  |  ['key2', '12']     // -! 3 entries (same key different values)    
    
// get() / getAll() -------------------------------------------------------------------
    formDt.get('key1');                                                                // -> '1'                // the first value of 'key1' is returned    
    formDt.getAll('key1');                                                             // -> ['1', '2']         // all values of 'key1' are returned    
    
// has() ------------------------------------------------------------------------------
    formDt.has('key2');                                                                // -> true               // key exist 
    formDt.has('key3');                                                                // -> false              // there's no such key 
    
// keys() / values() / entries() ------------------------------------------------------
    var formKeys = formDt.keys();                                                      // returns an iterator object holding the formData's keys    
    var formValues = formDt.values();                                                  // returns an iterator object holding the formData's values    
    var formEntries = formDt.entries();                                                // returns an iterator object holding the formData's entires    
    
    formKeys.next();                                                                   // -> {done:false, value:'key1'}
    formKeys.next();                                                                   // -> {done:false, value:'key1'}
    
    formValues.next();                                                                 // -> {done:false, value:'1'}
    formValues.next();                                                                 // -> {done:false, value:'2'}
    
    formEntries.next();                                                                // -> {done:false, value:['key1', '1']}
    formEntries.next();                                                                // -> {done:false, value:['key1', '2']}
    
// delete() ---------------------------------------------------------------------------
    formDt.delete('key1');                                                             // deletes 'key1' with all its values   
    
    for(i of formDt){ i };                                                             // ->  ['key2', '12']
</script>

append() / set() / get() / getAll() / has() / delete() TEST

CODE : 
<script>
    function testMethods(){
        var formDt = new FormData();
        
// append() / set() -------------------------------------------------------------------
        formDt.append('key-1', 21);
        formDt.set('key-2', 22);
        
        for(i of formDt){
            console.log(i);                                                            // -> ['key-1', '21']  |  ['key-2', '22']
        }
        
        formDt.append('key-1', 31);                                                    // appends this new key/value pair to the form data (because the 'key-1' already exist)
        formDt.set('key-2', 32);                                                       // overwrites the 'key-2' keys value 
        
        for(i of formDt){
            console.log(i);                                                            // -> ['key-1', '21']  |  ['key-2', '22']  |  ['key-1', '32']
        }
        
// get() / getAll() -------------------------------------------------------------------
        console.log( formDt.get('key-1') );                                            // -> '21'              // the first value of 'key-1'
        console.log( formDt.getAll('key-1') );                                         // -> ['21', '31']      // all values of 'key-1'
        
        console.log( formDt.get('key-2') );                                            // -> '22'  
        console.log( formDt.getAll('key-2') );                                         // -> ['32']            // 'key-2' has only 1 value 
        
// has() ------------------------------------------------------------------------------
        console.log( formDt.has('key-2') );                                            // -> true              // formData has this key 
        console.log( formDt.has('key-3') );                                            // -> false             // formData does not have this key 
        
// delete() ---------------------------------------------------------------------------    
        formDt.delete('key-1');                                                        // deletes 'key-1' with its value 
        
        for(i of formDt){
            console.log(i);                                                            // -> ['key-2', '22']  
        }
    }
</script>

keys() / values() / entries() TEST

CODE : 
    <button onclick="testMethods_()"> Run TEST </button>
<script>
    function testMethods_(){
        var formDt = new FormData();
        
        formDt.append('key-1', 1);
        formDt.append('key-1', 2);
        
        formDt.append('key-2', 10);
        formDt.append('key-3', 21);
        
// keys() -----------------------------------------------------------------------------
        var formKeys = formDt.keys();                                                  // creates and returns an iterator object containing the formData keys 
        
        console.log( formKeys.next() );                                                // -> {done:false, value:'key-1'}
        console.log( formKeys.next() );                                                // -> {done:false, value:'key-1'}
        console.log( formKeys.next() );                                                // -> {done:false, value:'key-2'}
        console.log( formKeys.next() );                                                // -> {done:false, value:'key-3'}
        console.log( formKeys.next() );                                                // -> {done:true, value:undefined}
        
// values() ---------------------------------------------------------------------------
        var formValues = formDt.values();                                              // creates and returns an iterator object containing the formData values 
        
        console.log( formValues.next() );                                              // -> {done:false, value:'1'}
        console.log( formValues.next() );                                              // -> {done:false, value:'2'}
        console.log( formValues.next() );                                              // -> {done:false, value:'10'}
        console.log( formValues.next() );                                              // -> {done:false, value:'21'}
        console.log( formValues.next() );                                              // -> {done:true, value:undefined}
    
// entries() --------------------------------------------------------------------------
        var formEntries = formDt.entries();                                            // creates and returns an iterator object containing the formData entries  
        
        console.log( formEntries.next() );                                             // -> {done:false, value:['key-1', '1']}
        console.log( formEntries.next() );                                             // -> {done:false, value:['key-1', '2']}
        console.log( formEntries.next() );                                             // -> {done:false, value:['key-2', '10']}
        console.log( formEntries.next() );                                             // -> {done:false, value:['key-3', '21']}
        console.log( formEntries.next() );                                             // -> {done:true, value:undefined}
    }
</script>
