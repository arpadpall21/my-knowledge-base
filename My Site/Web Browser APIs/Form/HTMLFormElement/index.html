- JavaScript interface of the <form> element

DEMO 
    <form action="demo.php" method="get" name="demoForm" target="_blank" >               
        <input type="text"><br>                                
        <input type="submit">                               
    </form>
<script>
    var myForm = document.querySelector("[action='demo.php']");
    
// elements ---------------------------------------------------------------------------
    myForm.elements;                                                                   // -> HTMLFormControlsCollection      // iterable object contains the form's elemets (<input type="text"> / <input type="submit">)    
    
// submit() / requestSubmit() / reset() -----------------------------------------------
    myForm.reset();                                                                    // resets the form (input fields are reset to default)  
    myForm.submit();                                                                   // submits the form as it is          // -! without constraint validation and does not trigger the 'submit' event when the form is submitted  
    myForm.requestSubmit();                                                            // submits the form with constraint validation and triggers the 'submit' event when the form is submitted    
    
// checkValidity() / reportValidity() ----------  
    myForm.checkValidity()                                                             // -> false          // because the fileds are empty (if all form inputs are filled correctly this would return "true")
    myForm.reportValidity()                                                            // -> false          // does the same as the "checkValidity()" method but also prints a warning message to the user where the field has not a valid value   
    
// length / name / method / target / action /enctype / encoding / acceptCharset / autocomplete / noValidate ------ 
    myForm.length                                                                      // -> 2              // number of input elements 
    myForm.name                                                                        // -> "demoForm"     // "name" attribute value 
    myForm.method                                                                      // -> "get"          // "method" attribute value 
    myForm.target                                                                      // -> "_blank"       // "target" attribute value 
    myForm.action                                                                      // -> "file:///D:/Safe/CODE%20+/My%20Site/03%20Improuve/Learn%20Webdesign/01%20HTML/Form/Form%20Object%20(HTMLFormElement)/demo.php"     // URL where the form will be submitted  
    myForm.enctype                                                                     // -> "application/x-www-form-urlencoded"    // "enctype" attribute value 
    myForm.encoding                                                                    // -> "application/x-www-form-urlencoded"    // "enctype" attribute value (same as "enctype") 
    myForm.acceptCharset                                                               // -> ""             // "accept-charset" attribute value (not specified that's why not returned)
    myForm.autocomplete                                                                // -> "on"           // "autocomplete" attribute value 
    myForm.noValidate                                                                  // -> false          // "novalidate" attribute value 
</script>
    
// onsubmit / onreset (submit / reset events) -----------------------------------------
    myForm.onsubmit = function(ev){ }                                                  // tirggered when the form is submitted    
    
    myForm.onreset = function(ev){ }                                                   // tirggered when the form is reset   
    
// onformdata (formdata event) --------------------------------------------------------
    myForm.onreset = function(ev){                                                     // tirggered when the form constructs its formData object    // when submitted or when the formData is explicitly constructed by the FormData() constructor    
        ev.formData                                                                    // -> FormData       // the constructed formData object  
    }

submit() VS requestSubmit() TEST




CODE : 
    <form id="reqSubmit" name="testFrom" method="POST" target="_self" accept-charset="utf-8">
        <input name="txt" required><br>
        <input name="nr" type="number" required><br>
        <input type="submit">
    </form>
    <button id="submit_"> submit with <mark>submit()</mark> method </button><br>
    <button id="reqSubmit_"> submit with <mark>requestSubmit</mark> method </button>
<script>
    var formElement = document.getElementById('reqSubmit');
    
    formElement.onsubmit = function(ev){
        console.log( 'submit event fired!' );
    }
    
    document.getElementById('submit_').onmouseup = function(ev){
        formElement.submit();                                                          // submits the form withouth constraint validation (inputs values are not validated) and does not fire the 'submit' event!    
    }
    
    document.getElementById('reqSubmit_').onmouseup = function(ev){
        formElement.requestSubmit();                                                   // submits the form with constraint validation (inputs values are checked and must be valid) and does fire the 'submit' event when the form is submitted!    
    }
</script>
