The validity / validationMessage / willValidate / checkValidity() / reportValidity() / setCustomValidity() methods and properties 

SYNTAX:     inputElement.validity                                                      // returns the ValidityState object which holds the validation state of the input element
            inputElement.checkValidity()                                               // checks if input element have a valid value and returns a boolean (if "false" fires the "invalid" event)
            inputElement.reportValidity()                                              // checks if input element have a valid value, if it returns false it triggers the error message to the user and the "invalid" event   
            
            
            inputElement.validationMessage                                             // returns the correspondant error message if the input element validity is "false"
            inputElement.setCustomValidity("errorMessage")                             // sets a custom error message (error message to the user) to the input element and invalidates it! (empty string "" removes the error message and sets the validity of the element to "true")  
            
            
            inputElement.willValidate                                                  // returns a boolean indicating if the element is subject of form constraint validation
DEMO 
    <form id="demo">
        <input type="text" required>
        <input type="text" disabled>
        <input type="text" pattern="[a-z]" placeholder="non-bolck letters only">      // only non-block letters are allowed 
        <input type="submit">
    </form>
<script>
    var myForm = document.getElementById("demo");
    
// checkValidity() / reportValidity() / validationMessage --------------- 
    myForm.elements[0].checkValidity();                                    // -> false      // element is not valid 
    myForm.elements[0].reportValidity();                                   // -> false      // error message "Veuillez renseigner ce champ." is shown to the user 
    myForm.elements[0].validationMessage;                                  // -> "Veuillez renseigner ce champ."    // error message returned because this input element is not valid   
    
// validity / willValidate ---------------------------------------------- 
    myForm.elements[0].validity;                                           // -> ValidityState  // "ValidityState" object returned which holds the validation state of this input element   
    myForm.elements[0].willValidate;                                       // -> true           // because this element is subject of constraint validaton
    myForm.elements[1].willValidate;                                       // -> false          // because this element is NOT subject of constraint validaton
    
// setCustomValidity() -------------------------------------------------- 
    myForm.elements[2].oninput = function(e) {                             // at every new input we check the validity 
        myForm.elements[2].setCustomValidity("");                          <-- me must set the customValidity to an empty string otherwise it will stays "false" from the previous checks 
        
        if (myForm.elements[2].validity.valid) {                           // this is where we check the validity of the input filed 
            myForm.elements[2].setCustomValidity("");                      // if the validity is "true" no error message set ("" empty string) 
        } else {
            myForm.elements[2].setCustomValidity("Only non-bolck letters are allowed!");    // if validity is false we set the error message this also set the element invalid! (that's the reasons why we remove the error message at te beginning of every check)  
        }
    }
</script>

DEMO 
    <form id="testForm">
        <input type="text" value='defalut value'>
        <input type="submit">
    </form>
    
<script>
    let testForm = document.getElementById("testForm");
    
// selectionStart / selectionEnd / selectionDirection ---------------------- 
    testForm.elements[0].onselect = function(e) {
        this.selectionStart;                                        // retunrs the index position where the user start the selection 
        this.selectionEnd;                                          // returns the index position where the user ends the selection 
        this.selectionDirection;                                    // returns a string indicating the selection direction (ex: "forward")
    }
    
// setSelectionRange() -----------------------------------------------------
    testForm.elements[0].setSelectionRange(2, 7, "forward");        // we set a selection by setting the "selectionStart", "selectionEnd" and "selectionDirection" properties in one call 
    
// setRangeText() ---------------------------------------------------------- 
    testForm.elements[0].setRangeText("new text ");                 // text added at the beginning, and the original text is pushed to the right 
    testForm.elements[0].setRangeText("new text ", 2, 5);           // text between index 2 and 5 replaced to "new text "
    testForm.elements[0].setRangeText("new text ", 2, 5, "select"); // the newly inserted text is selected (element does not get focus!)
    testForm.elements[0].setRangeText("new text ", 2, 5, "start");  // caret moved at the beginning of the inserted text part (element does not get focus!)
    testForm.elements[0].setRangeText("new text ", 2, 5, "end");    // caret moved at the end of the inserted text part (element does not get focus!)
    testForm.elements[0].setRangeText("new text ", 2, 5, "preserve"); // the original selection (previously selected text by the user) will be preserved after this operation (default)  
</script>

ValidityState API

- checks the validity of the input element 

SYNTAX :    validityState.badInput                                                     // returns true if the browser unable to convert the input provided by the user (ex: string given to a number input type)  
            validityState.customError                                                  // returns true if the .setCustomValidity() method has a custom message set, otherwise returns false
            validityState.rangeOverflow                                                // returns true if input's value is higher than its max attribute otherwise false
            validityState.rangeUnderflow                                               // returns true if input's value is lower than its min attribute oterwise false
            validityState.patternMismatch                                              // returns true if input's value doesn't match its pattern attribue otherwise false
            validityState.stepMismatch                                                 // returns true if element's value is invalid per its step attribute otherwise false
            validityState.typeMismatch                                                 // returns ture if elemen's value does not match the expected input syntax otherwise false
            validityState.tooLong                                                      // returns true if element's value exeeds its maxlength attribute otherwise false
            validityState.tooShort                                                     // returns true if element's value is shorter than its minlength attribute otherwise false    
            validityState.valid                                                        // returns true if an element has a valid value otherwise "false" (all other properties of the "ValidityState" object are "false")  
            validityState.valueMissing                                                 // returns true if an element with a required attribute has no value otherwise false
DEMO 
    <form id="demo">
        <input type="number" value="someText">
        <input type="number" min="5" max="15" step="2" value="8">
        <input type="text">
        <input type="text" pattern="^[A-Z].*" value="alma">
        <input type="email" value="Addf">
        <input type="text" maxlength="5" value="more than six characters">
        <input type="submit">
    </form>
<script>
    let testForm = document.getElementById("demo");
// badInput / valid / valueMissing ---------------------------------------- 
    testForm.elements[0].validity.valid;                                   // -> false  // element is NOT valid because this field must have a value in order to submit
    testForm.elements[0].validity.badInput;                                // -> false  // correct input type provided (empty is still a correnct value) (if the user would provide a text this would be "true")    
    testForm.elements[0].validity.valueMissing;                            // -> true   // becuase the filed is required and there is no value provided
    
// rangeOverflow / rangeUnderflow / stepMismatch -------------------------- 
    testForm.elements[1].validity.rangeUnderflow;                          // -> false  // because the provided value is higher the the minimal allowed 
    testForm.elements[1].validity.rangeOverflow;                           // -> false  // because the provided value is lower thant the maximal allowed
    testForm.elements[1].validity.stepMismatch;                            // -> true   // because the count start at 5 and every second numbers are allowed only (5 7 9 ...)
            
// customError ------------------------------------------------------------ 
    testForm.elements[2].setCustomValidity("myCustom Invalid Message");
    testForm.elements[2].validity.customError;                             // -> true   // because we previously set a custom message to the input 
    
// patternMismatch -------------------------------------------------------- 
    testForm.elements[3].validity.patternMismatch;                         // -> true   // because the entered input does not match the required pattern 
    
// typeMismatch -----------------------------------------------------------  
    testForm.elements[4].validity.typeMismatch;                            // -> true   // because an email input does not accept any capital letters therefor the entere "type" does not match   
    
// tooLong ---------------------------------------------------------------- 
    testForm.elements[5].validity.tooLong;                                 // -> true   // because there is more characters than the "maxlength" attribute allows it
</script>
