Description and Demonstration :

Example :

CODE : 
    <p id="navigatorInfo"></p>
<script>
    document.querySelector('#navigatorInfo').innerHTML = 
        " Cookies are enabled = " + navigator.cookieEnabled + "<br>" +
        " Application Name = " + navigator.appName + "<br>" +
        " Application Code Name = " + navigator.appCodeName + "<br>" +
        " Product Name = " + navigator.product + "<br>" +
        " Browser Version = " + navigator.appVersion + "<br>" +
        " User Agent = " + navigator.userAgent + "<br>" +
        " Platform (operating system) = " + navigator.platform + "<br>" +
        " Browser's language = " + navigator.language + "<br>" +
        " Is the browser Online now? = " + navigator.onLine + "<br>" +
        " is Java Enabled? = " + navigator.javaEnabled();
</script>
